22      @Test
23      public void testStatisticsOnEmptyStream() {
24          Flowable<Integer> nums = Flowable.empty();
25          Statistics s = nums.compose(Transformers.<Integer>collectStats()).blockingLast();
26          assertEquals(0, s.count());
27          assertEquals(0, s.sum(), 0.0001);
28          assertTrue(Double.isNaN(s.mean()));
29          assertTrue(Double.isNaN(s.sd()));
30      }

32      @Test
33      public void testStatisticsOnSingleElement() {
34          Flowable<Integer> nums = Flowable.just(1);
35          Statistics s = nums.compose(Transformers.<Integer>collectStats()).blockingLast();
36          assertEquals(1, s.count());
37          assertEquals(1, s.sum(), 0.0001);
38          assertEquals(1.0, s.mean(), 0.00001);
39          assertEquals(0, s.sd(), 0.00001);

42      @Test
43      public void testStatisticsOnMultipleElements() {
44          Flowable<Integer> nums = Flowable.just(1, 4, 10, 20);
45          Statistics s = nums.compose(Transformers.<Integer>collectStats()).blockingLast();
46          assertEquals(4, s.count());
47          assertEquals(35.0, s.sum(), 0.0001);
48          assertEquals(8.75, s.mean(), 0.00001);
49          assertEquals(7.258615570478987, s.sd(), 0.00001);
50          assertEquals(1+16+100+400, s.sumSquares(), 0.0001);
51      }

54      public void testStatisticsPairOnEmptyStream() {

61      @Test
62      public void testStatisticsPairOnSingleElement() {
63          Flowable<Integer> nums = Flowable.just(1);
64          Pair<Integer, Statistics> s = nums.compose(Transformers.collectStats(Functions.<Integer>identity()))
65                  .blockingLast();
66          assertEquals(1, (int) s.a());
67          assertEquals(1, s.b().count());
68          assertEquals(1, s.b().sum(), 0.0001);
69          assertEquals(1.0, s.b().mean(), 0.00001);
70          assertEquals(0, s.b().sd(), 0.00001);
71      }
