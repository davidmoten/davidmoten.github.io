20      @Test
21      public void testPrintStackTrace() throws Exception {
22          PrintStream err = System.err;
23          try {
24              ByteArrayOutputStream bytes = new ByteArrayOutputStream();
25              System.setErr(new PrintStream(bytes));
26              Consumers.printStackTrace().accept(new RuntimeException());
27              String message = new String(bytes.toByteArray());
28              assertTrue(message.startsWith("java.lang.RuntimeException"));
29              assertTrue(message.contains("ConsumersTest.testPrintStackTrace"));
30          } finally {
31              System.setErr(err);
32          }
33      }
