10  import io.reactivex.Flowable;
11  import io.reactivex.exceptions.MissingBackpressureException;
12  import io.reactivex.functions.BiFunction;
13  
14  public class FlowableFetchPagesByRequestTest {
15      final static BiFunction<Long, Long, Flowable<Long>> FETCH = new BiFunction<Long, Long, Flowable<Long>>() {
16          @Override
17          public Flowable<Long> apply(Long start, Long request) {
18              return Flowable.rangeLong(start, request);
19          }
20      };
21  
22      final static BiFunction<Long, Long, Flowable<Long>> FETCH_LESS = new BiFunction<Long, Long, Flowable<Long>>() {
23          @Override
24          public Flowable<Long> apply(Long start, Long request) {
25              return Flowable.rangeLong(start, request - 1);
26          }
27      };
28  
29      final static BiFunction<Long, Long, Flowable<Long>> FETCH_NONE = new BiFunction<Long, Long, Flowable<Long>>() {
30          @Override
31          public Flowable<Long> apply(Long start, Long request) {
32              return Flowable.empty();
33          }
34      };
35  
36      final static BiFunction<Long, Long, Flowable<Long>> FETCH_MORE = new BiFunction<Long, Long, Flowable<Long>>() {
37          @Override
38          public Flowable<Long> apply(Long start, Long request) {
39              return Flowable.rangeLong(start, request + 1);
40          }
41  
42      };
43  
44      @Test
45      public void isUtilityClass() {
46          Asserts.assertIsUtilityClass(FlowableFetchPagesByRequest.class);
47      }
48  
49      @Test
50      public void testFetchByRequest() {
51          Flowables.fetchPagesByRequest(FETCH) //
52                  .test(0) //
53                  .assertNoValues() //
54                  .requestMore(1) //
55                  .assertValue(0L) //
56                  .requestMore(2) //
57                  .assertValues(0L, 1L, 2L) //
58                  .requestMore(3) //
59                  .assertValues(0L, 1L, 2L, 3L, 4L, 5L) //
60                  .assertNotTerminated();
61      }
62  
63      @Test
64      public void testFetchByRequestNonZeroStart() {
65          Flowables.fetchPagesByRequest(FETCH, 3) //
66                  .test(0) //
67                  .assertNoValues() //
68                  .requestMore(1) //
69                  .assertValue(3L) //
70                  .requestMore(2) //
71                  .assertValues(3L, 4L, 5L) //
72                  .requestMore(3) //
73                  .assertValues(3L, 4L, 5L, 6L, 7L, 8L) //
74                  .assertNotTerminated();
75      }
76  
77      @Test
78      public void testFetchByRequestError() {
79          final BiFunction<Long, Long, Flowable<Integer>> fetch = new BiFunction<Long, Long, Flowable<Integer>>() {
80              @Override
81              public Flowable<Integer> apply(Long start, Long n) {
82                  throw new ThrowingException();
83              }
84  
85          };
86          Flowables.fetchPagesByRequest(fetch) //
87                  .test(1) //
88                  .assertNoValues() //
89                  .assertError(ThrowingException.class);
90      }
91  
92      @Test
93      public void testFetchCompletesIfReturnsLessThanRequested() {
94          Flowables.fetchPagesByRequest(FETCH_LESS) //
95                  .test(100) //
96                  .assertValueCount(99) //
97                  .assertComplete();
98      }
99  
100     @Test
101     public void testFetchCompletesIfNoneReturned() {
102         Flowables.fetchPagesByRequest(FETCH_NONE) //
103                 .test() //
104                 .assertNoValues() //
105                 .assertComplete();
106     }
107 
108     @Test
109     @Ignore
110     public void testFetchEmitsMissingBackpressureExceptionIfReturnsMoreThanRequested() {
111         Flowables.fetchPagesByRequest(FETCH_MORE) //
112                 .test(100) //
113                 .assertValueCount(100) //
114                 .assertError(MissingBackpressureException.class);
115     }
