161     @Test
162     public void testUnsubscribe() {
163         AtomicBoolean unsubA = new AtomicBoolean(false);
164         AtomicBoolean unsubB = new AtomicBoolean(false);
165         Flowable<Integer> a = Flowable.just(1, 2, 5, 7, 6, 8).doOnCancel(Actions.setToTrue(unsubA));
166         Flowable<Integer> b = Flowable.just(3, 4, 5, 6, 7).doOnCancel(Actions.setToTrue(unsubB));
167         final List<Integer> list = new ArrayList<Integer>();
168         final AtomicBoolean terminal = new AtomicBoolean();
169         matchThem(a, b) //
170                 .doOnTerminate(Actions.setToTrue(terminal)) //
171                 .doOnNext(Consumers.println()).subscribe(new Subscriber<Integer>() {
172 
173                     private Subscription s;
174 
175                     @Override
176                     public void onSubscribe(Subscription s) {
177                         this.s = s;
178                         s.request(Long.MAX_VALUE);
179                     }
180 
181                     @Override
182                     public void onComplete() {
183                     }
184 
185                     @Override
186                     public void onError(Throwable e) {
187                         e.printStackTrace();
188                     }
189 
190                     @Override
191                     public void onNext(Integer t) {
192                         list.add(t);
193                         s.cancel();
194                     }
195                 });
196         assertFalse(terminal.get());
197         assertEquals(Arrays.asList(5), list);
198         assertTrue(unsubA.get());
199         assertTrue(unsubB.get());
200     }

313     private void testShifted(int n, boolean async) {
314         Flowable<Integer> a = Flowable.just(0).concatWith(Flowable.range(1, n));
315         if (async) {
316             a = a.subscribeOn(Schedulers.computation());
317         }
318         Flowable<Integer> b = Flowable.range(1, n);
319         assertTrue(Flowable.sequenceEqual(matchThem(a, b), Flowable.range(1, n)).blockingGet());
320     }
321 
322     private static Flowable<Integer> matchThem(Flowable<Integer> a, Flowable<Integer> b) {
323         return a.compose(FlowableTransformers.matchWith(b, Functions.identity(), Functions.identity(), COMBINER));
324     }
325 
326     private static void match(Flowable<Integer> a, Flowable<Integer> b, Integer... expected) {
327         match(a, b).assertValues(expected).assertComplete();
328     }
329 
330     private static TestSubscriber<Integer> match(Flowable<Integer> a, Flowable<Integer> b) {
331         return matchThem(a, b).test();
332     }

334     private static final BiFunction<Integer, Integer, Integer> COMBINER = new BiFunction<Integer, Integer, Integer>() {
