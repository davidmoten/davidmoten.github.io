161     @Test
162     public void testUnsubscribe() {
163         AtomicBoolean unsubA = new AtomicBoolean(false);
164         AtomicBoolean unsubB = new AtomicBoolean(false);
165         Flowable<Integer> a = Flowable.just(1, 2, 5, 7, 6, 8).doOnCancel(Actions.setToTrue(unsubA));
166         Flowable<Integer> b = Flowable.just(3, 4, 5, 6, 7).doOnCancel(Actions.setToTrue(unsubB));
167         final List<Integer> list = new ArrayList<Integer>();
168         final AtomicBoolean terminal = new AtomicBoolean();
169         matchThem(a, b) //
170                 .doOnTerminate(Actions.setToTrue(terminal)) //
171                 .doOnNext(Consumers.println()) //
172                 .subscribe(new Subscriber<Integer>() {
173 
174                     private Subscription s;
175 
176                     @Override
177                     public void onSubscribe(Subscription s) {
178                         this.s = s;
179                         s.request(Long.MAX_VALUE);
180                     }
181 
182                     @Override
183                     public void onComplete() {
184                     }
185 
186                     @Override
187                     public void onError(Throwable e) {
188                         e.printStackTrace();
189                     }
190 
191                     @Override
192                     public void onNext(Integer t) {
193                         list.add(t);
194                         s.cancel();
195                     }
196                 });
197         assertFalse(terminal.get());
198         assertEquals(Arrays.asList(5), list);
199         assertTrue(unsubA.get());
200         assertTrue(unsubB.get());
201     }

354     private void testShifted(int n, boolean async) {
355         Flowable<Integer> a = Flowable.just(0).concatWith(Flowable.range(1, n));
356         if (async) {
357             a = a.subscribeOn(Schedulers.computation());
358         }
359         Flowable<Integer> b = Flowable.range(1, n);
360         assertTrue(Flowable.sequenceEqual(matchThem(a, b), Flowable.range(1, n)).blockingGet());
361     }
362 
363     private static Flowable<Integer> matchThem(Flowable<Integer> a, Flowable<Integer> b) {
364         return a.compose(
365                 Transformers.matchWith(b, Functions.identity(), Functions.identity(), COMBINER));
366     }
367 
368     private static void match(Flowable<Integer> a, Flowable<Integer> b, Integer... expected) {
369         match(a, b).assertValues(expected).assertComplete();
370     }
371 
372     private static TestSubscriber<Integer> match(Flowable<Integer> a, Flowable<Integer> b) {
373         return matchThem(a, b).test();
374     }

376     private static final BiFunction<Integer, Integer, Integer> COMBINER = new BiFunction<Integer, Integer, Integer>() {
377         @Override
378         public Integer apply(Integer x, Integer y) {
