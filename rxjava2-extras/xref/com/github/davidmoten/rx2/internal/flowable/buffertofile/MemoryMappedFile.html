3   import java.io.File;
4   import java.io.RandomAccessFile;
5   import java.lang.reflect.InvocationTargetException;

9   import sun.misc.Unsafe;
10  import sun.nio.ch.FileChannelImpl;
11  
12  @SuppressWarnings("restriction")
13  public final class MemoryMappedFile {
14  
15      private static final Unsafe unsafe;
16      private static final Method mmap;
17      private static final Method unmmap;
18      private static final int BYTE_ARRAY_OFFSET;
19  
20      private final File file;
21      private final long size;
22      private long addr;
23  
24      static {
25          unsafe = UnsafeAccess.unsafe();
26          mmap = getMethod(FileChannelImpl.class, "map0", int.class, long.class, long.class);
27          unmmap = getMethod(FileChannelImpl.class, "unmap0", long.class, long.class);
28          BYTE_ARRAY_OFFSET = unsafe.arrayBaseOffset(byte[].class);
29      }
30  
31      public MemoryMappedFile(File file, long len) {
32          this.file = file;
33          this.size = roundTo4096(len);
34          mapAndSetOffset();
35      }
36  
37      // visible for testing
38      // Bundle reflection calls to get access to the given method
39      static Method getMethod(Class<?> cls, String name, Class<?>... params) {
40          Method m;
41          try {
42              m = cls.getDeclaredMethod(name, params);
43          } catch (Exception e) {
44              throw new RuntimeException(e);
45          }
46          m.setAccessible(true);
47          return m;
48      }
49  
50      // Round to next 4096 bytes
51      private static long roundTo4096(long i) {
52          return (i + 0xfffL) & ~0xfffL;
53      }
54  
55      // Given that the location and size have been set, map that location
56      // for the given length and set this.addr to the returned offset
57      private void mapAndSetOffset() {
58          try {
59              final RandomAccessFile backingFile = new RandomAccessFile(this.file, "rw");
60              backingFile.setLength(this.size);
61              final FileChannel ch = backingFile.getChannel();
62              this.addr = (Long) mmap.invoke(ch, 1, 0L, this.size);
63  
64              ch.close();
65              backingFile.close();
66          } catch (Exception e) {
67              throw new RuntimeException(e);
68          }
69      }

79      public void close() {
80          try {
81              unmmap.invoke(null, addr, this.size);
82              if (!file.delete()) {
83                  throw new RuntimeException("could not delete " + file);
84              }
85          } catch (IllegalAccessException e) {
86              throw new RuntimeException(e);
87          } catch (IllegalArgumentException e) {
88              throw new RuntimeException(e);
89          } catch (InvocationTargetException e) {
90              throw new RuntimeException(e);
91          }
92      }
93  
94      public int getInt(long pos) {
95          return unsafe.getInt(pos + addr);
96      }
97  
98      public void putByte(long pos, byte val) {
99          unsafe.putByte(pos + addr, val);
100     }
101     
102     public void putInt(long pos, int val) {
103         unsafe.putInt(pos + addr, val);
104     }
105 
106     public void putOrderedInt(long pos, int val) {
107         unsafe.putOrderedInt(null, pos + addr, val);
108     }
109 
110     public int getIntVolatile(long pos) {
111         return unsafe.getIntVolatile(null, pos + addr);
112     }
113 
114     // May want to have offset & length within data as well, for both of these
115     public void getBytes(long pos, byte[] data, long offset, long length) {
116         unsafe.copyMemory(null, pos + addr, data, BYTE_ARRAY_OFFSET + offset, length);
117     }
118 
119     public void putBytes(long pos, byte[] data, long offset, long length) {
120         unsafe.copyMemory(data, BYTE_ARRAY_OFFSET + offset, null, pos + addr, length);
121     }
122 
123     public byte getByte(long pos) {
124         return unsafe.getByte(pos + addr);
