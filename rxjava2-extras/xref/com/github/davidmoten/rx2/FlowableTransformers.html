31      public static <State, In, Out> FlowableTransformer<In, Out> stateMachine(Callable<? extends State> initialState,
32              Function3<? super State, ? super In, ? super FlowableEmitter<Out>, ? extends State> transition,
33              BiPredicate<? super State, ? super FlowableEmitter<Out>> completion,
34              BackpressureStrategy backpressureStrategy, int requestBatchSize) {
35          return TransformerStateMachine.create(initialState, transition, completion, backpressureStrategy,
36                  requestBatchSize);
37      }

43      public static <T> FlowableTransformer<T, T> doOnEmpty(final Action action) {
44          return new FlowableTransformer<T, T>() {
45  
46              @Override
47              public Publisher<T> apply(Flowable<T> upstream) {
48                  return new FlowableDoOnEmpty<T>(upstream, action);
49              }
50          };
51      }

104     public static <T extends Number> FlowableTransformer<T, Statistics> collectStats() {
105         return new FlowableTransformer<T, Statistics>() {
106 
107             @Override
108             public Flowable<Statistics> apply(Flowable<T> source) {
109                 return source.scan(Statistics.create(), BiFunctions.collectStats());
110             }
111         };
112     }

114     public static <T, R extends Number> FlowableTransformer<T, Pair<T, Statistics>> collectStats(
115             final Function<? super T, ? extends R> function) {
116         return new FlowableTransformer<T, Pair<T, Statistics>>() {
117 
118             @Override
119             public Flowable<Pair<T, Statistics>> apply(Flowable<T> source) {
120                 return source.scan(Pair.create((T) null, Statistics.create()),
121                         new BiFunction<Pair<T, Statistics>, T, Pair<T, Statistics>>() {
122                             @Override
123                             public Pair<T, Statistics> apply(Pair<T, Statistics> pair, T t) throws Exception {
124                                 return Pair.create(t, pair.b().add(function.apply(t)));
125                             }
126                         }).skip(1);
127             }
128         };
129     }
