1   package com.github.davidmoten.grumpy.wms.demo;
2   
3   import java.io.IOException;
4   
5   import javax.servlet.ServletException;
6   import javax.servlet.http.HttpServlet;
7   import javax.servlet.http.HttpServletRequest;
8   import javax.servlet.http.HttpServletResponse;
9   
10  import com.github.davidmoten.grumpy.wms.Capabilities;
11  import com.github.davidmoten.grumpy.wms.WmsServletRequestProcessor;
12  import com.github.davidmoten.grumpy.wms.layer.darkness.DarknessLayer;
13  
14  public class WmsServlet extends HttpServlet {
15      private static final long serialVersionUID = 1518113833457077766L;
16  
17      private static final String SERVICE_TITLE = "Custom OGC Services";
18      private static final String SERVICE_NAME = "CustomOGC";
19      private static final String SERVICE_ABSTRACT = "Custom OGC WMS services including Custom, Fiddle and Darkness layers";
20  
21      private final WmsServletRequestProcessor processor;
22  
23      public WmsServlet() {
24  
25          // instantiate the layers
26          CustomLayer custom = new CustomLayer();
27          DarknessLayer darkness = new DarknessLayer();
28          FiddleLayer fiddle = new FiddleLayer();
29  
30          // setup the capabilities of the service which will extract features
31          // from the layers to fill in defaults for the layer fields in generated
32          // capabilities.xml
33          Capabilities cap = Capabilities.builder()
34          // set service name
35                  .serviceName(SERVICE_NAME)
36                  // set service title
37                  .serviceTitle(SERVICE_TITLE)
38                  // set service abstract
39                  .serviceAbstract(SERVICE_ABSTRACT)
40                  // add image format
41                  .imageFormat("image/png")
42                  // add info format
43                  .infoFormat("text/html")
44                  // add custom layer
45                  .layer(custom)
46                  // add darkness layer
47                  .layer(darkness)
48                  // add fiddle layer
49                  .layer(fiddle)
50                  // build caps
51                  .build();
52  
53          // initialize the request processor
54          processor = WmsServletRequestProcessor.builder()
55          // capabilities
56                  .capabilities(cap)
57                  // or use
58                  // .capabilitiesFromClasspath("/wms-capabilities.xml")
59                  // set image cache size
60                  .imageCache(200)
61                  // add custom layer as cached
62                  .addCachedLayer("Custom", custom)
63                  // add darkness, not cached
64                  .addLayer("Darkness", darkness)
65                  // add fiddles layer
66                  .addLayer("Fiddle", fiddle)
67                  // build it up
68                  .build();
69      }
70  
71      @Override
72      protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException,
73              IOException {
74  
75          // use the processor to handle requests
76          processor.doGet(req, resp);
77      }
78  
79  }
