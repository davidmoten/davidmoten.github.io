102         }

104         public Builder addLayer(String name, Layer layer, boolean cache) {
105             layersBuilder.add(name, layer);
106             if (cache)
107                 layersToCache.add(name);
108             return this;
109         }
110 
111         public Builder layers(Layers layers) {
112             this.layers = layers;
113             return this;
114         }
115 
116         public Builder imageWriter(ImageWriter imageWriter) {
117             this.imageWriter = imageWriter;
118             return this;
119         }
120 
121         public WmsServletRequestProcessor build() {
122             if (imageCacheSize != null)
123                 imageCache = new ImageCache(imageCacheSize);
124             for (String layer : layersToCache)
125                 imageCache.add(layer);
126             if (layers == null)
127                 layers = layersBuilder.build();
128             return new WmsServletRequestProcessor(capabilitiesProvider, layers, imageCache,
129                     imageWriter);
130         }

133     public void doGet(HttpServletRequest request, HttpServletResponse response)
134             throws ServletException, IOException {
135         long t = System.currentTimeMillis();
136         try {
137             log.info("httpGetUrl=" + request.getRequestURL() + "?" + request.getQueryString());
138             log.info("requestedByIP = ip " + request.getRemoteAddr());
139             String req = request.getParameter(PARAMETER_REQUEST);
140             setNoCacheParameters(response);
141             if (REQUEST_GET_CAPABILITIES.equals(req)) {
142                 writeCapabilities(request, response);
143             } else if (REQUEST_GET_MAP.equals(req)) {
144                 writeImage(request, response);
145             } else if (REQUEST_GET_FEATURE_INFO.equals(req)) {
146                 writeFeatureInfo(request, response);
147             } else
148                 throw new UnknownParameterException("Unrecognized REQUEST parameter: " + req);
149             // flush everything so timer below is realistic for delivery to
150             // client
151             response.getOutputStream().flush();
152         } catch (UnknownParameterException e) {
153             log.warn(e.getMessage());
154             throw new ServletException(e);
155         } catch (MissingMandatoryParameterException e) {
156             log.warn(e.getMessage(), e);
157             throw new ServletException(e);
158         } catch (Exception e) {
159             handleException(e);
160         } finally {
161             log.info("requestTimeSeconds="
162                     + new DecimalFormat("0.000").format((System.currentTimeMillis() - t) / 1000.0)
163                     + "s");
164         }
165     }

167     private void handleException(Exception e) throws ServletException {
168         if (e.getClass().getName().contains("ClientAbortException") || e.getMessage() != null
169                 && e.getMessage().contains("Broken pipe")
170                 || e.getCause() instanceof java.net.SocketException) {
171             String s = e.getMessage();
172             if (s == null)
173                 s = e.getClass().getName();
174             log.warn(e.getMessage());
175         } else {
176             log.error(e.getClass().getName());
177             log.error(e.getMessage(), e);
178             throw new ServletException(e);
179         }
180     }

182     private void writeCapabilities(HttpServletRequest request, HttpServletResponse response)
183             throws IOException {
184         response.setContentType("text/xml");
185         String capabilities = capabilitiesProvider.getCapabilities(request);
186         response.getOutputStream().write(capabilities.getBytes());

190     private void writeImage(HttpServletRequest request, HttpServletResponse response)
191             throws MissingMandatoryParameterException, IOException {
192         log.info("getting image");
193         WmsRequest wmsRequest = new WmsRequest(request);
194         byte[] bytes = null;
195         response.setContentType(wmsRequest.getFormat());
196 
197         // check the cache for the bytes of the image converted to the
198         // appropriate format. Note that the critical bottleneck is
199         // ImageIO.write rather than the layerManager.getImage call
200         bytes = imageCache.get(wmsRequest);
201 
202         // if cacheImage=false then don't use cache
203         if ("false".equals(request.getParameter("cacheImage")))
204             bytes = null;
205 
206         if (bytes == null) {
207             log.info("image cache empty");
208 
209             BufferedImage image = null;
210             // dynamic layers should clear the imageCache in a separate thread
211             // (for example, using a quartz job)
212             image = layerManager.getImage(wmsRequest);
213             // Note that we write the image to memory first to avoid this JRE
214             // bug:
215             // http://bugs.sun.com/bugdatabase/view_bug.do;jsessionid=dc84943191e06dffffffffdf200f5210dd319?bug_id=6967419
216             // which is commented on further in JIRA ER-95
217             log.info("writing image to memory for layers " + wmsRequest.getLayers());
218             ByteArrayOutputStream byteOs = new ByteArrayOutputStream();
219             String imageType = wmsRequest.getFormat().substring(
220                     wmsRequest.getFormat().indexOf('/') + 1);
221             // This call is slow!!
222             long t = System.currentTimeMillis();
223             imageWriter.writeImage(image, byteOs, imageType);
224             log.info("ImageIoWriteTimeMs=" + (System.currentTimeMillis() - t));
225             bytes = byteOs.toByteArray();
226             imageCache.put(wmsRequest, bytes);
227         } else
228             log.info("obtained image from cache for layers " + wmsRequest.getLayers());
229 
230         log.info("writing image to http output stream for layers " + wmsRequest.getLayers());
231         response.getOutputStream().write(bytes);
232         response.getOutputStream().flush();
233         log.info("imageSizeK=" + new DecimalFormat("0.000").format(bytes.length / 1000.0)
234                 + " for layers " + wmsRequest.getLayers());
235     }

237     private void writeFeatureInfo(HttpServletRequest request, HttpServletResponse response)
238             throws MissingMandatoryParameterException, IOException {
239         log.info("getting feature info");
240         int i = getI(request);
241         int j = getJ(request);
242         WmsRequest wmsRequest = new WmsRequest(request);
243         response.setContentType(wmsRequest.getInfoFormat());
244         BufferedOutputStream bos = new BufferedOutputStream(response.getOutputStream());
245         Map<String, String> infos = layerManager.getInfos(new Date(), wmsRequest, new Point(i, j),
246                 wmsRequest.getInfoFormat());
247 
248         for (Entry<String, String> entry : infos.entrySet()) {
249             log.debug(entry.getKey() + "=" + entry.getValue());
250             bos.write(("<p>" + entry.getKey() + "</p>").getBytes());
251             bos.write(entry.getValue().getBytes());
252         }
253         bos.close();
254     }

256     private int getJ(HttpServletRequest request) {
257         if (request.getParameter("J") != null)
258             return Math.round(Float.parseFloat(request.getParameter("J")));
259         else
260             // GAIA uses x, y instead of spec I,J!
261             return Math.round(Float.parseFloat(request.getParameter("Y")));
262 
263     }
264 
265     private int getI(HttpServletRequest request) {
266         if (request.getParameter("J") != null)
267             return Math.round(Float.parseFloat(request.getParameter("I")));
268         else
269             // GAIA uses x, y instead of spec I,J!
270             return Math.round(Float.parseFloat(request.getParameter("X")));
271 
272     }

274     private void setNoCacheParameters(HttpServletResponse response) {
275         // Set to expire far in the past.
276         response.setHeader("Expires", "-1");
277 
278         // Set standard HTTP/1.1 no-cache headers.
279         response.setHeader("Cache-Control", "no-store, no-cache, must-revalidate");
280 
281         // Set IE extended HTTP/1.1 no-cache headers (use addHeader).
282         response.addHeader("Cache-Control", "post-check=0, pre-check=0");
283 
284         // Set standard HTTP/1.0 no-cache header.
285         response.setHeader("Pragma", "no-cache");
286     }
