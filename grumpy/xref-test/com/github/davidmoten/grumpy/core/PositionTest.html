40      @Test
41      public final void testPredict() {
42          Position p = new Position(53, 3);
43          Position p2 = p.predict(100, 30);
44          Assert.assertEquals(53.77644258276322, p2.getLat(), 0.01);
45          Assert.assertEquals(3.7609191005595877, p2.getLon(), 0.01);
46  
47          // large distance tests around the equator
48  
49          double meanCircumferenceKm = 40041.47;
50  
51          // start on equator, just East of Greenwich
52          p = new Position(0, 3);
53  
54          // quarter circumference, Still in Eastern hemisphere
55  
56          p2 = p.predict(meanCircumferenceKm / 4.0, 90);
57          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
58          Assert.assertEquals(93.0, p2.getLon(), 0.1);
59  
60          // half circumference, Now in Western hemisphere,
61          // just East of the International date line
62  
63          p2 = p.predict(meanCircumferenceKm / 2.0, 90);
64          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
65          Assert.assertEquals(-177.0, p2.getLon(), 0.1);
66  
67          // three quarters circumference, Now in Western hemisphere,
68  
69          p2 = p.predict(3.0 * meanCircumferenceKm / 4.0, 90);
70          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
71          Assert.assertEquals(-87.0, p2.getLon(), 0.1);
72  
73          // full circumference, back to start
74          // relax Longitude tolerance slightly
75  
76          p2 = p.predict(meanCircumferenceKm, 90);
77          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
78          Assert.assertEquals(p.getLon(), p2.getLon(), 0.2);
79  
80          // same thing but backwards (heading west)
81  
82          // quarter circumference, Still in western hemisphere
83  
84          p2 = p.predict(meanCircumferenceKm / 4.0, 270);
85          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
86          Assert.assertEquals(-87.0, p2.getLon(), 0.1);
87  
88          // half circumference, Now in Western hemisphere,
89          // just East of the International date line
90  
91          p2 = p.predict(meanCircumferenceKm / 2.0, 270);
92          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
93          Assert.assertEquals(-177.0, p2.getLon(), 0.1);
94  
95          // three quarters circumference, Now in eastern hemisphere,
96  
97          p2 = p.predict(3.0 * meanCircumferenceKm / 4.0, 270);
98          Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
99          Assert.assertEquals(93.0, p2.getLon(), 0.1);
100 
101         // full circumference, back to start
102         // relax Longitude tolerance slightly
103 
104         p2 = p.predict(meanCircumferenceKm, 90);
105         Assert.assertEquals(p.getLat(), p2.getLat(), 0.001);
106         Assert.assertEquals(p.getLon(), p2.getLon(), 0.2);
107 
108         // OVER THE South POLE
109         // ===================
110 
111         // quarter circumference, should be at south pole
112 
113         p2 = p.predict(meanCircumferenceKm / 4.0, 180);
114         Assert.assertEquals(-90.0, p2.getLat(), 0.1);
115 
116         // this next assertion is by no means confident
117         // expecting 3 but getting it's reciprocal.
118         // Strange things happen at the pole!!
119 
120         Assert.assertEquals(-177, p2.getLon(), 0.00001);
121 
122         // half circumference, should be at the equator
123         // but in the Western hemisphere
124 
125         p2 = p.predict(meanCircumferenceKm / 2.0, 180);
126         Assert.assertEquals(p.getLat(), p2.getLat(), 0.1);
127         Assert.assertEquals(-177, p2.getLon(), 0.00001);
128 
129         // 3/4 circumference, should be at the north Pole
130         // but in the Western hemisphere
131 
132         p2 = p.predict(3.0 * meanCircumferenceKm / 4.0, 180);
133         Assert.assertEquals(90.0, p2.getLat(), 0.1);
134         Assert.assertEquals(p.getLon(), p2.getLon(), 0.00001);
135 
136         // full circumference, back to start
137         // relax latitude tolerance slightly
138 
139         p2 = p.predict(meanCircumferenceKm, 270);
140         Assert.assertEquals(p.getLat(), p2.getLat(), 0.1);
141         Assert.assertEquals(p.getLon(), p2.getLon(), 0.2);
142

241 
242         // Create new position objects
243         Position p1 = new Position(36.12, -86.67);
244         Position p2 = new Position(33.94, -118.4);
245 
246         {
247             double distanceKm = p1.getDistanceToKm(p2);
248             double bearingDegrees = p1.getBearingDegrees(p2);
249             Position p3 = p1.predict(distanceKm * 0.7, bearingDegrees);
250 
251             // Expected position
252             Position actual = p1.getPositionAlongPath(p2, 0.7);
253             // Test expected Lat return position
254             Assert.assertEquals(p3.getLat(), actual.getLat(), 0.01);
255             // Test expected Lon return position
256             Assert.assertEquals(p3.getLon(), actual.getLon(), 0.01);
257             // Test expected Lat return position
258             Assert.assertEquals(35.47, actual.getLat(), 0.01);
259             // Test expected Lon return position
260             Assert.assertEquals(-109.11, actual.getLon(), 0.01);
261 
262         }
263 
264         {
265             // If start point equals end point then a proportion
266             // along the path should equal the start point.
267             Position actual = p1.getPositionAlongPath(p1, 0.7);
268             // Test expected Lat return position
269             Assert.assertEquals(p1.getLat(), actual.getLat(), 0.01);
270             // Test expected Lon return position
271             Assert.assertEquals(p1.getLon(), actual.getLon(), 0.01);
272         }
273 
274         {
275             // If proportion is 0.0 then should return start point
276             Position actual = p1.getPositionAlongPath(p2, 0.0);
277             // Test expected Lat return position
278             Assert.assertEquals(p1.getLat(), actual.getLat(), 0.01);
279             // Test expected Lon return position
280             Assert.assertEquals(p1.getLon(), actual.getLon(), 0.01);
281         }
282 
283         {
284             // If proportion is 1.0 then should return end point
285             Position actual = p1.getPositionAlongPath(p2, 1.0);
286             // Test expected Lat return position
287             Assert.assertEquals(p2.getLat(), actual.getLat(), 0.01);
288             // Test expected Lon return position
289             Assert.assertEquals(p2.getLon(), actual.getLon(), 0.01);
290         }
