14      static List<Entry<Object, Rectangle>> entries1000(Precision precision) {
15          List<Entry<Object, Rectangle>> list = new ArrayList<Entry<Object, Rectangle>>();
16          BufferedReader br = new BufferedReader(
17                  new InputStreamReader(BenchmarksRTree.class.getResourceAsStream("/1000.txt")));
18          String line;
19          try {
20              while ((line = br.readLine()) != null) {
21                  String[] items = line.split(" ");
22                  double x = Double.parseDouble(items[0]);
23                  double y = Double.parseDouble(items[1]);
24                  Entry<Object, Rectangle> entry;
25                  if (precision == Precision.DOUBLE)
26                      entry = Entries.entry(new Object(), Geometries.rectangle(x, y, x + 1, y + 1));
27                  else
28                      entry = Entries.entry(new Object(), Geometries.rectangle((float) x, (float) y,
29                              (float) x + 1, (float) y + 1));
30                  list.add(entry);
31              }
32              br.close();
33          } catch (IOException e) {
34              throw new RuntimeException(e);
35          }
36          return list;
37      }
