59      @SuppressWarnings("unchecked")
60      @Test
61      public void testAddToFlatBuffers() throws IOException {
62          Entry<String, Point> a = Entries.entry("hello", Geometries.point(1, 2));
63          Entry<String, Point> b = Entries.entry("there", Geometries.point(3, 4));
64          Entry<String, Point> c = Entries.entry("you", Geometries.point(5, 6));
65          RTree<String, Point> tree = RTree.create();
66          tree = tree.add(a).add(b);
67          Serializer<String, Point> serializer = Serializers.flatBuffers().utf8();
68          ByteArrayOutputStream bytes = new ByteArrayOutputStream();
69          serializer.write(tree, bytes);
70          bytes.close();
71          byte[] array = bytes.toByteArray();
72          RTree<String, Point> tree2 = serializer.read(new ByteArrayInputStream(array), array.length,
73                  InternalStructure.SINGLE_ARRAY);
74          tree2 = tree2.add(c);
75          assertEquals(Sets.newHashSet(a, b, c),
76                  Sets.newHashSet(tree2.entries().toList().toBlocking().single()));
77      }

162     @SuppressWarnings("unchecked")
163     private static <S extends Geometry> void check(Serializer<String, S> serializer,
164             Entry<String, S> a, Entry<String, S> b) throws IOException {
165         RTree<String, S> tree = RTree.create();
166         tree = tree.add(a).add(b);
167         ByteArrayOutputStream bytes = new ByteArrayOutputStream();
168         serializer.write(tree, bytes);
169         bytes.close();
170         {
171             ByteArrayInputStream input = new ByteArrayInputStream(bytes.toByteArray());
172             RTree<String, S> tree2 = serializer.read(input, bytes.size(),
173                     InternalStructure.DEFAULT);
174             assertEquals(2, tree2.size());
175             assertEquals(Sets.newHashSet(a, b),
176                     Sets.newHashSet(tree2.entries().toList().toBlocking().single()));
177         }
178         {
179             ByteArrayInputStream input = new ByteArrayInputStream(bytes.toByteArray());
180             RTree<String, S> tree2 = serializer.read(input, bytes.size(),
181                     InternalStructure.SINGLE_ARRAY);
182             assertEquals(2, tree2.size());
183             assertEquals(Sets.newHashSet(a, b),
184                     Sets.newHashSet(tree2.entries().toList().toBlocking().single()));
185         }
186     }
