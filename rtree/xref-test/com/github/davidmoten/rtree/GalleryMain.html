15          List<Integer> sizes = Arrays.asList(100, 1000, 10000, 1000000);
16          List<Integer> maxChildrenValues = Arrays.asList(4, 8, 16, 32, 64, 128);
17          for (int size : sizes)
18              for (int maxChildren : maxChildrenValues) {
19                  if (size > maxChildren) {
20                      System.out.println("saving " + size + " m=" + maxChildren);
21                      RTree<Object, Point> tree = RTree.maxChildren(maxChildren)
22                              .<Object, Point> create().add(entries.take(size)).last().toBlocking()
23                              .single();
24                      tree.visualize(600, 600).save(
25                              "target/greek-" + size + "-" + maxChildren + "-quad.png");
26                      RTree<Object, Point> tree2 = RTree.star().maxChildren(maxChildren)
27                              .<Object, Point> create().add(entries.take(size)).last().toBlocking()
28                              .single();
29                      tree2.visualize(600, 600).save(
30                              "target/greek-" + size + "-" + maxChildren + "-star.png");
31                  }
32              }
