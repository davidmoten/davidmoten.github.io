8   import rx.Observable;

10  public class GalleryMain {
11  
12      public static void main(String[] args) {
13          Observable<Entry<Object, Point>> entries = GreekEarthquakes.entries(Precision.DOUBLE)
14                  .cache();
15  
16          List<Integer> sizes = Arrays.asList(100, 1000, 10000, 1000000);
17          List<Integer> maxChildrenValues = Arrays.asList(4, 8, 16, 32, 64, 128);
18          for (int size : sizes)
19              for (int maxChildren : maxChildrenValues) {
20                  if (size > maxChildren) {
21                      System.out.println("saving " + size + " m=" + maxChildren);
22                      RTree<Object, Point> tree = RTree.maxChildren(maxChildren)
23                              .<Object, Point>create().add(entries.take(size)).last().toBlocking()
24                              .single();
25                      tree.visualize(600, 600)
26                              .save("target/greek-" + size + "-" + maxChildren + "-quad.png");
27                      RTree<Object, Point> tree2 = RTree.star().maxChildren(maxChildren)
28                              .<Object, Point>create().add(entries.take(size)).last().toBlocking()
29                              .single();
30                      tree2.visualize(600, 600)
31                              .save("target/greek-" + size + "-" + maxChildren + "-star.png");
32                  }
33              }
34      }
35  }
