24      public void testHighPrecision() {
25          RTree<Integer, Rectangle> tree = RTree.create();
26          tree = tree.add(1, Geometries.rectangle(0, 0, 1, 1));
27          double x1 = 2.0000000001;
28          System.out.println((float) x1);
29          tree = tree.add(2, Geometries.rectangle(x1, 2, 3, 3));
30          tree.search(Geometries.rectangle((float) x1, 2.0, (float) x1, 2.0)) //
31                  .test() //
32                  .assertNoValues() //
33                  .assertCompleted();
34          tree.search(Geometries.rectangle(x1, 2.0, x1, 2.0)) //
35                  .test() //
36                  .assertValueCount(1) //
37                  .assertCompleted();
38      }
