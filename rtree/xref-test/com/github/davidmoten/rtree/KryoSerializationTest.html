21      public void testRTree() {

23          ByteArrayOutputStream bytes = new ByteArrayOutputStream();
24          Output output = new Output(bytes);
25          RTree<String, Point> tree = RTree.<String, Point> create()
26                  .add(Entries.entry("thing", Geometries.point(10, 20)))
27                  .add(Entries.entry("monster", Geometries.point(23, 45)));
28          kryo.writeObject(output, tree);
29          output.close();
30          Input input = new Input(new ByteArrayInputStream(bytes.toByteArray()));
31          @SuppressWarnings("unchecked")
32          RTree<String, Point> tree2 = kryo.readObject(input, RTree.class);
33          assertEquals(2, (int) tree2.entries().count().toBlocking().single());
34      }

36      @Test
37      public void testKryo() {
38          Kryo kryo = new Kryo();
39          ByteArrayOutputStream bytes = new ByteArrayOutputStream();
40          Output output = new Output(bytes);
41          Boo b = new Boo("hello");
42          kryo.register(Boo.class);
43          kryo.writeObject(output, b);
44          output.close();
45          Input input = new Input(new ByteArrayInputStream(bytes.toByteArray()));
46          Boo b2 = kryo.readObject(input, Boo.class);
47          assertEquals("hello", b2.name);
48      }
