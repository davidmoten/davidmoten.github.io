19  	@Test
20  	public void testSearchEmptyTree() {
21  		RTree tree = new RTree();
22  		assertTrue(tree.search(r(1)).isEmpty().toBlocking().single());
23  	}
24  
25  	@Test
26  	public void testSearchOnOneItem() {
27  		RTree tree = new RTree();
28  		Entry entry = new Entry(new Object(), r(1));
29  		tree = tree.add(entry);
30  		assertEquals(Arrays.asList(entry), tree.search(r(1)).toList()
31  				.toBlocking().single());
32  		System.out.println("entries="
33  				+ tree.entries().toList().toBlocking().single());
34  	}
35  
36  	@Test
37  	public void testPerformance() {
38  		long t = System.currentTimeMillis();
39  		RTree tree = new RTree();
40  		int n = 10000;
41  		for (int i = 0; i < n; i++) {
42  			Entry entry = new Entry(new Object(), random());
43  			tree = tree.add(entry);
44  		}
45  		long diff = System.currentTimeMillis() - t;
46  		System.out.println("inserts/second = " + ((double) n / diff * 1000));
47  		assertEquals(n, (int) tree.entries().count().toBlocking().single());
48  	}
49  
50  	@Test
51  	public void testVisualizer() {
52  		RTree tree = new RTree();
53  		int n = 100;
54  		for (int i = 0; i < n; i++) {
55  			Entry entry = new Entry(new Object(), random());
56  			tree = tree.add(entry);
57  		}
58  		tree.visualize(600, 600, new Rectangle(-20, -20, 1100, 1100), 5).save(
59  				new File("target/tree.png"), "PNG");
60  		;
61  	}
62  
63  	private static Rectangle r(int n) {
64  		return Rectangle.create(n, n, n + 1, n + 1);
65  	}
66  
67  	private static Rectangle r(int n, int m) {
68  		return Rectangle.create(n, m, n + 1, m + 1);
69  	}
70  
71  	private static Rectangle random() {
72  		return r((int) Math.round(Math.sqrt(Math.random()) * 1000),
73  				(int) Math.round(Math.sqrt(Math.random()) * 1000));
74  	}
