24      private static Comparator<Integer> integerComparator = new Comparator<Integer>() {
25  
26          @Override
27          public int compare(Integer i1, Integer i2) {
28              return i1.compareTo(i2);
29          }
30      };

32      @Test
33      public void testPriority() {
34          List<Integer> list = Observable.range(1, 5)
35                  .lift(new OperatorBoundedPriorityQueue<Integer>(2, integerComparator)).toSortedList()
36                  .toBlocking().single();
37          assertEquals(Arrays.asList(1, 2), list);
38      }

40      @Test
41      public void testUnsubscribeAfterFirst() {
42          final AtomicBoolean completed = new AtomicBoolean(false);
43          Observable.range(1, 5)
44                  // go through priority queue
45                  .lift(new OperatorBoundedPriorityQueue<Integer>(2, integerComparator))
46                  // subscribe

56                      }
