16  public final class Line implements Geometry {

23      private Line(float x1, float y1, float x2, float y2) {
24          this.x1 = x1;
25          this.y1 = y1;
26          this.x2 = x2;
27          this.y2 = y2;
28      }
29  
30      public static Line create(float x1, float y1, float x2, float y2) {
31          return new Line(x1, y1, x2, y2);
32      }
33  
34      @Override
35      public double distance(Rectangle r) {
36          if (r.contains(x1, y1) || r.contains(x2, y2)) {
37              return 0;
38          } else {
39              double d1 = distance(r.x1(), r.y1(), r.x1(), r.y2());
40              if (d1 == 0)
41                  return 0;
42              double d2 = distance(r.x1(), r.y2(), r.x2(), r.y2());
43              if (d2 == 0)
44                  return 0;
45              double d3 = distance(r.x2(), r.y2(), r.x2(), r.y1());
46              double d4 = distance(r.x2(), r.y1(), r.x1(), r.y1());
47              return Math.min(d1, Math.min(d2, Math.min(d3, d4)));
48          }
49      }
50  
51      private double distance(float x1, float y1, float x2, float y2) {
52          Float line = new Line2D.Float(x1, y1, x2, y2);
53          double d1 = line.ptSegDist(this.x1, this.y1);
54          double d2 = line.ptSegDist(this.x2, this.y2);
55          Float line2 = new Line2D.Float(this.x1, this.y1, this.x2, this.y2);
56          double d3 = line2.ptSegDist(x1, y1);
57          if (d3 == 0)
58              return 0;
59          double d4 = line2.ptSegDist(x2, y2);
60          if (d4 == 0)
61              return 0;
62          else
63              return Math.min(d1, Math.min(d2, Math.min(d3, d4)));
64  
65      }

67      @Override
68      public Rectangle mbr() {
69          return Geometries.rectangle(Math.min(x1, x2), Math.min(y1, y2), Math.max(x1, x2),
70                  Math.max(y1, y2));
71      }

73      @Override
74      public boolean intersects(Rectangle r) {
75          GeometryFactory gf = new GeometryFactory();
76          GeometricShapeFactory f = new GeometricShapeFactory(gf);
77          f.setBase(new Coordinate(r.x1(), r.y1()));
78          f.setWidth(r.x2() - r.x1());
79          f.setHeight(r.y2() - r.y1());
80          Polygon rect = f.createRectangle();
81          LineSegment line = new LineSegment(x1, y1, x2, y2);
82          return RectangleIntersects.intersects(rect, line.toGeometry(gf));
83      }

101     public boolean intersects(Line b) {
102         Line2D line1 = new Line2D.Float(x1, y1, x2, y2);
103         Line2D line2 = new Line2D.Float(b.x1(), b.y1(), b.x2(), b.y2());
104         return line2.intersectsLine(line1);
105     }

107     public boolean intersects(Point point) {
108         return intersects(point.mbr());
109     }

111     public boolean intersects(Circle circle) {
112         // using Vector Projection
113         // https://en.wikipedia.org/wiki/Vector_projection
114         Vector c = Vector.create(circle.x(), circle.y());
115         Vector a = Vector.create(x1, y1);
116         Vector cMinusA = c.minus(a);
117         float radiusSquared = circle.radius() * circle.radius();
118         if (x1 == x2 && y1 == y2) {
119             return cMinusA.modulusSquared() <= radiusSquared;
120         } else {
121             Vector b = Vector.create(x2, y2);
122             Vector bMinusA = b.minus(a);
123             float bMinusAModulus = bMinusA.modulus();
124             float lambda = cMinusA.dot(bMinusA) / bMinusAModulus;
125             // if projection is on the segment
126             if (lambda >= 0 && lambda <= bMinusAModulus) {
127                 Vector dMinusA = bMinusA.times(lambda / bMinusAModulus);
128                 // calculate distance to line from c using pythagoras' theorem
129                 return cMinusA.modulusSquared() - dMinusA.modulusSquared() <= radiusSquared;
130             } else {
131                 // return true if and only if an endpoint is within radius of
132                 // centre
133                 return cMinusA.modulusSquared() <= radiusSquared
134                         || c.minus(b).modulusSquared() <= radiusSquared;
135             }
136         }
137     }

139     private static final class Vector {
140         final float x;
141         final float y;
142 
143         static Vector create(float x, float y) {
144             return new Vector(x, y);
145         }
146 
147         Vector(float x, float y) {
148             this.x = x;
149             this.y = y;
150         }
151 
152         float dot(Vector v) {
153             return x * v.x + y * v.y;
154         }
155 
156         Vector times(float value) {
157             return create(value * x, value * y);
158         }
159 
160         Vector minus(Vector v) {
161             return create(x - v.x, y - v.y);
162         }
163 
164         float modulus() {
165             return (float) Math.sqrt(x * x + y * y);
166         }
167 
168         float modulusSquared() {
169             return x * x + y * y;
170         }
171 
172     }
