4   import com.github.davidmoten.guavamini.Optional;

8   public final class Rectangle implements Geometry, HasGeometry {
9       private final float x1, y1, x2, y2;
10  
11      protected Rectangle(float x1, float y1, float x2, float y2) {
12          Preconditions.checkArgument(x2 >= x1);
13          Preconditions.checkArgument(y2 >= y1);
14          this.x1 = x1;
15          this.y1 = y1;
16          this.x2 = x2;
17          this.y2 = y2;
18      }
19  
20      public float x1() {
21          return x1;
22      }
23  
24      public float y1() {
25          return y1;
26      }
27  
28      public float x2() {
29          return x2;
30      }
31  
32      public float y2() {
33          return y2;
34      }
35  
36      public float area() {
37          return (x2 - x1) * (y2 - y1);
38      }
39  
40      public Rectangle add(Rectangle r) {
41          return new Rectangle(min(x1, r.x1), min(y1, r.y1), max(x2, r.x2), max(y2, r.y2));
42      }
43  
44      public static Rectangle create(double x1, double y1, double x2, double y2) {
45          return new Rectangle((float) x1, (float) y1, (float) x2, (float) y2);
46      }
47  
48      public static Rectangle create(float x1, float y1, float x2, float y2) {
49          return new Rectangle(x1, y1, x2, y2);
50      }
51  
52      public boolean contains(double x, double y) {
53          return x >= x1 && x <= x2 && y >= y1 && y <= y2;
54      }
55  
56      @Override
57      public boolean intersects(Rectangle r) {
58          return r.x2 >= x1 && r.x1 <= x2 && r.y2 >= y1 && r.y1 <= y2;
59      }
60  
61      @Override
62      public double distance(Rectangle r) {
63          if (intersects(r))
64              return 0;
65          else {
66              Rectangle mostLeft = x1 < r.x1 ? this : r;
67              Rectangle mostRight = x1 > r.x1 ? this : r;
68              double xDifference = max(0,
69                      mostLeft.x1 == mostRight.x1 ? 0 : mostRight.x1 - mostLeft.x2);
70  
71              Rectangle upper = y1 < r.y1 ? this : r;
72              Rectangle lower = y1 > r.y1 ? this : r;
73  
74              double yDifference = max(0, upper.y1 == lower.y1 ? 0 : lower.y1 - upper.y2);
75  
76              return Math.sqrt(xDifference * xDifference + yDifference * yDifference);
77          }
78      }
79  
80      @Override
81      public Rectangle mbr() {
82          return this;
83      }
84  
85      @Override
86      public String toString() {
87          return "Rectangle [x1=" + x1 + ", y1=" + y1 + ", x2=" + x2 + ", y2=" + y2 + "]";
88      }
89  
90      @Override
91      public int hashCode() {
92          return Objects.hashCode(x1, y1, x2, y2);
93      }
94  
95      @Override
96      public boolean equals(Object obj) {
97          Optional<Rectangle> other = ObjectsHelper.asClass(obj, Rectangle.class);
98          if (other.isPresent()) {
99              return Objects.equal(x1, other.get().x1) && Objects.equal(x2, other.get().x2)
100                     && Objects.equal(y1, other.get().y1) && Objects.equal(y2, other.get().y2);
101         } else
102             return false;
103     }
104 
105     public float intersectionArea(Rectangle r) {
106         if (!intersects(r))
107             return 0;
108         else
109             return create(max(x1, r.x1), max(y1, r.y1), min(x2, r.x2), min(y2, r.y2)).area();
110     }
111 
112     public float perimeter() {
113         return 2 * (x2 - x1) + 2 * (y2 - y1);
114     }
115 
116     @Override
117     public Geometry geometry() {
118         return this;
119     }
120 
121     private static float max(float a, float b) {
122         if (a < b)
123             return b;
124         else
125             return a;
126     }
127 
128     private static float min(float a, float b) {
129         if (a < b)
130             return a;
131         else
132             return b;
133     }
134 
135 }
