4   /**
5    * A geometrical region that represents an Entry spatially. It is recommended
6    * that implementations of this interface implement equals() and hashCode()
7    * appropriately that entry equality checks work as expected.
8    */
9   public interface Geometry {
10  
11      /**
12       * <p>
13       * Returns the distance to the given {@link Rectangle}. For a
14       * {@link Rectangle} this might be Euclidean distance but for an EPSG4326
15       * lat-long Rectangle might be great-circle distance. The distance function
16       * should satisfy the following properties:
17       * </p>
18       * 
19       * <p>
20       * <code>distance(r) &gt;= 0</code>
21       * </p>
22       * 
23       * <p>
24       * <code>if r1 contains r2 then distance(r1)&lt;=distance(r2)</code>
25       * </p>
26       * 
27       * 
28       * @param r
29       *            rectangle to measure distance to
30       * @return distance to the rectangle r from the geometry
31       */
32      double distance(Rectangle r);
33  
34      /**
35       * Returns the minimum bounding rectangle of this geometry.
36       * 
37       * @return minimum bounding rectangle
38       */
39      Rectangle mbr();
40  
41      boolean intersects(Rectangle r);
42  }
