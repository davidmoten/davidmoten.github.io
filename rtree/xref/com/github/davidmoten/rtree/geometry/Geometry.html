3   /**
4    * A geometrical region that represents an Entry spatially. It is recommended
5    * that implementations of this interface implement equals() and hashCode()
6    * appropriately that entry equality checks work as expected.
7    */
8   public interface Geometry {
9   
10      /**
11       * <p>
12       * Returns the distance to the given {@link Rectangle}. For a {@link Rectangle}
13       * this might be Euclidean distance but for an EPSG4326 lat-long Rectangle might
14       * be great-circle distance. The distance function should satisfy the following
15       * properties:
16       * </p>
17       * 
18       * <p>
19       * <code>distance(r) &gt;= 0</code>
20       * </p>
21       * 
22       * <p>
23       * <code>if r1 contains r2 then distance(r1)&lt;=distance(r2)</code>
24       * </p>
25       * 
26       * 
27       * @param r
28       *            rectangle to measure distance to
29       * @return distance to the rectangle r from the geometry
30       */
31      double distance(Rectangle r);
32  
33      /**
34       * Returns the minimum bounding rectangle of this geometry.
35       * 
36       * @return minimum bounding rectangle
37       */
38      Rectangle mbr();
39  
40      boolean intersects(Rectangle r);
41  
42      boolean isDoublePrecision();
43  }
