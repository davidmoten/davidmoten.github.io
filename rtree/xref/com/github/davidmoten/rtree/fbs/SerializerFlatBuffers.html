36  public final class SerializerFlatBuffers<T, S extends Geometry> implements Serializer<T, S> {
37  
38      private final FactoryFlatBuffers<T, S> factory;
39  
40      private SerializerFlatBuffers(Func1<? super T, byte[]> serializer,
41              Func1<byte[], ? extends T> deserializer) {
42          this.factory = new FactoryFlatBuffers<T, S>(serializer, deserializer);
43      }
44  
45      public static <T, S extends Geometry> Serializer<T, S> create(
46              Func1<? super T, byte[]> serializer, Func1<byte[], ? extends T> deserializer) {
47          return new SerializerFlatBuffers<T, S>(serializer, deserializer);
48      }

50      /*
51       * (non-Javadoc)
52       * 
53       * @see com.github.davidmoten.rtree.fbs.Serializer#serialize(com.github.
54       * davidmoten.rtree.RTree, java.io.OutputStream)
55       */
56      @Override
57      public void write(RTree<T, S> tree, OutputStream os) throws IOException {
58          FlatBufferBuilder builder = new FlatBufferBuilder();
59          final Rectangle mbb;
60          if (tree.root().isPresent()) {
61              mbb = tree.root().get().geometry().mbr();
62          } else {
63              mbb = Geometries.rectangle(0, 0, 0, 0);
64          }
65          int b = Box_.createBox_(builder, mbb.x1(), mbb.y1(), mbb.x2(), mbb.y2());
66          Context_.startContext_(builder);
67          Context_.addBounds(builder, b);
68          Context_.addMinChildren(builder, tree.context().minChildren());
69          Context_.addMaxChildren(builder, tree.context().maxChildren());
70          int c = Context_.endContext_(builder);
71          final int n;
72          if (tree.root().isPresent()) {
73              n = addNode(tree.root().get(), builder, factory.serializer());
74          } else {
75              // won't be used
76              n = 0;
77          }
78          // int t = Tree_.createTree_(builder, c, n, tree.size());
79          Tree_.startTree_(builder);
80          Tree_.addContext(builder, c);
81          Tree_.addSize(builder, tree.size());
82          if (tree.size() > 0) {
83              Tree_.addRoot(builder, n);
84          }
85          int t = Tree_.endTree_(builder);
86          Tree_.finishTree_Buffer(builder, t);
87  
88          ByteBuffer bb = builder.dataBuffer();
89          os.write(bb.array(), bb.position(), bb.remaining());
90      }
91  
92      private static <T, S extends Geometry> int addNode(Node<T, S> node, FlatBufferBuilder builder,
93              Func1<? super T, byte[]> serializer) {
94          if (node instanceof Leaf) {
95              Leaf<T, S> leaf = (Leaf<T, S>) node;
96              return FlatBuffersHelper.addEntries(leaf.entries(), builder, serializer);
97          } else {
98              NonLeaf<T, S> nonLeaf = (NonLeaf<T, S>) node;
99              int[] nodes = new int[nonLeaf.count()];
100             for (int i = 0; i < nonLeaf.count(); i++) {
101                 Node<T, S> child = nonLeaf.child(i);
102                 nodes[i] = addNode(child, builder, serializer);
103             }
104             int ch = Node_.createChildrenVector(builder, nodes);
105             Node_.startNode_(builder);
106             Node_.addChildren(builder, ch);
107             Rectangle mbb = nonLeaf.geometry().mbr();
108             int b = Box_.createBox_(builder, mbb.x1(), mbb.y1(), mbb.x2(), mbb.y2());
109             Node_.addMbb(builder, b);
110             return Node_.endNode_(builder);
111         }
112     }

114     /*
115      * (non-Javadoc)
116      * 
117      * @see com.github.davidmoten.rtree.fbs.Serializer#deserialize(long,
118      * java.io.InputStream, com.github.davidmoten.rtree.InternalStructure)
119      */
120     @Override
121     public RTree<T, S> read(InputStream is, long sizeBytes, InternalStructure structure)
122             throws IOException {
123         byte[] bytes = readFully(is, (int) sizeBytes);
124         Tree_ t = Tree_.getRootAsTree_(ByteBuffer.wrap(bytes));
125         Context<T, S> context = new Context<T, S>(t.context().minChildren(),
126                 t.context().maxChildren(), new SelectorRStar(), new SplitterRStar(), factory);
127         Node_ node = t.root();
128         if (node == null) {
129             return SerializerHelper.create(Optional.<Node<T, S>> absent(), 0, context);
130         } else {
131             final Node<T, S> root;
132             if (structure == InternalStructure.SINGLE_ARRAY) {
133                 if (node.childrenLength() > 0) {
134                     root = new NonLeafFlatBuffers<T, S>(node, context, factory.deserializer());
135                 } else {
136                     root = new LeafFlatBuffers<T, S>(node, context, factory.deserializer());
137                 }
138             } else {
139                 root = toNodeDefault(node, context, factory.deserializer());
140             }
141             return SerializerHelper.create(Optional.of(root), (int) t.size(), context);
142         }
143     }
144 
145     private static <T, S extends Geometry> Node<T, S> toNodeDefault(Node_ node,
146             Context<T, S> context, Func1<byte[], ? extends T> deserializer) {
147         int numChildren = node.childrenLength();
148         if (numChildren > 0) {
149             List<Node<T, S>> children = new ArrayList<Node<T, S>>(numChildren);
150             for (int i = 0; i < numChildren; i++) {
151                 children.add(toNodeDefault(node.children(i), context, deserializer));
152             }
153             return new NonLeafDefault<T, S>(children, context);
154         } else {
155             List<Entry<T, S>> entries = FlatBuffersHelper.createEntries(node, deserializer);
156             return new LeafDefault<T, S>(entries, context);
157         }
158     }
159 
160     @VisibleForTesting
161     static byte[] readFully(InputStream is, int numBytes) throws IOException {
162         byte[] b = new byte[numBytes];
163         int n = is.read(b);
164         if (n != numBytes)
165             throw new RuntimeException("unexpected");
166         return b;
167     }
