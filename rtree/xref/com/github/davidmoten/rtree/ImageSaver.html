3   import java.awt.image.BufferedImage;
4   import java.io.File;
5   import java.util.concurrent.Callable;
6   
7   import javax.imageio.ImageIO;
8   
9   final class ImageSaver {
10  
11      private ImageSaver() {
12          // prevent instantiation
13      }
14  
15      static void save(final BufferedImage image, final File file, final String imageFormat) {
16          // this is an ugly workaround to achieve 100% coverage (the pesky
17          // IOException catch)
18          // The performance hit of making this allocation should be negligible
19          // compared to the cost of IO
20          Callable<Void> callable = new Callable<Void>() {
21              @Override
22              public Void call() throws Exception {
23                  ImageIO.write(image, imageFormat, file);
24                  return null;
25              }
26          };
27          run(callable);
28      }
29  
30      static void run(Callable<Void> callable) {
31          try {
32              callable.call();
33          } catch (Exception e) {
34              throw new RuntimeException(e);
35          }
36      }
37  
38  }
