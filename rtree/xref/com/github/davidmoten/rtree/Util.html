18  	static Rectangle mbr(Collection<? extends HasMbr> items) {
19  		Preconditions.checkArgument(!items.isEmpty());
20  		Optional<Rectangle> r = Optional.absent();
21  		for (final HasMbr mbr : items) {
22  			if (r.isPresent())
23  				r = of(r.get().add(mbr.mbr()));
24  			else
25  				r = of(mbr.mbr());
26  		}
27  		return r.get();
28  	}
29  
30  	static <T> List<T> add(List<T> list, T element) {
31  		final ArrayList<T> result = new ArrayList<T>(list);
32  		result.add(element);
33  		return result;
34  	}
35  
36  	static <T> List<? extends T> replace(List<? extends T> list, T node,
37  			List<? extends T> replacements) {
38  		final ArrayList<T> result = new ArrayList<T>(list);
39  		result.remove(node);
40  		result.addAll(replacements);
41  		return result;
42  	}
43  
44  	static HasMbr findLeastIncreaseInMbrArea(Rectangle r,
45  			List<? extends HasMbr> list) {
46  		Preconditions.checkArgument(!list.isEmpty());
47  		Optional<Double> minDifference = Optional.absent();
48  		Optional<HasMbr> minDiffItem = Optional.absent();
49  		for (final HasMbr m : list) {
50  			final double diff = m.mbr().add(r).area() - m.mbr().area();
51  			if (!minDifference.isPresent() || diff < minDifference.get()) {
52  				minDifference = of(diff);
53  				minDiffItem = of(m);
54  			}
55  		}
56  		return minDiffItem.get();
57  	}
58  
59  	private static <T extends HasMbr> List<T> sort(List<T> entries,
60  			final Comparator<? super Rectangle> comparator) {
61  		List<T> list = new ArrayList<T>(entries);
62  		Collections.sort(list, new Comparator<T>() {
63  			@Override
64  			public int compare(T e1, T e2) {
65  				return comparator.compare(e1.mbr(), e2.mbr());
66  			}
67  		});
68  		return list;
69  	}
70  
71  	static <T extends HasMbr> List<T> sort(List<T> list,
72  			Optional<Comparator<Rectangle>> comparator) {
73  		if (!comparator.isPresent())
74  			return list;
75  		else
76  			return sort(list, comparator.get());
77  	}
