1   package com.github.davidmoten.rtree.internal.util;
2   
3   import java.util.ArrayList;
4   import java.util.Collections;
5   import java.util.Comparator;
6   import java.util.List;
7   import java.util.PriorityQueue;
8   
9   import com.github.davidmoten.guavamini.Preconditions;
10  
11  public final class BoundedPriorityQueue<T> {
12  
13      private final PriorityQueue<T> queue; /* backing data structure */
14      private final Comparator<? super T> comparator;
15      private final int maxSize;
16  
17      /**
18       * Constructs a {@link BoundedPriorityQueue} with the specified
19       * {@code maxSize} and {@code comparator}.
20       *
21       * @param maxSize
22       *            - The maximum size the queue can reach, must be a positive
23       *            integer.
24       * @param comparator
25       *            - The comparator to be used to compare the elements in the
26       *            queue, must be non-null.
27       */
28      public BoundedPriorityQueue(final int maxSize, final Comparator<? super T> comparator) {
29          Preconditions.checkArgument(maxSize > 0, "maxSize must be > 0");
30          Preconditions.checkNotNull(comparator, "comparator cannot be null");
31          this.queue = new PriorityQueue<T>(reverse(comparator));
32          this.comparator = comparator;
33          this.maxSize = maxSize;
34      }
35  
36      private static <T> Comparator<T> reverse(final Comparator<T> comparator) {
37          return (o1, o2) -> comparator.compare(o2, o1);
38      }
39  
40      public static <T> BoundedPriorityQueue<T> create(final int maxSize,
41              final Comparator<? super T> comparator) {
42          return new BoundedPriorityQueue<T>(maxSize, comparator);
43      }
44  
45      /**
46       * Adds an element to the queue. If the queue contains {@code maxSize}
47       * elements, {@code e} will be compared to the lowest element in the queue
48       * using {@code comparator}. If {@code e} is greater than or equal to the
49       * lowest element, that element will be removed and {@code e} will be added
50       * instead. Otherwise, the queue will not be modified and {@code e} will not
51       * be added.
52       *
53       * @param t
54       *            - Element to be added, must be non-null.
55       */
56      public void add(final T t) {
57          if (t == null) {
58              throw new NullPointerException("cannot add null to the queue");
59          }
60          if (queue.size() >= maxSize) {
61              final T maxElement = queue.peek();
62              if (comparator.compare(maxElement, t) < 1) {
63                  return;
64              } else {
65                  queue.poll();
66              }
67          }
68          queue.add(t);
69      }
70  
71      /**
72       * @return Returns a view of the queue as a
73       *         {@link Collections#unmodifiableList(java.util.List)}
74       *         unmodifiableList sorted in reverse order.
75       */
76      public List<T> asList() {
77          return Collections.unmodifiableList(new ArrayList<>(queue));
78      }
79  
80      public List<T> asOrderedList() {
81          List<T> list = new ArrayList<>(queue);
82          list.sort(comparator);
83          return list;
84      }
85  
86  }
