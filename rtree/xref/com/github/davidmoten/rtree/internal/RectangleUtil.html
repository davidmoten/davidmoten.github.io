9       /**
10       * The bitmask that indicates that a point lies to the left of this
11       * <code>Rectangle2D</code>.
12       * 
13       * @since 1.2
14       */
15      public static final int OUT_LEFT = 1;
16  
17      /**
18       * The bitmask that indicates that a point lies above this
19       * <code>Rectangle2D</code>.
20       * 
21       * @since 1.2
22       */
23      public static final int OUT_TOP = 2;
24  
25      /**
26       * The bitmask that indicates that a point lies to the right of this
27       * <code>Rectangle2D</code>.
28       * 
29       * @since 1.2
30       */
31      public static final int OUT_RIGHT = 4;
32  
33      /**
34       * The bitmask that indicates that a point lies below this
35       * <code>Rectangle2D</code>.
36       * 
37       * @since 1.2
38       */
39      public static final int OUT_BOTTOM = 8;

41      public static boolean rectangleIntersectsLine(double rectX, double rectY, double rectWidth, double rectHeight,
42              double x1, double y1, double x2, double y2) {
43          return _rectangleIntersectsLine(rectX, rectY, rectWidth, rectHeight, x1, y1, x2, y2)
44                  || Line2D.linesIntersect(rectX, rectY, rectX + rectWidth, rectY + rectHeight, x1, y1, x2, y2);
45      }
46      
47      private static boolean _rectangleIntersectsLine(double rectX, double rectY, double rectWidth,
48              double rectHeight, double x1, double y1, double x2, double y2) {
49          if (rectangleCornerOnSegment(rectX, rectY, rectWidth, rectHeight, x1, y1, x2, y2)) {
50              return true;
51          }
52          int out1, out2;
53          if ((out2 = outcode(rectX, rectY, rectWidth, rectHeight, x2, y2)) == 0) {
54              return true;
55          }
56          while ((out1 = outcode(rectX, rectY, rectWidth, rectHeight, x1, y1)) != 0) {
57              if ((out1 & out2) != 0) {
58                  return false;
59              }
60              if ((out1 & (OUT_LEFT | OUT_RIGHT)) != 0) {
61                  double x = rectX;
62                  if ((out1 & OUT_RIGHT) != 0) {
63                      x += rectWidth;
64                  }
65                  y1 = y1 + (x - x1) * (y2 - y1) / (x2 - x1);
66                  x1 = x;
67              } else {
68                  double y = rectY;
69                  if ((out1 & OUT_BOTTOM) != 0) {
70                      y += rectHeight;
71                  }
72                  x1 = x1 + (y - y1) * (x2 - x1) / (y2 - y1);
73                  y1 = y;
74              }
75          }
76          return true;
77      }
78  
79      private static boolean rectangleCornerOnSegment(double rectX, double rectY, double rectWidth,
80              double rectHeight, double x1, double y1, double x2, double y2) {
81          if (pointOnSegment(rectX, rectY, x1, y1, x2, y2)) {
82              return true;
83          } else if (pointOnSegment(rectX + rectWidth, rectY, x1, y1, x2, y2)) {
84              return true;
85          } else if (pointOnSegment(rectX, rectY + rectHeight, x1, y1, x2, y2)) {
86              return true;
87          } else if (pointOnSegment(rectX + rectWidth, rectY + rectHeight, x1, y1, x2, y2)) {
88              return true;
89          } else {
90              return false;
91          }
92      }
93  
94      private static final double PRECISION = 0.00000001;
95  
96      private static boolean pointOnSegment(double x, double y, double x1, double y1, double x2,
97              double y2) {
98          if (x < x1 || x > x2 || y < y1 || y > y2) {
99              return false;
100         } else {
101             double v = (y2 - y1) * (x - x1) - (x2 - x1) * (y - y1);
102             return Math.abs(v) < PRECISION;
103         }
104     }
105 
106     private static int outcode(double rectX, double rectY, double rectWidth, double rectHeight,
107             double x, double y) {
108         int out = 0;
109         if (rectWidth <= 0) {
110             out |= OUT_LEFT | OUT_RIGHT;
111         } else if (x < rectX) {
112             out |= OUT_LEFT;
113         } else if (x > rectX + rectWidth) {
114             out |= OUT_RIGHT;
115         }
116         if (rectHeight <= 0) {
117             out |= OUT_TOP | OUT_BOTTOM;
118         } else if (y < rectY) {
119             out |= OUT_TOP;
120         } else if (y > rectY + rectHeight) {
121             out |= OUT_BOTTOM;
122         }
123         return out;
124     }
