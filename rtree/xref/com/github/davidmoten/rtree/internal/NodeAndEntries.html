1   package com.github.davidmoten.rtree.internal;
2   
3   import java.util.List;
4   import java.util.Optional;
5   
6   import com.github.davidmoten.rtree.Entry;
7   import com.github.davidmoten.rtree.Node;
8   import com.github.davidmoten.rtree.geometry.Geometry;
9   
10  /**
11   * Used for tracking deletions through recursive calls.
12   * 
13   * @param <T>
14   *            value type
15   * @param <S> geometry type
16   */
17  public final class NodeAndEntries<T, S extends Geometry> {
18  
19      private final Optional<? extends Node<T, S>> node;
20      private final List<Entry<T, S>> entries;
21      private final int count;
22  
23      /**
24       * Constructor.
25       * 
26       * @param node
27       *            absent = whole node was deleted present = either an unchanged
28       *            node because of no removal or the newly created node without
29       *            the deleted entry
30       * @param entries
31       *            from nodes that dropped below minChildren in size and thus
32       *            their entries are to be redistributed (readded to the tree)
33       * @param countDeleted
34       *            count of the number of entries removed
35       */
36      public NodeAndEntries(Optional<? extends Node<T, S>> node, List<Entry<T, S>> entries,
37              int countDeleted) {
38          this.node = node;
39          this.entries = entries;
40          this.count = countDeleted;
41      }
42  
43      public Optional<? extends Node<T, S>> node() {
44          return node;
45      }
46  
47      public List<Entry<T, S>> entriesToAdd() {
48          return entries;
49      }
50  
51      public int countDeleted() {
52          return count;
53      }
54  
55  }
