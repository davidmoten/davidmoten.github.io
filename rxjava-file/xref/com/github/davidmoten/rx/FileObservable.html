27      /**
28       * Returns an {@link Observable} that uses NIO WatchService (and a dedicated
29       * thread) to push modify events to an observable that reads and reports new
30       * sequences of bytes to a subscriber. The NIO WatchService events are
31       * sampled according to <code>sampleTimeMs</code> so that lots of discrete
32       * activity on a file (for example a log file with very frequent entries)
33       * does not prompt an inordinate number of file reads to pick up changes.
34       * 
35       * @param file
36       *            the file to tail
37       * @param startPosition
38       *            start tailing file at position in bytes
39       * @param sampleTimeMs
40       *            sample time in millis
41       * @return
42       */
43      public final static Observable<byte[]> tailFile(File file, long startPosition, long sampleTimeMs) {
44          return from(file, StandardWatchEventKinds.ENTRY_CREATE, StandardWatchEventKinds.ENTRY_MODIFY,
45                  StandardWatchEventKinds.OVERFLOW)
46          // don't care about the event details, just that there is one
47                  .cast(Object.class)
48                  // get lines once on subscription so we tail the lines
49                  // in the
50                  // file at startup
51                  .startWith(new Object())
52                  // emit a max of 1 event per sample period
53                  .sample(sampleTimeMs, TimeUnit.MILLISECONDS)
54                  // tail file triggered by events
55                  .lift(new OperatorFileTailer(file, startPosition));
56      }

58      /**
59       * Returns an {@link Observable} that uses NIO WatchService (and a dedicated
60       * thread) to push modify events to an observable that reads and reports new
61       * lines to a subscriber. The NIO WatchService events are sampled according
62       * to <code>sampleTimeMs</code> so that lots of discrete activity on a file
63       * (for example a log file with very frequent entries) does not prompt an
64       * inordinate number of file reads to pick up changes.
65       * 
66       * @param file
67       *            the file to tail
68       * @param startPosition
69       *            start tailing file at position in bytes
70       * @param sampleTimeMs
71       *            sample time in millis
72       * @param charset
73       *            the character set to use to decode the bytes to a string
74       * @return
75       */
76      public final static Observable<String> tailTextFile(File file, long startPosition, long sampleTimeMs,
77              Charset charset) {
78          return toLines(tailFile(file, startPosition, sampleTimeMs), charset);
79      }

81      /**
82       * Returns an {@link Observable} of String that uses the given events stream
83       * to trigger checks on file change so that new lines can be read and
84       * emitted.
85       * 
86       * @param file
87       *            the file to tail
88       * @param startPosition
89       *            start tailing file at position in bytes
90       * @param charset
91       *            the character set to use to decode the bytes to a string
92       * @param events
93       *            trigger a check for file changes. Use
94       *            {@link Observable#interval(long, TimeUnit)} for example.
95       * @return
96       */
97      public final static Observable<String> tailTextFile(File file, long startPosition, Charset charset,
98              Observable<?> events) {
99          return toLines(events.lift(new OperatorFileTailer(file, startPosition)), charset);
100     }

102     /**
103      * Returns an {@link Observable} of {@link WatchEvent}s from a
104      * {@link WatchService}.
105      * 
106      * @param watchService
107      *            {@link WatchService} to generate events for
108      * @return
109      */
110     public final static Observable<WatchEvent<?>> from(WatchService watchService) {
111         return Observable.from(watchService).lift(new OperatorWatchServiceEvents());
112     }

114     /**
115      * If file does not exist at subscribe time then is assumed to not be a
116      * directory. If the file is not a directory (bearing in mind the aforesaid
117      * assumption) then a {@link WatchService} is set up on its parent and
118      * {@link WatchEvent}s of the given kinds are filtered to concern the file
119      * in question. If the file is a directory then a {@link WatchService} is
120      * set up on the directory and all events are passed through of the given
121      * kinds.
122      * 
123      * @param file
124      *            file to watch
125      * @param kinds
126      *            event kinds to watch for and emit
127      * @return
128      */
129     @SafeVarargs
130     public final static Observable<WatchEvent<?>> from(final File file, Kind<?>... kinds) {
131         return watchService(file, kinds)
132         // emit events from the WatchService
133                 .flatMap(TO_WATCH_EVENTS)
134                 // restrict to events related to the file
135                 .filter(onlyRelatedTo(file));
136     }
137 
138     /**
139      * Creates a {@link WatchService} on subscribe for the given file and event
140      * kinds.
141      * 
142      * @param file
143      *            the file to watch
144      * @param kinds
145      *            event kinds to watch for
146      * @return
147      */
148     @SafeVarargs
149     public final static Observable<WatchService> watchService(final File file, final Kind<?>... kinds) {
150         return Observable.create(new OnSubscribe<WatchService>() {
151 
152             @Override
153             public void call(Subscriber<? super WatchService> subscriber) {
154                 try {
155                     final Path path = getBasePath(file);
156                     WatchService watchService = path.getFileSystem().newWatchService();
157                     path.register(watchService, kinds);
158                     subscriber.onNext(watchService);
159                     subscriber.onCompleted();
160                 } catch (Exception e) {
161                     subscriber.onError(e);
162                 }
163             }
164 
165         });
166     }
167 
168     private final static Path getBasePath(final File file) {
169         final Path path;
170         if (file.exists() && file.isDirectory())
171             path = Paths.get(file.toURI());
172         else
173             path = Paths.get(file.getParentFile().toURI());
174         return path;
175     }
176 
177     /**
178      * Returns true if and only if the path corresponding to a WatchEvent
179      * represents the given file. This will be the case for Create, Modify,
180      * Delete events.
181      * 
182      * @param file
183      *            the file to restrict events to
184      * @return
185      */
186     private final static Func1<WatchEvent<?>, Boolean> onlyRelatedTo(final File file) {
187         return new Func1<WatchEvent<?>, Boolean>() {
188 
189             @Override
190             public Boolean call(WatchEvent<?> event) {
191 
192                 final boolean ok;
193                 if (file.isDirectory())
194                     ok = true;
195                 else if (StandardWatchEventKinds.OVERFLOW.equals(event.kind()))
196                     // TODO allow overflow events through?
197                     ok = true;
198                 else {
199                     Object context = event.context();
200                     if (context != null && context instanceof Path) {
201                         Path p = (Path) context;
202                         Path basePath = getBasePath(file);
203                         File pFile = new File(basePath.toFile(), p.toString());
204                         ok = pFile.getAbsolutePath().equals(file.getAbsolutePath());
205                     } else
206                         ok = false;
207                 }
208                 return ok;
209             }
210         };
211     }

213     private static Observable<String> toLines(Observable<byte[]> bytes, Charset charset) {
214         return StringObservable.split(StringObservable.decode(bytes, charset), "\n");
215     }
216 
217     private final static Func1<WatchService, Observable<WatchEvent<?>>> TO_WATCH_EVENTS = new Func1<WatchService, Observable<WatchEvent<?>>>() {
218 
219         @Override
220         public Observable<WatchEvent<?>> call(WatchService watchService) {
221             return from(watchService);
222         }
223     };
