438     private static Serializer<Character> createCharacterSerializer() {
439         Serializer<Character> serializer = new Serializer<Character>() {
440 
441             @Override
442             public Reader<Character> createReader(InputStream in) {
443                 return new Reader<Character>() {
444 
445                     java.io.Reader r = new InputStreamReader(in, StandardCharsets.UTF_8);
446 
447                     @Override
448                     public Character read() throws IOException {
449                         int c = r.read();
450                         if (c == -1) {
451                             return null;
452                         } else {
453                             return Character.valueOf((char) c);
454                         }
455                     }
456 
457                     @Override
458                     public void close() throws IOException {
459                         r.close();
460                     }
461 
462                 };
463             }
464 
465             @Override
466             public Writer<Character> createWriter(OutputStream out) {
467                 return new Writer<Character>() {
468 
469                     java.io.Writer w = new OutputStreamWriter(out, StandardCharsets.UTF_8);
470 
471                     @Override
472                     public void write(Character value) throws IOException {
473                         w.write((int) value.charValue());
474                     }
475 
476                     @Override
477                     public void close() throws IOException {
478                         w.close();
479                     }
480 
481                     @Override
482                     public void flush() throws IOException {
483                         w.flush();
484                     }
485                 };
486             }
487 
488         };
489         return serializer;
490     }

492     @Test
493     public void testBig() throws IOException {
494         final long N = Long.parseLong(System.getProperty("n", "1000000"));
495         for (long n = 1000; n <= N; n = n * 10) {
496             File input = new File("target/large");
497 
498             Serializer<Integer> serializer = new DataSerializer<Integer>() {
499 
500                 @Override
501                 public Integer read(DataInputStream dis) throws IOException {
502                     try {
503                         return dis.readInt();
504                     } catch (EOFException e) {
505                         return null;
506                     }
507                 }
508 
509                 @Override
510                 public void write(DataOutputStream dos, Integer value) throws IOException {
511                     dos.writeInt(value);
512                 }
513             };
514             long total = 0;
515             try (OutputStream out = new BufferedOutputStream(new FileOutputStream(input))) {
516                 Writer<Integer> writer = serializer.createWriter(out);
517                 SecureRandom r = new SecureRandom();
518                 for (int i = 0; i < n; i++) {
519                     int v = r.nextInt(1000);
520                     total += v;
521                     writer.write(v);
522                 }
523             }
524             long t = System.currentTimeMillis();
525             Sorter //
526                     .serializer(serializer) //
527                     .comparator((x, y) -> Integer.compare(x, y)) //
528                     .input(input) //
529                     .output(OUTPUT) //
530                     .sort();
531             System.out.println(n + " integers sorted in " + (System.currentTimeMillis() - t) / 1000.0 + "s");
532             // ensure ascending
533             try (InputStream in = new BufferedInputStream(new FileInputStream(OUTPUT))) {
534                 Reader<Integer> reader = serializer.createReader(in);
535                 Integer v;
536                 int last = Integer.MIN_VALUE;
537                 int count = 0;
538                 long total2 = 0;
539                 while ((v = reader.read()) != null) {
540                     assertTrue(v >= last);
541                     total2 += v;
542                     last = v;
543                     count++;
544                 }
545                 assertEquals(n, count);
546                 assertEquals(total, total2);
547             }
548             input.delete();
549         }
550     }
