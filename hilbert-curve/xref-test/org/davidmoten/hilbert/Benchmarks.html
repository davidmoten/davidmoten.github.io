3   import org.openjdk.jmh.annotations.Benchmark;
4   import org.openjdk.jmh.infra.Blackhole;
5   
6   public class Benchmarks {
7   
8       private static final int BITS = 10;
9       private static final long N = Math.round(Math.pow(2, BITS)) - 1;
10      private static final HilbertCurve c = HilbertCurve.bits(BITS).dimensions(5);
11  
12      @Benchmark
13      public void roundTrip(Blackhole b) {
14          for (long i = 0; i < N; i++) {
15              long[] point = c.point(i);
16              b.consume(i == c.index(point).longValue());
17          }
18      }
19  
20      @Benchmark
21      public void point(Blackhole b) {
22          for (long i = 0; i < N; i++) {
23              b.consume(c.point(i));
24          }
25      }
26  
27  }
