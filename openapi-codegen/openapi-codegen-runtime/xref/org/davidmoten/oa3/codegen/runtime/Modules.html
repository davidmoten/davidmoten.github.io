6   import com.fasterxml.jackson.databind.Module;

9   public final class Modules {
10  
11      public static final Module STRICT_DESERIALIZERS = createStrictDeserializersModule();
12  
13      private static Module createStrictDeserializersModule() {
14          SimpleModule m = new SimpleModule();
15          m.addDeserializer(Boolean.class, new StrictBooleanDeserializer());
16          m.addDeserializer(Short.class, new StrictShortDeserializer());
17          m.addDeserializer(Integer.class, new StrictIntegerDeserializer());
18          m.addDeserializer(Long.class, new StrictLongDeserializer());
19          m.addDeserializer(Float.class, new StrictFloatDeserializer());
20          m.addDeserializer(Double.class, new StrictDoubleDeserializer());
21          m.addDeserializer(OffsetDateTime.class, new StrictOffsetDateTimeDeserializer());
22          m.addDeserializer(LocalDate.class, new StrictLocalDateDeserializer());
23          m.addDeserializer(String.class, new StrictStringDeserializer());
24          return m;
25      }
