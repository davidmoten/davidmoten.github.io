3   import java.util.Map;
4   
5   import com.fasterxml.jackson.core.JsonProcessingException;
6   import com.fasterxml.jackson.databind.ObjectMapper;
7   
8   public final class RuntimeUtil {
9       
10      private static final ObjectMapper MAPPER = new ObjectMapper();
11      
12      private RuntimeUtil() {
13          // prevent instantiation
14      }
15      
16      public static void checkCanSerialize(Config config, Object o) {
17          try {
18              config.mapper().writeValueAsString(o);
19          } catch (JsonProcessingException e) {
20              throw new IllegalArgumentException(e);
21          }
22      }
23      
24      @SuppressWarnings("unchecked")
25      public static Map<String, Object> toMap(String json) {
26          try {
27              return MAPPER.readValue(json, Map.class);
28          } catch (JsonProcessingException e) {
29              throw new RuntimeException(e);
30          }
31      }
32  }
