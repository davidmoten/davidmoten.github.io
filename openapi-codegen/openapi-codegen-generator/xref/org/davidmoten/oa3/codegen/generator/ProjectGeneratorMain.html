3   import java.io.File;
4   import java.io.IOException;
5   import java.nio.file.Files;
6   
7   public class ProjectGeneratorMain {
8   
9       public static void main(String[] args) throws IOException {
10          String groupId = System.getProperty("groupId", "com.demo");
11          String artifactId = System.getProperty("artifactId", "client");
12          String basePackage = System.getProperty("basePackage", "com.demo.api");
13          String version = System.getProperty("version", "1.0");
14          String file = System.getProperty("file", "openapi.yaml");
15          String output = System.getProperty("output");
16          if (output == null) {
17              output = Files.createTempDirectory("client").toFile().getCanonicalPath();
18          }
19          File out = new File(output);
20          out.mkdirs();
21          boolean clean = System.getProperty("clean", "false").equals("true");
22          if (clean) {
23              deleteContents(out);
24          }
25          ProjectGenerator.generate(file, groupId, artifactId, version, basePackage, true, false, out);
26          System.out.println(output);
27      }
28  
29      private static void deleteContents(File file) {
30          if (file == null) {
31              return;
32          }
33          File[] list = file.listFiles();
34          if (list != null) {
35              for (File f : list) {
36                  if (f.isDirectory()) {
37                      deleteContents(f);
38                  }
39                  f.delete();
40              }
41          }
42      }
