42      public ClientServerGenerator(Definition definition) {
43          this.names = new Names(definition);
44          // make a ref map
45          Map<String, Cls> refCls = new HashMap<>();
46          Map<Schema<?>, Cls> schemaCls = new HashMap<>();
47          MyVisitor v = new MyVisitor(names);
48          Apis.visitSchemas(names.api(), v);
49          for (Result result : v.results()) {
50              Cls c = result.cls;
51              if (c.topLevel) {
52                  String refPrefix = c.category.refPrefix();
53                  String ref = refPrefix + c.name.get();
54                  refCls.put(ref, c);
55                  schemaCls.put(c.schema.get(), c);
56              }
57          }
58          this.refCls = refCls;
59          this.schemaCls = schemaCls;
60      }

72      private List<Method> collectMethods() {
73          // want a method per path, operation combo
74          List<Method> methods = new ArrayList<>();
75          names.api().getPaths().forEach((pathName, pathItem) -> {
76              gatherMethods(pathName, pathItem, methods);
77          });
78          return methods;
79      }

81      private void gatherMethods(String pathName, PathItem pathItem, List<Method> methods) {
82          pathItem = Apis.resolveRefs(names.api(), pathItem);
83          pathItem.readOperationsMap() //
84                  .forEach((method, operation) -> gatherMethods(pathName, method, operation, methods));
85      }

87      private void gatherMethods(String pathName, HttpMethod method, Operation operation, List<Method> methods) {
88          final String methodName;
89          if (operation.getOperationId() != null) {
90              methodName = Names.toIdentifier(operation.getOperationId());
91          } else {
92              methodName = Names.toIdentifier(ImmutableList.of(pathName, method.toString().toLowerCase(Locale.ENGLISH)));
93          }
94          Optional<Integer> statusCode = Optional.empty();
95          List<Param> params = new ArrayList<>();
96          Optional<String> returnFullClassName = Optional.empty();
97          List<String> consumes = new ArrayList<>();
98          List<String> produces = new ArrayList<>();
99          Set<String> parameterNames = new HashSet<>();
100         if (operation.getParameters() != null) {
101             operation.getParameters() //
102                     .forEach(p -> {
103                         // TODO handle refs to components/headers
104                         p = resolveParameterRefs(p);
105                         boolean isArray = false;
106                         Schema<?> s = p.getSchema();
107                         if (Util.isArray(s)) {
108                             isArray = true;
109                             s = s.getItems();
110                         }
111                         s = resolveRefs(s);
112                         Optional<Object> defaultValue = Optional.ofNullable(s.getDefault());
113                         String parameterName = Names.toIdentifier(p.getName());
114                         int i = 2;
115                         while (parameterNames.contains(parameterName)) {
116                             parameterName = Names.toIdentifier(p.getName()) + i;
117                             i++;
118                         }
119                         parameterNames.add(parameterName);
120                         final Param param;
121                         if (Util.isPrimitive(s)) {
122                             // handle simple schemas
123                             Class<?> c = Util.toClass(s.getType(), s.getFormat(), s.getExtensions(),
124                                     names.mapIntegerToBigInteger(), names.mapNumberToBigDecimal());
125                             param = new Param(p.getName(), parameterName, defaultValue, p.getRequired(),
126                                     c.getCanonicalName(), isArray, false, constraints(s),
127                                     ParamType.valueOf(p.getIn().toUpperCase(Locale.ENGLISH)), false,
128                                     Optional.ofNullable(p.getDescription()));
129                         } else {
130                             // is complex schema
131                             Cls cls = schemaCls.get(s);
132                             param = new Param(p.getName(), parameterName, defaultValue, p.getRequired(),
133                                     cls.fullClassName, isArray, false, constraints(s),
134                                     ParamType.valueOf(p.getIn().toUpperCase(Locale.ENGLISH)), true,
135                                     Optional.ofNullable(p.getDescription()));
136                         }
137                         params.add(param);
138                     });
139         }
140         if (operation.getRequestBody() != null) {
141             RequestBody b = resolveRefs(operation.getRequestBody());
142             MediaType mediaType = mediaType(b.getContent(), "application/json");
143             if (mediaType == null) {
144                 mediaType = mediaType(b.getContent(), "application/xml");
145             }
146             if (mediaType != null) {
147                 Schema<?> schema = mediaType.getSchema();
148                 if (schema != null) {
149                     if (schemaCls.get(schema) != null) {
150                         String fullClassName = resolveRefsFullClassName(schema);
151                         params.add(new Param("requestBody", "requestBody",
152                                 Optional.ofNullable((Object) schema.getDefault()),
153                                 org.davidmoten.oa3.codegen.util.Util.orElse(b.getRequired(), true), fullClassName,
154                                 false, true, constraints(schema), ParamType.BODY, false,
155                                 Optional.ofNullable(schema.getDescription())));
156                     } else {
157                         throw new RuntimeException("unexpected");
158                     }
159                 } else {
160                     // if no schema specified then is octet-stream
161                     String fullClassName = byte[].class.getCanonicalName();
162                     params.add(new Param("requestBody", "requestBody", Optional.empty(),
163                             org.davidmoten.oa3.codegen.util.Util.orElse(b.getRequired(), true), fullClassName, false,
164                             true, Constraints.empty(), ParamType.BODY, false,
165                             Optional.empty()));
166                 }
167             } else {
168                 System.out.println("TODO handle request body with media types " + b.getContent().keySet());
169             }
170             consumes = new ArrayList<>(b.getContent().keySet());
171         }
172         Optional<StatusCodeApiResponse> response = primaryResponse(operation.getResponses());
173         Optional<Integer> primaryStatusCode = response.map(x -> x.statusCode);
174         Mutable<String> primaryMimeType = Mutable.create(null);
175         if (response.isPresent() && response.get().response != null) {
176             Content content = resolveResponseRefs(response.get().response).getContent();
177             // if content is null then their is no response body
178             if (content != null) {
179                 primaryMimeType.value = "application/json";
180                 MediaType mediaType = mediaType(content, "application/json");
181                 if (mediaType == null) {
182                     primaryMimeType.value = "application/xml";
183                     mediaType = mediaType(content, primaryMimeType.value);
184                 }
185                 if (mediaType != null) {
186                     if (mediaType.getSchema() == null) {
187                         // Any schema
188                         returnFullClassName = Optional.of(Object.class.getCanonicalName());
189                     } else {
190                         returnFullClassName = Optional.of(resolveRefsFullClassName(mediaType.getSchema()));
191                     }
192                 } else {
193                     // loop through all mime-types and pick first non-default to infer return class
194                     // name
195                     final String defaultReturnClassFullName = Resource.class.getCanonicalName();
196                     returnFullClassName = content. //
197                             keySet() //
198                             .stream() //
199                             .filter(x -> !"default".equals(x)) //
200                             .map(x -> {
201                                 primaryMimeType.value = x;
202                                 if (x.startsWith("text/")) {
203                                     return String.class.getCanonicalName();
204                                 } else {
205                                     return defaultReturnClassFullName;
206                                 }
207                             }) //
208                             .findFirst();
209                     if (!returnFullClassName.isPresent()) {
210                         primaryMimeType.value = "default";
211                         MediaType a = content.get(primaryMimeType.value);
212                         if (a == null) {
213                             returnFullClassName = Optional.empty();
214                         } else {
215                             returnFullClassName = Optional.of(resolveRefsFullClassName(a.getSchema()));
216                         }
217                     }
218                     if (!returnFullClassName.isPresent()) {
219                         primaryMimeType.value = null;
220                     }
221                 }
222                 statusCode = Optional.of(response.get().statusCode);
223                 produces = new ArrayList<>(content.keySet());
224             }
225         }
226 
227         List<ResponseDescriptor> responseDescriptors = responseDescriptors(operation);
228 
229         Method m = new Method(methodName, statusCode, params, returnFullClassName, pathName, method, consumes, produces,
230                 Optional.ofNullable(operation.getDescription()), primaryStatusCode,
231                 Optional.ofNullable(primaryMimeType.value), responseDescriptors);
232         methods.add(m);
233     }

235     private MediaType mediaType(Content content, String mimeType) {
236         return content.entrySet().stream().filter(x -> x.getKey().replaceAll(";.*", "").equalsIgnoreCase(mimeType))
237                 .map(x -> x.getValue()) //
238                 .findFirst() //
239                 .orElse(null);
240     }

242     private List<ResponseDescriptor> responseDescriptors(Operation operation) {
243         List<ResponseDescriptor> list = new ArrayList<>();
244         operation.getResponses().forEach((statusCode, response) -> {
245             response = resolveResponseRefs(response);
246             if (response.getContent() != null) {
247                 response.getContent().forEach((contentType, mediaType) -> {
248                     final String fullClassName;
249                     if (mediaType.getSchema() == null) {
250                         fullClassName = byte[].class.getCanonicalName();
251                     } else {
252                         fullClassName = resolveRefsFullClassName(mediaType.getSchema());
253                     }
254                     list.add(new ResponseDescriptor(statusCode, contentType, fullClassName));
255                 });
256             }
257         });
258         return list;
259     }

261     private RequestBody resolveRefs(RequestBody b) {
262         while (b.get$ref() != null) {
263             b = names.lookupRequestBody(b.get$ref());
264         }
265         return b;

268     private static Constraints constraints(Schema<?> schema) {
269         return new Constraints(Optional.ofNullable(schema.getMinLength()), Optional.ofNullable(schema.getMaxLength()),
270                 Optional.ofNullable(schema.getMinimum()), Optional.ofNullable(schema.getMaximum()),
271                 Optional.ofNullable(schema.getExclusiveMinimumValue()),
272                 Optional.ofNullable(schema.getExclusiveMaximumValue()), Optional.ofNullable(schema.getMinItems()),
273                 Optional.ofNullable(schema.getMaxItems()), Optional.ofNullable(schema.getPattern()));

276     private static Optional<StatusCodeApiResponse> primaryResponse(ApiResponses responses) {
277         if (responses.get("200") != null) {
278             return Optional.of(new StatusCodeApiResponse(200, responses.get("200")));
279         } else {
280             for (Entry<String, ApiResponse> r : responses.entrySet()) {
281                 if (is2XX(r.getKey())) {
282                     return Optional.of(new StatusCodeApiResponse(Integer.parseInt(r.getKey()), r.getValue()));
283                 }
284             }
285         }
286         return Optional.empty();
287     }

300     private static boolean is2XX(String key) {
301         return key.length() == 3 && key.startsWith("2");
302     }

304     private Parameter resolveParameterRefs(Parameter p) {
305         while (p.get$ref() != null) {
306             p = names.lookupParameter(p.get$ref());
307         }
308         return p;
309     }
310 
311     private ApiResponse resolveResponseRefs(ApiResponse r) {
312         while (r.get$ref() != null) {
313             String ref = r.get$ref();
314             r = names.lookupResponse(ref);
315             if (r == null) {
316                 throw new RuntimeException("could not find response " + ref);
317             }
318         }
319         return r;
320     }

322     private String resolveRefsFullClassName(Schema<?> schema) {
323         return schemaCls.get(resolveRefs(schema)).fullClassName;
324     }
325 
326     private Schema<?> resolveRefs(Schema<?> schema) {
327         Schema<?> s = schema;
328         while (s.get$ref() != null) {
329             s = refCls.get(s.get$ref()).schema.get();
330         }
331         return s;
332     }
