45      public ClientServerGenerator(Definition definition) {
46          this.names = new Names(definition);
47          // make a ref map
48          Map<String, Cls> refCls = new HashMap<>();
49          Map<Schema<?>, Cls> schemaCls = new HashMap<>();
50          MyVisitor v = new MyVisitor(names);
51          Apis.visitSchemas(names.api(), v);
52          for (Result result : v.results()) {
53              Cls c = result.cls;
54              if (c.topLevel) {
55                  String refPrefix = c.category.refPrefix();
56                  String ref = refPrefix + c.name.get();
57                  refCls.put(ref, c);
58                  schemaCls.put(c.schema.get(), c);
59              }
60          }
61          this.refCls = refCls;
62          this.schemaCls = schemaCls;
63      }

75      private List<Method> collectMethods() {
76          // want a method per path, operation combo
77          List<Method> methods = new ArrayList<>();
78          if (names.api().getPaths() != null) {
79              names.api().getPaths().forEach((pathName, pathItem) -> {
80                  gatherMethods(pathName, pathItem, methods);
81              });
82          }
83          return methods;
84      }

86      private void gatherMethods(String pathName, PathItem pathItem, List<Method> methods) {
87          pathItem = Apis.resolveRefs(names.api(), pathItem);
88          pathItem.readOperationsMap() //
89                  .forEach((method, operation) -> gatherMethods(pathName, method, operation, methods));
90      }

92      private void gatherMethods(String pathName, HttpMethod method, Operation operation, List<Method> methods) {
93          final String methodName;
94          if (operation.getOperationId() != null) {
95              methodName = Names.toIdentifier(operation.getOperationId());
96          } else {
97              methodName = Names.toIdentifier(ImmutableList.of(pathName, method.toString().toLowerCase(Locale.ENGLISH)));
98          }
99          Optional<String> statusCode = Optional.empty();
100         List<Param> params = new ArrayList<>();
101         Optional<String> returnFullClassName = Optional.empty();
102         List<String> consumes = new ArrayList<>();
103         List<String> produces = new ArrayList<>();
104         Set<String> parameterNames = new HashSet<>();
105         if (operation.getParameters() != null) {
106             operation.getParameters() //
107                     .forEach(p -> {
108                         // TODO handle refs to components/headers
109                         p = resolveParameterRefs(p);
110                         boolean isArray = false;
111                         Schema<?> s = p.getSchema();
112                         if (s == null) {
113                             // TODO do something with p.getContent() instead of schema
114                             System.out.println("[WARN] parameter not without schema not supported yet, skipping:\n" + p);
115                             return;
116                         }
117                         final Schema<?> resolvedOriginal = resolveRefs(s);
118                         if (Util.isArray(s)) {
119                             isArray = true;
120                             s = s.getItems();
121                         } 
122                         s = resolveRefs(s);
123                         Optional<Object> defaultValue = Optional.ofNullable(s.getDefault());
124                         String parameterName = Names.toIdentifier(p.getName());
125                         int i = 2;
126                         while (parameterNames.contains(parameterName)) {
127                             parameterName = Names.toIdentifier(p.getName()) + i;
128                             i++;
129                         }
130                         parameterNames.add(parameterName);
131                         final Param param;
132                         if (Util.isPrimitive(s)) {
133                             // handle simple schemas
134                             Class<?> c = Util.toClass(Util.getTypeOrThrow(s), s.getFormat(), s.getExtensions(),
135                                     names.mapIntegerToBigInteger(), names.mapNumberToBigDecimal());
136                             param = new Param(p.getName(), parameterName, defaultValue, p.getRequired(),
137                                     c.getCanonicalName(), isArray, false, constraints(s), toParamType(p), false,
138                                     Optional.ofNullable(p.getDescription()), Optional.empty(), Optional.empty());
139                         } else {
140                             // is complex schema
141                             Cls cls = schemaCls.get(resolvedOriginal);
142                             param = new Param(p.getName(), parameterName, defaultValue, p.getRequired(),
143                                     cls.fullClassName, isArray, false, constraints(s), toParamType(p), true,
144                                     Optional.ofNullable(p.getDescription()), Optional.empty(), Optional.empty());
145                         }
146                         params.add(param);
147                     });
148         }
149         if (operation.getRequestBody() != null) {
150             RequestBody b = resolveRefs(operation.getRequestBody());
151             MediaType mediaType = mediaType(b.getContent(), "application/json").map(Entry::getValue).orElse(null);
152             if (mediaType == null) {
153                 mediaType = mediaType(b.getContent(), "application/xml").map(Entry::getValue).orElse(null);
154             }
155             final boolean isMultipartFormData;
156             if (mediaType == null) {
157                 mediaType = mediaType(b.getContent(), "multipart/form-data").map(Entry::getValue).orElse(null);
158                 isMultipartFormData = mediaType != null;
159             } else {
160                 isMultipartFormData = false;
161             }
162             final boolean isFormUrlEncoded;
163             if (mediaType == null) {
164                 mediaType = mediaType(b.getContent(), "application/x-www-form-urlencoded").map(Entry::getValue).orElse(null);
165                 isFormUrlEncoded = mediaType != null;
166             } else {
167                 isFormUrlEncoded = false;
168             }
169             if (mediaType != null) {
170                 Schema<?> schema = mediaType.getSchema();
171                 if (schema != null) {
172                     if (schemaCls.get(schema) != null) {
173                         String fullClassName = resolveRefsFullClassName(schema);
174                         ParamType paramType;
175                         if (isMultipartFormData) {
176                             paramType = ParamType.MULTIPART_FORM_DATA;
177                         } else if (isFormUrlEncoded){
178                             paramType = ParamType.FORM_URLENCODED;
179                         } else {
180                             paramType = ParamType.BODY;
181                         }
182                         params.add(new Param("requestBody", "requestBody",
183                                 Optional.ofNullable((Object) schema.getDefault()),
184                                 org.davidmoten.oa3.codegen.util.Util.orElse(b.getRequired(), true), fullClassName,
185                                 false, true, constraints(schema), paramType, false,
186                                 Optional.ofNullable(schema.getDescription()), Optional.empty(), Optional.empty()));
187                     } else {
188                         throw new RuntimeException("unexpected");
189                     }
190                 } else {
191                     addRequestBodyOctetStreamParameter(params, b, Optional.empty());
192                 }
193             } else {
194                 System.out.println("TODO handle request body with media types " + b.getContent().keySet());
195             }
196             consumes = new ArrayList<>(b.getContent().keySet());
197         }
198         Optional<StatusCodeApiResponse> response = primaryResponse(operation.getResponses());
199         Optional<String> primaryStatusCode = response.map(x -> x.statusCode);
200         Mutable<String> primaryMimeType = Mutable.create(null);
201         if (response.isPresent() && response.get().response != null) {
202             Content content = resolveResponseRefs(response.get().response).getContent();
203             // if content is null then their is no response body
204             if (content != null) {
205                 primaryMimeType.value = "application/json";
206                 MediaType mediaType = mediaType(content, "application/json").map(Entry::getValue).orElse(null);
207                 if (mediaType == null) {
208                     primaryMimeType.value = "application/xml";
209                     mediaType = mediaType(content, primaryMimeType.value).map(Entry::getValue).orElse(null);
210                 }
211                 if (mediaType != null) {
212                     if (mediaType.getSchema() == null) {
213                         // Any schema
214                         returnFullClassName = Optional.of(Object.class.getCanonicalName());
215                     } else {
216                         returnFullClassName = Optional.of(resolveRefsFullClassName(mediaType.getSchema()));
217                     }
218                 } else {
219                     // loop through all mime-types and pick first non-default to infer return class
220                     // name
221                     final String defaultReturnClassFullName = byte[].class.getCanonicalName();
222                     returnFullClassName = content. //
223                             keySet() //
224                             .stream() //
225                             .filter(x -> !"default".equals(x)) //
226                             .map(x -> {
227                                 primaryMimeType.value = x;
228                                 if (x.startsWith("text/")) {
229                                     return String.class.getCanonicalName();
230                                 } else {
231                                     return defaultReturnClassFullName;
232                                 }
233                             }) //
234                             .findFirst();
235                     if (!returnFullClassName.isPresent()) {
236                         primaryMimeType.value = "default";
237                         MediaType a = content.get(primaryMimeType.value);
238                         if (a == null) {
239                             returnFullClassName = Optional.empty();
240                         } else {
241                             returnFullClassName = Optional.of(resolveRefsFullClassName(a.getSchema()));
242                         }
243                     }
244                     if (!returnFullClassName.isPresent()) {
245                         primaryMimeType.value = null;
246                     }
247                 }
248                 statusCode = Optional.of(response.get().statusCode);
249                 produces = new ArrayList<>(content.keySet());
250             }
251         }
252 
253         List<ResponseDescriptor> responseDescriptors = responseDescriptors(operation);
254         boolean includeForServerGeneration;
255         if (operation.getExtensions() != null) {
256             includeForServerGeneration = Boolean.TRUE.equals(
257                     operation.getExtensions().getOrDefault(ExtensionKeys.INCLUDE_FOR_SERVER_GENERATION, ""));
258         } else {
259             includeForServerGeneration = true;
260         }
261         Method m = new Method(methodName, statusCode, params, returnFullClassName, pathName, method, consumes, produces,
262                 Optional.ofNullable(operation.getDescription()), primaryStatusCode,
263                 Optional.ofNullable(primaryMimeType.value), responseDescriptors, includeForServerGeneration);
264         methods.add(m);
265     }

267     private static ParamType toParamType(Parameter p) {
268         if (p instanceof QueryParameter) {
269             return ParamType.QUERY;
270         } else if (p instanceof PathParameter) {
271             return ParamType.PATH;
272         } else if (p instanceof HeaderParameter) {
273             return ParamType.HEADER;
274         } else if (p instanceof CookieParameter) {
275             return ParamType.COOKIE;
276         } else {
277             return ParamType.valueOf(p.getIn().toUpperCase(Locale.ENGLISH));
278         }
279     }

281     private void addRequestBodyOctetStreamParameter(List<Param> params, RequestBody b, Optional<String> contentType) {
282         // if no schema specified then is octet-stream
283         String fullClassName = byte[].class.getCanonicalName();
284         params.add(new Param("requestBody", "requestBody", Optional.empty(),
285                 org.davidmoten.oa3.codegen.util.Util.orElse(b.getRequired(), true), fullClassName, false,
286                 true, Constraints.empty(), ParamType.BODY, false,
287                 Optional.empty(), contentType, Optional.empty()));
288     }

290     private Optional<Entry<String, MediaType>> mediaType(Content content, String mimeType) {
291         return content.entrySet() //
292                 .stream() //
293                 .filter(x -> x.getKey().replaceAll(";.*", "").equalsIgnoreCase(mimeType))
294                 .findFirst();
295     }

297     private List<ResponseDescriptor> responseDescriptors(Operation operation) {
298         List<ResponseDescriptor> list = new ArrayList<>();
299         operation.getResponses().forEach((statusCode, response) -> {
300             response = resolveResponseRefs(response);
301             if (response.getContent() != null) {
302                 response.getContent().forEach((contentType, mediaType) -> {
303                     final String fullClassName;
304                     if (mediaType.getSchema() == null) {
305                         fullClassName = byte[].class.getCanonicalName();
306                     } else {
307                         fullClassName = resolveRefsFullClassName(mediaType.getSchema());
308                     }
309                     list.add(new ResponseDescriptor(statusCode, contentType, fullClassName));
310                 });
311             }
312         });
313         return list;
314     }

331     private static Optional<StatusCodeApiResponse> primaryResponse(ApiResponses responses) {
332         if (responses.get("200") != null) {
333             return Optional.of(new StatusCodeApiResponse("200", responses.get("200")));
334         } else {
335             for (Entry<String, ApiResponse> r : responses.entrySet()) {
336                 if (is2XX(r.getKey())) {
337                     return Optional.of(new StatusCodeApiResponse(r.getKey(), r.getValue()));
338                 }
339             }
340         }
341         return Optional.empty();
342     }

359     private Parameter resolveParameterRefs(Parameter p) {
360         while (p.get$ref() != null) {
361             p = names.lookupParameter(p.get$ref());
362         }
363         return p;

366     private ApiResponse resolveResponseRefs(ApiResponse r) {
367         while (r.get$ref() != null) {
368             String ref = r.get$ref();
369             r = names.lookupResponse(ref);
370             if (r == null) {
371                 throw new RuntimeException("could not find response " + ref);
372             }
373         }
374         return r;
375     }

377     private String resolveRefsFullClassName(Schema<?> schema) {
378         return schemaCls.get(resolveRefs(schema)).fullClassName;
379     }
380 
381     private Schema<?> resolveRefs(Schema<?> schema) {
382         Schema<?> s = schema;
383         while (s.get$ref() != null) {
384             String ref = s.get$ref();
385             Cls c = refCls.get(ref);
386             if (c == null) {
387                 throw new IllegalArgumentException("Cls not found for schema=\n" + schema + "\nrefCls map keys=\n" + refCls.keySet());
388             }
389             s = c.schema.get();
390             if (s == null) {
391                 throw new IllegalArgumentException("$ref not found: " + ref);
392             }
393         }
394         return s;
395     }

431         public Optional<Integer> statusCodeFirstInRange() {
432             return statusCode //
433                     .map(x -> x.toUpperCase(Locale.ENGLISH)) //
434                     .map(x -> {
435                         if (x.endsWith("XX")) {
436                             return Integer.parseInt(x.substring(0, 1)) * 100;
437                         } else {
438                             return Integer.parseInt(x);
