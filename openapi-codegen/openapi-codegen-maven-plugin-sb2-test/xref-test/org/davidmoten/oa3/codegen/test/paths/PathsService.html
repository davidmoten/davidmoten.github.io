55      @Override
56      public Response1 responseMultiTypeGet(String accept, String username) throws ServiceException {
57          List<String> accepts = Arrays.asList(accept.split(", "));
58          if (accepts.stream().anyMatch(x -> MediaType.valueOf(x).isCompatibleWith(MediaType.APPLICATION_OCTET_STREAM))) {
59              byte[] bytes = "hello there".getBytes(StandardCharsets.UTF_8);
60              InputStream in = new ByteArrayInputStream(bytes);
61              InputStreamResource res = new InputStreamResource(in);
62              HttpHeaders headers = new HttpHeaders();
63              headers.put(HttpHeaders.CONTENT_TYPE, Arrays.asList(MediaType.APPLICATION_OCTET_STREAM_VALUE));
64              headers.put(HttpHeaders.CONTENT_LENGTH, Arrays.asList(bytes.length + ""));
65              return response(new ResponseEntity<>(res, headers, HttpStatus.OK));
66          } else if (accepts.stream().anyMatch(x -> MediaType.valueOf(x).isCompatibleWith(MediaType.APPLICATION_JSON))) {
67              return new Response1(username);
68          } else {
69              throw new ServiceException(HttpStatus.BAD_REQUEST.value(), "unsupported Accept header: " + accept);
70          }
71      }
