13      public static String encodeOctets(byte[] a) {
14          if (a == null) {
15              return "";
16          } else {
17              return new BigInteger(a).toString(16);
18          }
19      }
20      
21      public static String encodeOctets(Optional<byte[]> a) {
22          return a.map(Util::encodeOctets).orElse("");
23      }
24  
25      public static byte[] decodeOctets(String s) {
26          return new BigInteger(s, 16).toByteArray();
27      }
28  
29      public static <T> T orElse(T value, T defaultValue) {
30          return value == null ? defaultValue : value;
31      }
32  
33      public static String toString(Class<?> cls, Object... items) {
34          Preconditions.checkArgument(items.length % 2 == 0);
35          StringBuilder s = new StringBuilder();
36          int i = 0;
37          while (i < items.length) {
38              if (i > 0) {
39                  s.append(", ");
40              }
41              s.append(items[i]);
42              s.append("=");
43              s.append(items[i + 1]);
44              i += 2;
45          }
46          return cls.getSimpleName() + "[" + s + "]";
47      }
48  
49      public static byte[] read(InputStream in) throws IOException {
50          byte[] buffer = new byte[8192];
51          int n = 0;
52          ByteArrayOutputStream bytes = new ByteArrayOutputStream();
53          while ((n = in.read(buffer)) != -1) {
54              bytes.write(buffer, 0, n);
55          }
56          return bytes.toByteArray();
57      }
