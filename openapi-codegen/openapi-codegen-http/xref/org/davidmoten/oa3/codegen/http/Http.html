239     public static HttpResponse call(//
240             HttpMethod method, //
241             String basePath, //
242             String pathTemplate, //
243             Serializer serializer, //
244             List<Interceptor> interceptors, //
245             Headers requestHeaders, //
246             List<ParameterValue> parameters, //
247             // (statusCode, contentType, class)
248             List<ResponseDescriptor> descriptors, boolean allowPatch) {
249         return call(method, basePath, pathTemplate, serializer, interceptors, requestHeaders, parameters,
250                 (statusCode, contentType) -> match(descriptors, statusCode, contentType), allowPatch);
251     }
252 
253     private static Optional<Class<?>> match(List<ResponseDescriptor> descriptors, Integer statusCode,
254             String contentType) {
255         List<ResponseDescriptor> matches = new ArrayList<>();
256         for (ResponseDescriptor d : descriptors) {
257             if (d.matches(statusCode, contentType)) {
258                 matches.add(d);
259             }
260         }
261         Collections.sort(matches, ResponseDescriptor.specificity());
262         return matches.stream().findFirst().map(d -> d.cls());
263     }

265     private static HttpResponse call(//
266             HttpMethod method, //
267             String basePath, //
268             String pathTemplate, //
269             Serializer serializer, //
270             List<Interceptor> interceptors, //
271             Headers requestHeaders, //
272             List<ParameterValue> parameters, //
273             // (statusCode x contentType) -> class
274             BiFunction<? super Integer, ? super String, Optional<Class<?>>> responseCls, boolean allowPatch) {
275         String url = buildUrl(basePath, pathTemplate, parameters);
276         Optional<ParameterValue> requestBody = parameters.stream().filter(x -> x.type() == ParameterType.BODY)
277                 .findFirst();
278         try {
279             Headers headers = new Headers(requestHeaders);
280             final HttpMethod requestMethod;
281             if (!allowPatch && method.equals(HttpMethod.PATCH)) {
282                 headers.put("X-HTTP-Method-Override", HttpMethod.PATCH.name());
283                 requestMethod = HttpMethod.POST;
284             } else {
285                 requestMethod = method;
286             }
287             // modify request metadata (like insert auth related headers)
288             RequestBase r = new RequestBase(requestMethod, url, headers);
289             for (Interceptor interceptor : interceptors) {
290                 r = interceptor.intercept(r);
291             }
292             log.debug("connecting to method=" + r.method() + ", url=" + url + ", headers=" + r.headers());
293             return connectAndProcess(serializer, parameters, responseCls, r.url(), requestBody, r.headers(),
294                     r.method());
295         } catch (IOException e) {
296             throw new UncheckedIOException(e);
297         }
298     }

300     private static HttpResponse connectAndProcess(Serializer serializer, List<ParameterValue> parameters,
301             BiFunction<? super Integer, ? super String, Optional<Class<?>>> responseCls, String url,
302             Optional<ParameterValue> requestBody, Headers headers, final HttpMethod method)
303             throws IOException, MalformedURLException, ProtocolException, StreamReadException, DatabindException {
304         log.debug("Http.headers={}", headers);
305         HttpURLConnection con = (HttpURLConnection) new URL(url).openConnection();
306         con.setRequestMethod(method.name());
307         parameters.stream() //
308                 .filter(p -> p.type() == ParameterType.HEADER && p.value().isPresent()) //
309                 .forEach(p -> headers.put(p.name(), String.valueOf(p.value().get())));
310         // add request body content type (should just be one)
311         parameters.stream().filter(p -> p.contentType().isPresent())
312                 .forEach(p -> headers.put("Content-Type", p.contentType().get()));
313         headers.forEach((key, list) -> {
314             con.setRequestProperty(key, list.stream().collect(Collectors.joining(", ")));
315         });
316         con.setDoInput(true);
317         if (requestBody.isPresent()) {
318             con.setDoOutput(true);
319             Optional<?> body = requestBody.get().value();
320             if (body.isPresent()) {
321                 try (OutputStream out = con.getOutputStream()) {
322                     serializer.serialize(body.get(), requestBody.get().contentType().get(), out);
323                 }
324             }
325         }
326         int statusCode = con.getResponseCode();
327         Headers responseHeaders = Headers.create(con.getHeaderFields());
328         String responseContentType = Optional.ofNullable(con.getHeaderField("Content-Type"))
329                 .orElse("application/octet-stream");
330         Object data;
331         Optional<Class<?>> responseClass = responseCls.apply(statusCode, responseContentType);
332         try (InputStream in = log(con.getInputStream())) {
333             data = readResponse(serializer, responseClass, responseContentType, in);
334         } catch (IOException e) {
335             try (InputStream err = log(con.getErrorStream())) {
336                 data = readResponse(serializer, responseClass, responseContentType, err);
337             }
338         }
339         return new HttpResponse(statusCode, responseHeaders, Optional.of(data));
340     }

342     private static InputStream log(InputStream inputStream) {
343         if (!log.isDebugEnabled()) {
344             return inputStream;
345         }
346         ByteArrayOutputStream bytes = new ByteArrayOutputStream();
347         return new InputStream() {
348 
349             @Override
350             public int read() throws IOException {
351                 int v = inputStream.read();
352                 if (v == -1) {
353                     log.debug("Http.inputStream=\n{}", new String(bytes.toByteArray(), StandardCharsets.UTF_8));
354                 }
355                 bytes.write(v);
356                 return v;
357             }
358 
359         };

362     @VisibleForTesting
363     static String buildUrl(String basePath, String pathTemplate, List<ParameterValue> parameters) {
364         Preconditions.checkArgument(pathTemplate.startsWith("/"));
365         // substitute path parameters
366         String path = stripFinalSlash(basePath) + insertParametersIntoPath(pathTemplate, parameters);
367         // build query string
368         String queryString = parameters //
369                 .stream() //
370                 .filter(p -> p.type() == ParameterType.QUERY) //
371                 .filter(p -> p.value().isPresent()) //
372                 .map(p -> urlEncode(p.name()) + "=" + p.value().map(x -> valueToString(x)).orElse("")) //
373                 .collect(Collectors.joining("&"));
374         return path + "?" + queryString;
375     }

377     private static Object readResponse(Serializer serializer, Optional<Class<?>> responseType,
378             String responseContentType, InputStream in) throws IOException, StreamReadException, DatabindException {
379         if (responseType.isPresent()) {
380             return serializer.deserialize(responseType.get(), responseContentType, in);
381         } else {
382             return new String(Util.read(in), StandardCharsets.UTF_8);
383         }
384     }

386     private static String valueToString(Object value) {
387         if (value == null) {
388             return "";
389         } else if (value instanceof Collection) {
390             Collection<?> c = (Collection<?>) value;
391             return c.stream().map(x -> valueToString(x)).collect(Collectors.joining(","));
392         } else {
393             return urlEncode(value.toString());
394         }
395     }

397     private static String stripFinalSlash(String s) {
398         if (s.endsWith("/")) {
399             return s.substring(0, s.length() - 2);
400         } else {
401             return s;
402         }
403     }

405     private static String urlEncode(String s) {
406         try {
407             return URLEncoder.encode(s, "UTF-8").replaceAll("\\+", "%20");
408         } catch (UnsupportedEncodingException e) {
409             throw new RuntimeException(e);
410         }
411     }

413     private static String insertParametersIntoPath(String pathTemplate, List<ParameterValue> parameters) {
414         String s = pathTemplate;
415         for (ParameterValue p : parameters) {
416             if (p.type() == ParameterType.PATH) {
417                 s = insertParameter(s, p.name(), p.value().get());
418             }
419         }
420         return s;
421     }
422 
423     private static String insertParameter(String s, String name, Object object) {
424         return s.replace("{" + name + "}", urlEncode(object.toString()));
425     }
