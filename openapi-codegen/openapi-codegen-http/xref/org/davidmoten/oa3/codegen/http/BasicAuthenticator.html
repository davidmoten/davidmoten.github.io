1   package org.davidmoten.oa3.codegen.http;
2   
3   import java.nio.charset.StandardCharsets;
4   import java.util.Base64;
5   
6   public interface BasicAuthenticator extends Interceptor {
7   
8       String username();
9   
10      String password();
11  
12      default RequestBase intercept(RequestBase r) {
13          Headers h = new Headers(r.headers());
14          String encoded = Base64.getEncoder()
15                  .encodeToString((username() + ":" + password()).getBytes(StandardCharsets.UTF_8));
16          h.put("Authorization", "Basic " + encoded);
17          return new RequestBase(r.method(), r.url(), h);
18      }
19  
20  }
