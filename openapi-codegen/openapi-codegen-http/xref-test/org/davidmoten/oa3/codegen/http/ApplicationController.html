3   import org.springframework.http.HttpStatus;
4   import org.springframework.http.ResponseEntity;
5   import org.springframework.web.bind.annotation.RequestBody;
6   import org.springframework.web.bind.annotation.RequestMapping;
7   import org.springframework.web.bind.annotation.RequestMethod;
8   import org.springframework.web.bind.annotation.RequestParam;
9   import org.springframework.web.bind.annotation.RestController;
10  
11  @RestController
12  public class ApplicationController {
13  
14      @RequestMapping(method = RequestMethod.GET, value = "/thing", produces = { "application/json" })
15      public ResponseEntity<?> thingGet(@RequestParam(name = "id", required = true) String id) {
16          if (id.equals("a")) {
17              return ResponseEntity.ok(new Thing("janice", 34));
18          } else {
19              return ResponseEntity.status(HttpStatus.BAD_REQUEST)
20                      .body(new Problem("bad id", HttpStatus.BAD_REQUEST.value()));
21          }
22      }
23  
24      @RequestMapping(method = RequestMethod.POST, value = "/thing", produces = { "application/json" })
25      public ResponseEntity<?> thingPost(@RequestBody(required = true) Thing thing) {
26          if (thing.age == 20) {
27              return ResponseEntity.ok().body(thing);
28          } else {
29              return ResponseEntity.status(HttpStatus.BAD_REQUEST)
30                      .body(new Problem("bad age", HttpStatus.BAD_REQUEST.value()));
31          }
32  
33      }
34  }
