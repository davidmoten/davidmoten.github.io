3   import java.util.Locale;
4   import java.util.Set;
5   
6   import com.github.davidmoten.guavamini.Sets;
7   
8   public final class MediaType {
9   
10      private static final Set<String> BINARY_MEDIA_TYPES = Sets.newHashSet( //
11              "application/octet-stream", //
12              "application/pdf", //
13              "image/jpeg", //
14              "image/gif", //
15              "image/png");
16  
17      public static boolean isJson(String mediaType) {
18          return lc(mediaType).endsWith("/json") || lc(mediaType).endsWith("+json");
19      }
20  
21      public static boolean isText(String mediaType) {
22          return lc(mediaType).startsWith("text/");
23      }
24  
25      public static boolean isOctetStream(String mediaType) {
26          return BINARY_MEDIA_TYPES.contains(lc(mediaType));
27      }
28      
29      private static String lc(String s) {
30          return s.toLowerCase(Locale.ENGLISH);
31      }
32  
33  }
