3   import java.io.ByteArrayOutputStream;
4   import java.io.InputStream;
5   import java.io.OutputStream;
6   import java.util.Map;

8   import com.fasterxml.jackson.databind.JsonNode;

10  public interface Serializer {
11  
12      void serialize(Object o, String contentType, OutputStream out);
13  
14      <T> T deserialize(Class<T> cls, String contentType, InputStream in);
15      
16      /**
17       * Returns the properties of object. If has no properties or is not an object
18       * type that supports properties then returns an empty map. In the case of JSON
19       * the map value is of an imprecise type but will serialize to JSON in an 
20       * expected deterministic way using Jackson serialization (i.e the object could be 
21       * an annotated Jackson object or an instance of {@link JsonNode}).
22       * 
23       * @param o object to get properties from, if null then returns an empty map
24       * @param contentType content type of the object, cannot be null
25       */
26      Map<String, Object> properties(Object o, String contentType);
27      
28      default byte[] serialize(Object o, String contentType) {
29          ByteArrayOutputStream bytes = new ByteArrayOutputStream();
30          serialize(o, contentType, bytes);
31          return bytes.toByteArray();
32      }

34  }
