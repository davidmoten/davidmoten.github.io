1   package org.davidmoten.oa3.codegen.runtime;

3   import com.fasterxml.jackson.core.JacksonException;
4   import com.fasterxml.jackson.core.JsonParser;
5   import com.fasterxml.jackson.core.JsonToken;
6   import com.fasterxml.jackson.databind.DeserializationContext;
7   import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
8   import java.io.IOException;
9   
10  public final class StrictFloatDeserializer extends StdDeserializer<Float> {
11  
12      private static final long serialVersionUID = 2207323065789635630L;
13  
14      public StrictFloatDeserializer() {
15          super(Float.class);
16      }
17  
18      @Override
19      public Float deserialize(JsonParser p, DeserializationContext ctxt)
20              throws IOException, JacksonException {
21          JsonToken t = p.currentToken();
22          if (t == JsonToken.VALUE_NUMBER_INT) {
23              return p.getFloatValue();
24          } else if (t == JsonToken.VALUE_NUMBER_FLOAT) {
25              return p.getFloatValue();
26          } else {
27              return (Float) ctxt.handleUnexpectedToken(Float.class, p);
28          }
29      }
30  }
