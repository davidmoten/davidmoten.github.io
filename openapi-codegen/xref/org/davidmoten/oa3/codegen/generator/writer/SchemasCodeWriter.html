88      public static void writeSchemaClass(Names names, Map<String, Set<Cls>> fullClassNameInterfaces, Cls cls,
89              String schemaName) {
90          if ((cls.category == SchemaCategory.PATH || cls.category == SchemaCategory.RESPONSE) && cls.schema.isPresent()
91                  && cls.schema.get().get$ref() != null) {
92              // when a cls has a ref and is used with a Path or Response then the ref class
93              // is used in generated code
94              return;
95          }
96          CodePrintWriter out = CodePrintWriter.create(cls.fullClassName, names.simpleNameInPackage(cls.fullClassName));
97          writeClass(out, cls, fullClassNameInterfaces, names);
98          WriterUtil.writeContent(names, out);
99      }

123     private static void writeClass(CodePrintWriter out, Cls cls, Map<String, Set<Cls>> fullClassNameInterfaces,
124             Names names) {
125         if (cls.topLevel) {
126             out.line("package %s;", cls.pkg());
127             out.println();
128             out.format("%s", IMPORTS_HERE);
129         }
130         // reserve class names in Imports for member classes
131         reserveMemberClassNamesInImports(out.imports(), cls);
132         writeClassDeclaration(out, cls, fullClassNameInterfaces);
133         writeEnumMembers(out, cls);
134         if (isPolymorphic(cls)) {
135             writePolymorphicClassContent(out, cls, names, fullClassNameInterfaces);
136         } else {
137             writeFields(out, cls);
138             writeConstructor(out, cls, fullClassNameInterfaces, names);
139             writeGetters(out, cls, fullClassNameInterfaces);
140             writePropertiesMapGetter(out, cls);
141             writeMutators(out, cls, fullClassNameInterfaces);
142             writeBuilder(out, cls, fullClassNameInterfaces);
143         }
144         writeEnumCreator(out, cls);
145         writeEnumDeserializer(out, cls);
146         writeMemberClasses(out, cls, fullClassNameInterfaces, names);
147         if (cls.classType != ClassType.ENUM && cls.classType != ClassType.ONE_OR_ANY_OF_DISCRIMINATED) {
148             writeEqualsMethod(out, cls);
149             writeHashCodeMethod(out, cls);
150             writeToStringMethod(out, cls);
151         }
152         out.closeParen();
153     }

175     private static void writeEnumCreator(CodePrintWriter out, Cls cls) {
176         if (cls.classType == ClassType.ENUM) {
177             String simpleClassName = Names.simpleClassName(cls.fullClassName);
178             out.println();
179             out.line("@%s", JsonCreator.class);
180             out.line("public static %s fromValue(%s value) {", simpleClassName, Object.class);
181             out.line("for (%s x: %s.values()) {", simpleClassName, simpleClassName);
182             // be careful because x.value can be primitive
183             if (cls.isNullableEnum()) {
184                 out.line("if (%s.equals(value, x.value.get())) {", Objects.class);
185             } else {
186                 out.line("if (%s.equals(value, x.value)) {", Objects.class);
187             }
188             out.line("return x;");
189             out.closeParen();
190             out.closeParen();
191             out.line("throw new %s(\"unexpected enum value: '\" + value + \"'\");", IllegalArgumentException.class);
192             out.closeParen();
193         }

211     private static void writeClassDeclaration(CodePrintWriter out, Cls cls,
212             Map<String, Set<Cls>> fullClassNameInterfaces) {
213         String modifier = classModifier(cls);
214         Set<Cls> interfaces = fullClassNameInterfaces.get(cls.fullClassName);
215         String implementsClause = implementsClause(out.imports(), interfaces, cls);
216         // TODO ensure contentType() and value() methods of HasEncoding are annotated
217         // with @Override
218 
219         final boolean javadocExists;
220         if (cls.description.isPresent()) {
221             String html = WriterUtil.markdownToHtml(cls.description.get());
222             javadocExists = Javadoc.printJavadoc(out, out.indent(), html, true);
223         } else {
224             javadocExists = false;
225         }
226         if (!javadocExists) {
227             out.println();
228         }
229         if (cls.classType == ClassType.ONE_OR_ANY_OF_DISCRIMINATED) {
230             writeJsonTypeInfoAnnotation(out, cls);
231         } else if (cls.classType == ClassType.ONE_OF_NON_DISCRIMINATED || cls.classType == ClassType.ALL_OF) {
232             writePolymorphicDeserializerAnnotation(out, cls);
233         } else if (cls.classType == ClassType.ANY_OF_NON_DISCRIMINATED) {
234             writeAnyOfSerializerAnnotations(out, cls);
235         }
236         if (cls.classType != ClassType.ENUM && cls.classType != ClassType.ONE_OR_ANY_OF_DISCRIMINATED
237                 && cls.classType != ClassType.ANY_OF_NON_DISCRIMINATED) {
238             writeJsonIncludeAnnotation(out);
239             writeAutoDetectAnnotation(out);
240         }
241         if (cls.classType == ClassType.ENUM && cls.hasEnumNullValue()) {
242             writeEnumNullValueDeserializerAnnotation(out, cls);
243         }
244         if (cls.topLevel) {
245             WriterUtil.addGeneratedAnnotation(out);
246         }
247         out.line("public %s%s %s%s {", modifier, cls.classType.word(), cls.simpleName(), implementsClause);
248     }

263     private static String classModifier(Cls cls) {
264         final String modifier;
265         if (cls.classType == ClassType.ONE_OR_ANY_OF_DISCRIMINATED || cls.classType == ClassType.ENUM) {
266             modifier = "";
267         } else {

273     private static String implementsClause(Imports imports, Set<Cls> interfaces, Cls cls) {
274         interfaces = Util.orElse(interfaces, Collections.emptySet());
275         final String implemented;
276         if (interfaces.isEmpty() && !cls.hasEncoding()) {
277             implemented = "";
278         } else {
279             Stream<String> a = interfaces //
280                     .stream() //
281                     .map(x -> x.fullClassName);
282             Stream<String> b = Stream.of(HasEncoding.class.getCanonicalName()) //
283                     .filter(x -> cls.hasEncoding() && cls.classType != ClassType.ENUM);
284             // for use with ContentType class
285             Stream<String> c = Stream.of(HasStringValue.class.getCanonicalName()) //
286                     .filter(x -> cls.hasEncoding() && cls.classType == ClassType.ENUM);
287             implemented = " implements " + Stream.concat(a, Stream.concat(b, c)) //
288                     .map(imports::add) //
289                     .collect(Collectors.joining(", "));
290         }
291         return implemented;
292     }

294     private static void writeJsonTypeInfoAnnotation(CodePrintWriter out, Cls cls) {
295         out.line("@%s(use = %s.NAME, property = \"%s\", include = %s.EXISTING_PROPERTY, visible = true)",
296                 JsonTypeInfo.class, Id.class, cls.discriminator.propertyName, As.class);
297         out.right().right();
298         String types = cls.fields.stream().map(x -> {
299             final String fieldImportedType;
300             if (x.fullClassName.startsWith(cls.fullClassName)) {
301                 fieldImportedType = Names.simpleClassName(cls.fullClassName)
302                         + x.fullClassName.substring(cls.fullClassName.length());
303             } else {
304                 fieldImportedType = out.add(x.fullClassName);
305             }
306             return String.format("\n%s@%s.Type(value = %s.class, name = \"%s\")", out.indent(),
307                     out.add(JsonSubTypes.class), fieldImportedType,
308                     cls.discriminator.discriminatorValueFromFullClassName(x.fullClassName));
309         }).collect(Collectors.joining(", "));
310         out.left().left();
311         out.line("@%s({%s})", JsonSubTypes.class, types);
312     }

337     private static void writeEnumMembers(CodePrintWriter out, Cls cls) {
338         final String parameterFullClassName;
339         if (!cls.fields.isEmpty()) {
340             parameterFullClassName = cls.fields.get(0).fullClassName;
341         } else {
342             parameterFullClassName = "NotUsed";
343         }
344         int[] index = new int[] { -1 };
345         String text = cls.enumMembers.stream() //
346                 .map(x -> {
347                     index[0]++;
348                     final String memberName;
349                     if (!cls.enumNames.isEmpty()) {
350                         memberName = cls.enumNames.get(index[0]);
351                     } else {
352                         memberName = x.name;
353                     }
354                     if (x.parameter instanceof ObjectNode) {
355                         return String.format("%s%s(%s.toMap(\"%s\"))", out.indent(), memberName,
356                                 out.add(RuntimeUtil.class), escapedJson((ObjectNode) x.parameter));
357                     } else if (parameterFullClassName.equals(BigInteger.class.getCanonicalName())
358                             || parameterFullClassName.equals(BigDecimal.class.getCanonicalName())) {
359                         return String.format("%s%s(new %s(\"\"))", out.indent(), memberName,
360                                 out.add(parameterFullClassName));
361                     } else {
362                         String delim = x.parameter instanceof String //
363                                 || //
364                                 cls.enumValueFullType.equals(String.class.getCanonicalName()) //
365                                         && x.parameter instanceof Boolean ? "\"" : "";
366                         if (x.nullable) {
367                             if (x.parameter == null) {
368                                 return String.format("%s%s(%s.empty())", out.indent(), memberName,
369                                         out.add(Optional.class));
370                             } else {
371                                 return String.format("%s%s(%s.of(%s%s%s))", out.indent(), memberName,
372                                         out.add(Optional.class), delim, x.parameter, delim);
373                             }
374                         } else {
375                             return String.format("%s%s(%s%s%s)", out.indent(), memberName, delim, x.parameter, delim);
376                         }
377                     }
378                 }).collect(Collectors.joining(",\n"));
379         if (!text.isEmpty()) {
380             out.println("\n" + text + ";");
381         }
382     }

384     private static void writePolymorphicClassContent(CodePrintWriter out, Cls cls, Names names, Map<String, Set<Cls>> fullClassNameInterfaces) {
385         if (cls.classType == ClassType.ONE_OR_ANY_OF_DISCRIMINATED) {
386             out.println();
387             out.line("%s %s();", String.class, cls.discriminator.fieldName);
388         } else {
389             if (cls.classType == ClassType.ONE_OF_NON_DISCRIMINATED) {
390                 out.println();
391                 writeJsonValueAnnotation(out);
392                 out.line("private final %s %s;", Object.class, "value");
393 
394                 // add constructor for each member of the oneOf (fieldTypes)
395                 // as there are multiple constructors we cannot add ConstructorBinding
396                 // annotations so polymorphic stuff can't be used to bind to rest method
397                 // parameters
398                 writeOneOfAnyOfNonDiscriminatedObjectConstructor(out, cls);
399                 Set<String> used = new HashSet<>();
400                 cls.fields.forEach(f -> writeOneOfAnyOfNonDiscriminatedMemberSpecificConstructor(out, cls, f, used));
401                 out.println();
402                 writeGetter(out, out.add(Object.class), "value", "value");
403                 writeNonDiscriminatedFactoryMethod(out, cls, names, fullClassNameInterfaces);
404             } else if (cls.classType == ClassType.ANY_OF_NON_DISCRIMINATED) {
405                 writeFields(out, cls);
406 
407                 // write constructor
408                 out.right().right();
409                 final String parameters = cls.fields //
410                         .stream() ///
411                         .map(x -> String.format("\n%s%s %s", out.indent(), x.resolvedType(out.imports()),
412                                 x.fieldName(cls)))
413                         .collect(Collectors.joining(","));
414                 out.left().left();
415                 out.println();
416                 out.line("private %s(%s) {", Names.simpleClassName(cls.fullClassName), parameters);
417                 ifValidate(cls, out, names, //
418                         o -> cls.fields.forEach(x -> checkNotNull(cls, o, x)));
419                 cls.fields.forEach(x -> assignField(out, cls, x));
420                 out.closeParen();
421                 out.println();
422                 out.line("public static %s of(%s) {", cls.simpleName(), parameters);
423                 String fields = cls.fields.stream().map(x -> x.fieldName).collect(Collectors.joining(", "));
424                 out.line("%s $o = new %s(%s);", cls.simpleName(), cls.simpleName(), fields);
425                 out.line("%s.checkCanSerialize(%s.config(), $o);", //
426                         RuntimeUtil.class, //
427                         out.add(names.globalsFullClassName()));
428                 out.line("return $o;");
429                 out.closeParen();
430 
431                 // write getters
432                 cls.fields.forEach(f -> {
433                     out.println();
434                     writeGetter(out, f.resolvedType(out.imports()), f.fieldName(cls), f.fieldName(cls));
435                 });
436 
437                 writeAnyOfOrAllOfBuilder(out, cls, true);
438             } else if (cls.classType == ClassType.ALL_OF) {
439                 // allof
440                 writeFields(out, cls);
441 
442                 out.right().right();
443                 final String parameters = cls //
444                         .fields //
445                         .stream() //
446                         .map(x -> String.format("\n%s%s %s", out.indent(), x.resolvedType(out.imports()),
447                                 x.fieldName(cls))) //
448                         .collect(Collectors.joining(","));
449                 out.left().left();
450                 out.println();
451                 out.line("public %s(%s) {", Names.simpleClassName(cls.fullClassName), parameters);
452                 ifValidate(cls, out, names, //
453                         o -> cls.fields.stream().forEach(x -> {
454                             if (!x.isPrimitive() && x.required) {
455                                 checkNotNull(cls, o, x);
456                             } else {
457                                 o.line("// TODO %s", x.fieldName);
458                             }
459                             validateMore(o, cls, x);
460                         }));
461                 cls.fields.stream().forEach(x -> assignField(out, cls, x));
462                 out.closeParen();
463 
464                 // write getters for allOf members
465                 {   
466                     // avoid name clashes
467                     Map<Field, String> getterNames = new HashMap<>();
468                     Set<String> usedNames = new HashSet<>(); 
469                     cls.fields.forEach(f -> {
470                         String name = "as" + Names.simpleClassName(f.resolvedType(out.imports()));
471                         int i = 0;
472                         String modifiedName = name;
473                         while (usedNames.contains(modifiedName)) {
474                             i++;
475                             modifiedName = name + i;
476                         }
477                         getterNames.put(f, modifiedName);
478                         usedNames.add(modifiedName);
479                     });
480                     cls.fields.forEach(f -> {
481                         out.println();
482                         writeGetter(out, f.resolvedType(out.imports()), getterNames.get(f), f.fieldName(cls));
483                     });
484                 }
485 
486                 // write all field getters
487                 Set<String> used = new HashSet<>();
488                 cls.fields.forEach(field -> {
489                     Optional<Cls> c = names.cls(field.fullClassName);
490                     if (c.isPresent() && c.get().classType != ClassType.ONE_OF_NON_DISCRIMINATED) {
491                         c.get().fields //
492                                 .stream() //
493                                 .filter(f -> !f.mapType.isPresent()) //
494                                 .forEach(f -> {
495                                     String fieldName = f.fieldName(c.get());
496                                     if (!used.contains(fieldName)) {
497                                         used.add(fieldName);
498                                         String type = f.resolvedTypePublicConstructor(out.imports());
499                                         out.println();
500                                         out.line("public %s %s() {", type, fieldName);
501                                         final String getter;
502                                         if (c.get().classType == ClassType.ALL_OF) {
503                                             getter = "as" + Names.upperFirst(fieldName);
504                                         } else {
505                                             getter = fieldName;
506                                         }
507                                         out.line("return %s.%s();", field.fieldName(cls), getter);
508                                         out.closeParen();
509                                     }
510                                 });
511                     }
512                 });
513                 // write allof builder
514                 writeAnyOfOrAllOfBuilder(out, cls, false);
515             }
516 
517             out.println();
518             out.line("@%s(\"serial\")", SuppressWarnings.class);
519             final Class<?> polymorphicDeserializer;
520             if (cls.classType == ClassType.ANY_OF_NON_DISCRIMINATED) {
521                 polymorphicDeserializer = AnyOfDeserializer.class;
522             } else {
523                 polymorphicDeserializer = PolymorphicDeserializer.class;
524             }
525             out.line("public static final class _Deserializer extends %s<%s> {", polymorphicDeserializer,
526                     cls.simpleName());
527             out.println();
528             out.line("public _Deserializer() {");
529             String classes = cls.fields //
530                     .stream() //

676     private static void writeConstructor(CodePrintWriter out, Cls cls, Map<String, Set<Cls>> fullClassNameInterfaces,
677             Names names) {
678 
679         Set<Cls> interfaces = interfaces(cls, fullClassNameInterfaces);
680 
681         boolean hasAdditionalProperties = cls.fields.stream().anyMatch(Field::isAdditionalProperties);
682         boolean hasDiscriminator = cls.fields.stream().anyMatch(x -> isDiscriminator(interfaces, x));
683         boolean extraConstructor = hasAdditionalProperties && cls.classType == ClassType.CLASS;
684         if (extraConstructor) {
685             // if has additionalProperties then we make the JsonCreator constructor private
686             // (excluding properties) and make another public constructor that includes the
687             // Map for additional properties as a parameter)
688             out.right().right();
689             String parameters = cls //
690                     .fields //
691                     .stream() //
692                     // ignore discriminators that should be constants
693 //                    .filter(x -> !isDiscriminator(interfaces, x)) //
694                     .map(x -> {
695                         final String t;
696                         if (x.mapType.isPresent()) {
697                             t = x.resolvedTypeMapPublic(out.imports());
698                         } else {
699                             t = x.resolvedTypePublicConstructor(out.imports());
700                         }
701                         return String.format("\n%s%s %s", out.indent(), t, x.fieldName(cls));
702                     }) //
703                     .collect(Collectors.joining(","));
704             out.left().left();
705             out.println();
706             out.line("public %s(%s) {", Names.simpleClassName(cls.fullClassName), parameters);
707             writeConstructorBody(out, cls, names, interfaces, true);
708             out.closeParen();
709         }
710         out.right().right();
711         String parameters = cls //
712                 .fields //
713                 .stream() //
714                 // ignore discriminators that should be constants
715 //                    .filter(x -> !isDiscriminator(interfaces, x)) //
716                 .filter(x -> !x.isAdditionalProperties()) //
717                 .map(f -> {
718                     final String t;
719                     if (f.mapType.isPresent()) {
720                         t = f.resolvedTypeMapPublic(out.imports());
721                     } else {
722                         t = f.resolvedTypePublicConstructor(out.imports());
723                     }
724                     String annotations = cls.unwrapSingleField() ? "" //
725                             : String.format("@%s(\"%s\") ", out.add(JsonProperty.class), f.name);
726                     if (f.writeOnly) {
727                         annotations += String.format("@%s(using = %s.class) ", out.add(JsonDeserialize.class),
728                                 out.add(OptionalEmptyDeserializer.class));
729                     } else if (f.isOctets()) {
730                         if (f.required && f.readOnly) {
731                             annotations += String.format("@%s(using = %s.class) ", out.add(JsonDeserialize.class),
732                                     out.add(OptionalPresentOctetsDeserializer.class));
733                         } else if (!f.required && f.nullable) {
734                             annotations += String.format("@%s(using = %s.class) ", out.add(JsonDeserialize.class),
735                                     out.add(JsonNullableOctetsDeserializer.class));
736                         } else if (!f.required || f.nullable) {
737                             annotations += String.format("@%s(using = %s.class) ", out.add(JsonDeserialize.class),
738                                     out.add(OptionalOctetsDeserializer.class));
739                         } else {
740                             annotations += String.format("@%s(using = %s.class) ", out.add(JsonDeserialize.class),
741                                     out.add(OctetsDeserializer.class));
742                         }
743                     } else if (f.required && f.readOnly && !f.nullable) {
744                         annotations += String.format("@%s(converter = %s.class) ", out.add(JsonDeserialize.class),
745                                 out.add(OptionalMustBePresentConverter.class));
746                     }
747                     return String.format("\n%s%s%s %s", out.indent(), annotations, t, f.fieldName(cls));
748                 }) //
749                 .collect(Collectors.joining(","));
750         out.left().left();
751         out.println();
752         final String modifier;
753         if (cls.classType == ClassType.ENUM) {
754             modifier = "";
755         } else if (hasDiscriminator || extraConstructor) {
756             modifier = "private ";
757         } else {
758             modifier = "public ";
759         }
760         if (modifier.equals("private ")) {
761             addConstructorBindingAnnotation(out, names);
762         }
763         if (cls.classType != ClassType.ENUM) {
764             out.line("@%s", JsonCreator.class);
765         }
766         out.line("%s%s(%s) {", modifier, Names.simpleClassName(cls.fullClassName), parameters);
767         writeConstructorBody(out, cls, names, interfaces, false);
768         out.closeParen();
769     }

794     private static void writeConstraintValidations(CodePrintWriter out, Cls cls, Names names, Set<Cls> interfaces,
795             boolean additionalPropertiesIsParameter, List<Field> fields) {

799                         .forEach(x -> {

801                                 if (x.isOctets() || !x.isPrimitive()) {

809     private static void writeConstructorBodyFieldAssignment(CodePrintWriter out, Cls cls, Set<Cls> interfaces, Field x,
810             boolean additionalPropertiesIsParameter) {
811         if (x.isAdditionalProperties() && !additionalPropertiesIsParameter) {
812             out.line("this.%s = new %s<>();", x.fieldName(cls), HashMap.class);
813         } else if (x.mapType.isPresent()) {
814             if (x.isArray) {
815                 out.line("this.%s = new %s<>();", x.fieldName(cls), ArrayList.class);
816             } else if (x.required && !x.nullable) {
817                 out.line("this.%s = %s.createMapIfNull(%s);", x.fieldName(cls), Util.class, x.fieldName(cls));
818             } else {
819                 out.line("this.%s = %s;", x.fieldName(cls), x.fieldName(cls));
820             }
821         } else {
822             Optional<Discriminator> disc = discriminator(interfaces, x);
823             if (disc.isPresent()) {
824                 // write constant value for discriminator, if is enum then
825                 // grab it's value using the DiscriminatorHelper
826                 out.line("%s.checkEquals(%s.value(%s.class, \"%s\"), %s, \"%s\");", //
827                         Preconditions.class, //
828                         DiscriminatorHelper.class, //
829                         out.add(x.fullClassName), //
830                         disc.get().discriminatorValueFromFullClassName(cls.fullClassName), //
831                         x.fieldName(cls), //
832                         x.fieldName(cls));
833                 out.line("this.%s = %s;", x.fieldName(cls), x.fieldName(cls));
834             } else if (x.nullable) {
835                 if (x.required) {
836                     out.line("this.%s = %s;", x.fieldName(cls), x.fieldName(cls));
837                 } else {
838                     assignField(out, cls, x);
839                 }
840             } else if (!x.isPrimitive()) {
841                 if (x.required) {
842                     assignField(out, cls, x);
843                 } else {
844                     assignOptionalField(out, cls, x);
845                 }
846             } else {
847                 assignField(out, cls, x);
848             }
849         }
850     }

857         if (cls.classType == ClassType.ENUM) {

907     private static void validateMore(CodePrintWriter out, Cls cls, Field x) {
908         if (x.isAdditionalProperties()) {
909             // TODO check values of map
910             return;
911         }
912         String raw = x.fieldName(cls);
913         if (x.minLength.isPresent() && !x.isDateOrTime()) {
914             out.line("%s.checkMinLength(%s, %s, \"%s\");", Preconditions.class, raw, x.minLength.get(), raw);
915         }
916         if (x.maxLength.isPresent() && !x.isDateOrTime()) {
917             out.line("%s.checkMaxLength(%s, %s, \"%s\");", Preconditions.class, raw, x.maxLength.get(), raw);
918         }
919         if (x.pattern.isPresent() && !x.isDateOrTime() && !x.isByteArray() && !x.isOctets()) {

921                     WriterUtil.escapePattern(x.pattern.get()), raw);

923         if (x.min.isPresent() && x.isNumber()) {
924             out.line("%s.checkMinimum(%s, \"%s\", \"%s\", %s);", Preconditions.class, raw, x.min.get().toString(), raw,
925                     x.exclusiveMin);
926         }
927         if (x.max.isPresent() && x.isNumber()) {
928             out.line("%s.checkMaximum(%s, \"%s\", \"%s\", %s);", Preconditions.class, raw, x.max.get().toString(), raw,
929                     x.exclusiveMax);
930         }
931         if (x.isArray && x.minItems.isPresent()) {
932             out.line("%s.checkMinSize(%s, %s, \"%s\");", Preconditions.class, raw, x.minItems.get(), raw);
933         }
934         if (x.isArray && x.maxItems.isPresent()) {
935             out.line("%s.checkMaxSize(%s, %s, \"%s\");", Preconditions.class, raw, x.maxItems.get(), raw);
936         }
937     }

1064     private static void writeMutators(CodePrintWriter out, Cls cls, Map<String, Set<Cls>> fullClassNameInterfaces) {
1065         List<Field> fields = cls.fields //
1066                 .stream() //
1067                 .collect(Collectors.toList());
1068         if (fields.size() <= 1) {
1069             return;
1070         }
1071         fields //
1072                 .stream() //
1073                 // ignore discriminators that should be constants so don't need a mutator
1074                 .filter(x -> !isDiscriminator(interfaces(cls, fullClassNameInterfaces), x)) //
1075                 .forEach(x -> {
1076                     String t = x.mapType.isPresent() ? x.resolvedTypeMapPublic(out.imports())
1077                             : x.resolvedTypePublicConstructor(out.imports());
1078                     out.println();
1079                     out.line("public %s with%s(%s %s) {", cls.simpleName(), Names.upperFirst(x.fieldName(cls)), t,
1080                             x.fieldName(cls));
1081                     {
1082                         String params = fields.stream() //
1083                                 .map(y -> y.fieldName(cls)) //
1084                                 .collect(Collectors.joining(", "));
1085                         out.line("return new %s(%s);", cls.simpleName(), params);
1086                         out.closeParen();
1087                     }
1088                     if (!x.mapType.isPresent()) {
1089                         Optional<String> tNonOptional = x.resolvedTypePublicConstructorNonOptional(out.imports());
1090                         if (tNonOptional.isPresent() && !tNonOptional.get().equals(t)) {
1091                             out.println();
1092                             out.line("public %s with%s(%s %s) {", cls.simpleName(), Names.upperFirst(x.fieldName(cls)),
1093                                     tNonOptional.get(), x.fieldName(cls));
1094                             String params = fields.stream() //
1095                                     .map(y -> {
1096                                         if (y.fieldName(cls).equals(x.fieldName(cls))) {
1097                                             if (y.nullable && !y.required) {
1098                                                 return String.format("%s.of(%s)", out.add(JsonNullable.class),
1099                                                         y.fieldName(cls));
1100                                             } else {
1101                                                 return String.format("%s.of(%s)", out.add(Optional.class),
1102                                                         y.fieldName(cls));
1103                                             }
1104                                         } else {
1105                                             return y.fieldName(cls);
1106                                         }
1107                                     }).collect(Collectors.joining(", "));
1108                             out.line("return new %s(%s);", cls.simpleName(), params);
1109                             out.closeParen();
1110                         }
1111                     }
1112                 });
1113     }
