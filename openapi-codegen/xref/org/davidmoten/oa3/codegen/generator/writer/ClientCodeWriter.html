21      public static void writeClientClass(Names names, List<Method> methods) {
22          String fullClassName = names.clientFullClassName();
23          CodePrintWriter out = CodePrintWriter.create(fullClassName);
24          writeClientClass(out, names, methods);
25          WriterUtil.writeContent(names, out);
26      }
27  
28      private static void writeClientClass(CodePrintWriter out, Names names, List<Method> methods) {
29          out.line("package %s;", out.pkg());
30          out.println();
31          out.format("%s", WriterUtil.IMPORTS_HERE);
32          WriterUtil.writeApiJavadoc(out, names);
33          WriterUtil.addGeneratedAnnotation(out);
34          out.line("public class %s {", out.simpleClassName());
35          writeClientClassFieldsConstructorAndBuilder(out, names);
36          writeClientClassMethods(out, methods);
37          out.closeParen();
38      }
39  
40      private static void writeClientClassFieldsConstructorAndBuilder(CodePrintWriter out, Names names) {
41          // add fields
42          out.println();
43          out.line("private final %s serializer;", Serializer.class);
44          out.line("private final %s<%s> interceptors;", List.class, Interceptor.class);
45          out.line("private final %s basePath;", String.class);
46  
47          // write constructor
48          out.println();
49          out.line("private %s(%s serializer, %s<%s> interceptors, %s basePath) {", out.simpleClassName(),
50                  Serializer.class, List.class, Interceptor.class, String.class);
51          out.line("this.serializer = serializer;");
52          out.line("this.interceptors = interceptors;");
53          out.line("this.basePath = basePath;");
54          out.closeParen();
55  
56          // write builder
57          out.println();
58          out.line("public static %s<%s> basePath(%s basePath) {", ClientBuilder.class, out.simpleClassName(),
59                  String.class);
60          out.line("return new %s<>(b -> new %s(b.serializer(), b.interceptors(), b.basePath()), %s.config(), basePath);",
61                  ClientBuilder.class, out.simpleClassName(), out.add(names.globalsFullClassName()));
62          out.closeParen();
63      }
64  
65      private static final String FULL_RESPONSE_SUFFIX = "FullResponse";
66  
67      private static void writeClientClassMethods(CodePrintWriter out, List<Method> methods) {
68          methods.forEach(m -> {
69              out.right().right();
70              String params = m.parameters //
71                      .stream() //
72                      .map(p -> String.format("\n%s%s %s", out.indent(),
73                              ServerCodeWriterSpringBoot.toImportedType(p, out.imports()), p.identifier)) //
74                      .collect(Collectors.joining(", "));
75              out.left().left();
76              final String importedReturnType;
77              if (!m.returnFullClassName.isPresent()) {
78                  importedReturnType = "void";
79              } else {
80                  importedReturnType = out.add(m.returnFullClassName.get());
81              }
82              if (m.primaryStatusCode.isPresent() && m.primaryMediaType.isPresent()) {
83                  final Optional<String> returns;
84                  if (m.returnFullClassName.isPresent()) {
85                      returns = m.primaryStatusCode.map(x -> "primary response with status code " + x);
86                  } else {
87                      returns = Optional.empty();
88                  }
89                  ServerCodeWriterSpringBoot.writeMethodJavadoc(out, m, returns);
90                  out.line("public %s %s(%s) {", importedReturnType, m.methodName, params);
91                  final String paramIdentifiers;
92                  if (m.parameters.size() <= 3) {
93                      paramIdentifiers = m.parameters.stream().map(p -> p.identifier).collect(Collectors.joining(", "));
94                  } else {
95                      out.right().right().right();
96                      paramIdentifiers = m.parameters.stream()
97                              .map(p -> String.format("\n%s%s", out.indent(), p.identifier))
98                              .collect(Collectors.joining(","));
99                      out.left().left().left();
100                 }
101                 out.line("return %s%s(%s)", m.methodName, FULL_RESPONSE_SUFFIX, paramIdentifiers);
102                 out.right().right();
103                 out.line(".assertStatusCodeMatches(\"%s\")", m.primaryStatusCode.get());
104                 out.line(".assertContentTypeMatches(\"%s\")", m.primaryMediaType.get());
105                 out.line(".dataUnwrapped();");
106                 out.left().left();
107                 out.closeParen();
108             }
109             ServerCodeWriterSpringBoot.writeMethodJavadoc(out, m,
110                     Optional.of("full response with status code, body and headers"));
111             out.line("public %s %s%s(%s) {", HttpResponse.class, m.methodName, FULL_RESPONSE_SUFFIX, params);
112             out.line("return %s", Http.class);
113             out.right().right();
114             out.line(".method(%s.%s)", HttpMethod.class, m.httpMethod.name());
115             out.line(".basePath(this.basePath)");
116             out.line(".path(\"%s\")", m.path);
117             out.line(".serializer(this.serializer)");
118             out.line(".interceptors(this.interceptors)");
119             out.line(".acceptApplicationJson()");
120             m.parameters.forEach(p -> {
121                 if (p.type == ParamType.QUERY) {
122                     out.line(".queryParam(\"%s\", %s)", p.name, p.identifier);
123                 } else if (p.type == ParamType.PATH) {
124                     out.line(".pathParam(\"%s\", %s)", p.name, p.identifier);
125                 } else if (p.type == ParamType.BODY) {
126                     out.line(".body(%s)", p.identifier);
127                     out.line(".contentTypeApplicationJson()");
128                 } else if (p.type == ParamType.COOKIE) {
129                     out.line(".cookie(\"%s\", %s)", p.name, p.identifier);
130                 } else if (p.type == ParamType.HEADER) {
131                     out.line(".header(\"%s\", %s)", p.name, p.identifier);
132                 }
133             });
134             m.responseDescriptors.forEach(r -> {
135                 out.line(".responseAs(%s.class)", out.add(r.fullClassName()));
136                 out.line(".whenStatusCodeMatches(\"%s\")", r.statusCode());
137                 out.line(".whenContentTypeMatches(\"%s\")", r.mediaType());
138             });
139             out.line(".call();");
140             out.left().left();
141             out.closeParen();
142         });
143     }
