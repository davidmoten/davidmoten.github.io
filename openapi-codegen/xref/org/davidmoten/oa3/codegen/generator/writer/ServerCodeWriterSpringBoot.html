49      public static void writeServiceClasses(Names names, List<Method> methods) {
50          writeApplicationClass(names);
51          writeJacksonConfigurationClass(names);
52          List<Method> includedMethods = methods //
53                  .stream() //
54                  .filter(x-> x.includeForServerGeneration) //
55                  .collect(Collectors.toList());
56          writeServiceControllerClass(names, includedMethods);
57          writeServiceInterfaceClass(names, includedMethods);
58      }

60      private static void writeApplicationClass(Names names) {
61          CodePrintWriter out = CodePrintWriter.create(names.applicationFullClassName(), names.simpleNameInPackage(names.applicationFullClassName()));
62          writeApplicationClass(out);
63          WriterUtil.writeContent(names, out);
64      }

68          out.println();
69          out.format("%s", IMPORTS_HERE);

71          out.line("@%s", SpringBootApplication.class);
72          WriterUtil.addGeneratedAnnotation(out);

78          out.left();
79          out.println();
80          out.line("}");
81      }

162     private static void writeServiceMethods(CodePrintWriter out, List<Method> methods, boolean isController,
163             Names names) {
164         methods.forEach(m -> {
165             final Optional<String> returns;
166             if (m.returnFullClassName.isPresent()) {
167                 returns = m.primaryStatusCode.map(x -> "primary response status code " + x);
168             } else {
169                 returns = Optional.empty();
170             }
171             writeMethodJavadoc(out, m, returns, Maps.empty());
172             out.right().right();
173             String params = m.parameters.stream().map(p -> {
174                 if (p.isRequestBody) {
175                     final String annotations;
176                     if (isController) {
177                         annotations = String.format("@%s ", out.add(RequestBody.class));
178                     } else {
179                         annotations = "";
180                     }
181                     return String.format("\n%s%s%s %s", out.indent(), annotations, toImportedType(p, out.imports()),
182                             "requestBody");
183                 } else {
184                     final String annotations;
185                     if (isController) {
186                         String defValue = p.defaultValue.map(x -> ", defaultValue = \"" + x + "\"").orElse("");
187                         String required = ", required = " + p.required;
188                         Class<?> ann = annotation(p.type);
189                         if (ann.equals(RequestParam.class) && p.isComplexQueryParameter) {
190                             ann = ModelAttribute.class;
191                             required = "";
192                             defValue = "";
193                         }
194                         annotations = String.format("@%s(name = \"%s\"%s%s) ", out.add(ann), p.name, defValue,
195                                 required);
196                     } else {
197                         annotations = "";
198                     }
199                     return String.format("\n%s%s%s %s", out.indent(), annotations, toImportedType(p, out.imports()),
200                             p.identifier);
201                 }
202             }).collect(Collectors.joining(", "));
203             out.left().left();
204             final String importedReturnType;
205             if (isController) {
206                 importedReturnType = String.format("%s<?>", out.add(ResponseEntity.class));
207             } else if (!m.returnFullClassName.isPresent()) {
208                 importedReturnType = "void";
209             } else {
210                 final String fullClassName;
211                 if (m.returnFullClassName.orElse("").equals(byte[].class.getCanonicalName())) {
212                     fullClassName = Resource.class.getCanonicalName();
213                 } else {
214                     fullClassName = m.returnFullClassName.get();
215                 }
216                 importedReturnType = out.add(fullClassName);
217             }
218 //            @RequestMapping(
219 //                    method = RequestMethod.POST,
220 //                    value = "/postWithRequestBodyNotRequired",
221 //                    produces = { "application/json" },
222 //                    consumes = { "application/json" }
223 //                )
224             if (isController) {
225                 out.line("@%s(", RequestMapping.class);
226                 out.right();
227                 String consumes = m.consumes.stream().map(x -> "\"" + x + "\"").collect(Collectors.joining(", "));
228                 if (!consumes.isEmpty()) {
229                     consumes = String.format(",\n%sconsumes = {%s}", out.indent(), consumes);
230                 }
231                 String produces = m.produces.stream().map(x -> "\"" + x + "\"").collect(Collectors.joining(", "));
232                 if (!produces.isEmpty()) {
233                     produces = String.format(",\n%sproduces = {%s}", out.indent(), produces);
234                 }
235                 out.line("method = %s.%s,", RequestMethod.class, m.httpMethod);
236                 out.line("value = \"%s\"%s%s)", m.path, consumes, produces);
237                 out.left();
238                 out.line("public %s %s(%s) {", importedReturnType, m.methodName, params);
239                 out.line("try {");
240                 addValidationChecks(out, m, names);
241                 if (m.returnFullClassName.isPresent()) {
242                     out.line("return %s.status(%s).body(service.%s(%s));", ResponseEntity.class, //
243                             m.statusCodeFirstInRange().orElse(200), //
244                             m.methodName, //
245                             m.parameters.stream().map(p -> p.identifier).collect(Collectors.joining(", ")));
246                 } else {
247                     out.line("service.%s(%s);", m.methodName,
248                             m.parameters.stream().map(p -> p.identifier).collect(Collectors.joining(", ")));
249                     out.line("return %s.status(%s).build();", ResponseEntity.class, m.statusCodeFirstInRange().orElse(200));
250                 }
251                 out.left();
252                 out.line("} catch (%s e) {", Throwable.class);
253                 out.line("return service.errorResponse(e);");
254                 out.closeParen();
255                 out.closeParen();
256             } else {
257                 out.line("default %s %s(%s) throws %s {", importedReturnType, m.methodName, params,
258                         ServiceException.class);
259                 out.line("throw notImplemented();");
260                 out.closeParen();
261             }
262         });
263     }

265     static void writeMethodJavadoc(CodePrintWriter out, Method m, Optional<String> returns, Map<String, String> throwing) {
266         Map<String, String> parameterDescriptions = m.parameters //
267                 .stream() //
268                 .collect(Collectors.toMap(x -> x.identifier,
269                         x -> WriterUtil.markdownToHtml(x.description.orElse(x.identifier))));
270         String html = m.description.map(x -> WriterUtil.markdownToHtml(x))
271                 .orElse("<p>Returns response from call to path <i>%s</i>.</p>");
272         String more = m.responseDescriptors.stream() //
273                 .map(rd -> {
274                     String full = rd.fullClassName();
275                     String link = Util.isPrimitiveFullClassName(full) || full.equals("byte[]") ? full
276                             : String.format("{@link %s}", out.add(full));
277                     return String.format("\n<p>[status=%s, %s] --&gt; %s</p>", rd.statusCode(), rd.mediaType(), link);
278                 }).collect(Collectors.joining());
279         if (!Javadoc.printJavadoc(out, Optional.of(html + more), Collections.emptyList(), Optional.empty(), returns,
280                 parameterDescriptions, true, throwing)) {
281             out.println();
282         }
283     }

285     private static Class<?> annotation(ParamType t) {
286         if (t == ParamType.BODY) {
287             return RequestBody.class;
288         } else if (t == ParamType.QUERY) {
289             return RequestParam.class;
290         } else if (t == ParamType.PATH) {
291             return PathVariable.class;
292         } else if (t == ParamType.HEADER) {
293             return RequestHeader.class;
294         } else if (t == ParamType.COOKIE) {
295             return CookieValue.class;
296         } else {
297             throw new IllegalArgumentException("unexpected " + t);
298         }
299     }

301     private static void addValidationChecks(CodePrintWriter out, Method m, Names names) {
302         boolean found = m.parameters.stream().anyMatch(p -> p.constraints.atLeastOnePresent());
303         if (found) {
304             out.line("if (%s.config().validateInControllerMethod().test(\"%s\")) {",
305                     out.add(names.globalsFullClassName()), m.methodName);
306             m.parameters.forEach(p -> {
307                 Constraints x = p.constraints;
308                 if (x.atLeastOnePresent()) {
309                     if (x.minLength.isPresent()) {
310                         out.line("%s.checkMinLength(%s, %s, \"%s\");", RequestPreconditions.class, p.identifier,
311                                 x.minLength.get(), p.identifier);
312                     }
313                     if (x.maxLength.isPresent()) {
314                         out.line("%s.checkMaxLength(%s, %s, \"%s\");", RequestPreconditions.class, p.identifier,
315                                 x.maxLength.get(), p.identifier);
316                     }
317                     if (x.pattern.isPresent()) {
318                         out.line("%s.checkMatchesPattern(%s, \"%s\", \"%s\");", RequestPreconditions.class,
319                                 p.identifier, WriterUtil.escapePattern(x.pattern.get()), p.identifier);
320                     }
321                     if (x.min.isPresent()) {
322                         out.line("%s.checkMinimum(%s, \"%s\", \"%s\", %s);", RequestPreconditions.class, p.identifier,
323                                 x.min.get().toString(), p.identifier, false);
324                     }
325                     if (x.max.isPresent()) {
326                         out.line("%s.checkMaximum(%s, \"%s\", \"%s\", %s);", RequestPreconditions.class, p.identifier,
327                                 x.max.get().toString(), p.identifier, false);
328                     }
329                     if (x.minExclusive.isPresent()) {
330                         out.line("%s.checkMinimum(%s, \"%s\", \"%s\", %s);", RequestPreconditions.class, p.identifier,
331                                 x.minExclusive.get().toString(), p.identifier, true);
332                     }
333                     if (x.maxExclusive.isPresent()) {
334                         out.line("%s.checkMaximum(%s, \"%s\", \"%s\", %s);", RequestPreconditions.class, p.identifier,
335                                 x.maxExclusive.get().toString(), p.identifier, true);
336                     }
337                     if (p.isArray && x.minItems.isPresent()) {
338                         out.line("%s.checkMinSize(%s, %s, \"%s\");", RequestPreconditions.class, p.identifier,
339                                 x.minItems.get(), p.identifier);
340                     }
341                     if (p.isArray && x.maxItems.isPresent()) {
342                         out.line("%s.checkMaxSize(%s, %s, \"%s\");", RequestPreconditions.class, p.identifier,
343                                 x.maxItems.get(), p.identifier);
344                     }
345                 }
346             });
347             out.closeParen();
348         }

351     static String toImportedType(Param p, Imports imports) {
352         if (p.isArray) {
353             if (p.required) {
354                 return String.format("%s<%s>", imports.add(List.class), imports.add(p.fullClassName));
355             } else {
356                 return String.format("%s<%s<%s>>", imports.add(Optional.class), imports.add(List.class),
357                         imports.add(p.fullClassName));
358             }
359         } else if (p.required || p.defaultValue.isPresent()) {
360             return imports.add(Util.toPrimitive(p.fullClassName));
361         } else {
362             return String.format("%s<%s>", imports.add(Optional.class), imports.add(p.fullClassName));
363         }
364     }
