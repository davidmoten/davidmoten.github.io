26      public static String getTypeOrThrow(Schema<?> schema) {
27          return getType(schema) //
28                  .orElseThrow(() -> new IllegalStateException("no type found for\n" + schema));
29      }

31      public static Optional<String> getType(Schema<?> schema) {
32          if (schema.getType() == null) {
33              if (schema.getTypes() != null) {
34                  // mainly want to cover 3.1 nullable: ["string", "null"]
35                  // oneOf should take another path hopefully, depending on what parser does
36                  return schema //
37                          .getTypes() //
38                          .stream() //
39                          .filter(x -> !"null".equals(x)) //
40                          .findFirst();
41              } else {
42                  return Optional.empty();
43              }
44          } else {
45              return Optional.of(schema.getType());
46          }
47      }

107     public static Class<?> toClass(String type, String format, Map<String, Object> extensions,
108             boolean mapIntegerToBigInteger, boolean mapDoubleToBigDouble) {
109         Preconditions.checkNotNull(type);
110         if ("string".equals(type)) {
111             if ("date-time".equals(format)) {
112                 return OffsetDateTime.class;
113             } else if ("date".equals(format)) {
114                 return LocalDate.class;
115             } else if ("time".equals(format)) {
116                 return OffsetTime.class;
117             } else if ("byte".equals(format)) {
118                 return byte[].class;
119             } else if ("binary".equals(format)) {
120                 return byte[].class;
121             } else {
122                 return String.class;
123             }
124         } else if ("boolean".equals(type)) {
125             return Boolean.class;
126         } else if ("integer".equals(type)) {
127             if ("int32".equals(format)) {
128                 return Integer.class;
129             } else if ("int64".equals(format)) {
130                 return Long.class;
131             } else {
132                 return mapIntegerToBigInteger || isArbitraryPrecision(extensions) ? BigInteger.class : Long.class;
133             }
134         } else if ("number".equals(type)) {
135             if ("float".equals(format)) {
136                 return Float.class;
137             } else if ("double".equals(format)) {
138                 return Double.class;
139             } else {
140                 return mapDoubleToBigDouble || isArbitraryPrecision(extensions) ? BigDecimal.class : Double.class;
141             }
142         } else if ("array".equals(type)) {
143             return List.class;
144         } else if ("object".equals(type)) {
145             return Map.class;
146         } else {
147             throw new RuntimeException("unexpected type and format: " + type + ", " + format);
148         }
149     }
150 
151     private static boolean isArbitraryPrecision(Map<String, Object> extensions) {
152         return extensions != null
153                 && "true".equalsIgnoreCase(String.valueOf(extensions.get("x-openapi-codegen-arbitrary-precision")));
154     }

156     public static String toPrimitive(String canonicalClassName) {
157         if (canonicalClassName.equals(Integer.class.getCanonicalName())) {
158             return "int";
159         } else if (canonicalClassName.equals(Short.class.getCanonicalName())) {
160             return "short";
161         } else if (canonicalClassName.equals(Long.class.getCanonicalName())) {
162             return "long";
163         } else if (canonicalClassName.equals(Float.class.getCanonicalName())) {
164             return "float";
165         } else if (canonicalClassName.equals(Double.class.getCanonicalName())) {
166             return "double";
167         } else if (canonicalClassName.equals(Boolean.class.getCanonicalName())) {
168             return "boolean";
169         } else if (canonicalClassName.equals(Byte.class.getCanonicalName())) {
170             return "byte";
171         } else {
172             return canonicalClassName;
173         }
174     }

176     public static Class<?> toPrimitive(Class<?> c) {
177         if (c.equals(Integer.class)) {
178             return int.class;
179         } else if (c.equals(Long.class)) {
180             return long.class;
181         } else if (c.equals(Float.class)) {
182             return float.class;
183         } else if (c.equals(Boolean.class)) {
184             return boolean.class;
185         } else if (c.equals(Short.class)) {
186             return short.class;
187         } else if (c.equals(Byte.class)) {
188             return byte.class;
189         } else if (c.equals(BigInteger.class)) {
190             return c;
191         } else {
192             return c;
193         }
194     }
