3   public final class LinkedStack<T> {
4       LinkedStackNode<T> last;
5   
6       public void push(T value) {
7           LinkedStackNode<T> node = new LinkedStackNode<>(value);
8           if (last != null) {
9               node.previous = last;
10          }
11          last = node;
12      }
13  
14      public T pop() {
15          if (last == null) {
16              return null;
17          } else {
18              T v = last.value;
19              last = last.previous;
20              return v;
21          }
22      }
23  
24      public T peek() {
25          if (last == null) {
26              return null;
27          } else {
28              return last.value;
29          }
30      }
31  
32      public boolean isEmpty() {
33          return last == null;
34      }
35  
36      private static final class LinkedStackNode<T> {
37          final T value;
38          LinkedStackNode<T> previous;
39  
40          LinkedStackNode(T value) {
