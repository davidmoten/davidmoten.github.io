5   import org.springframework.http.ResponseEntity;

7   public final class ServiceException extends Exception {
8   
9       private static final long serialVersionUID = -4115693210890378905L;
10  
11      private final int statusCode;
12      private final Optional<? extends ResponseEntity<?>> response;
13  
14      private ServiceException(int statusCode, String message, Optional<? extends ResponseEntity<?>> response) {
15          super(message);
16          this.statusCode = statusCode;
17          this.response = response;
18      }
19  
20      public ServiceException(int statusCode, String message) {
21          this(statusCode, message, Optional.empty());
22      }
23  
24      public ServiceException(int statusCode, Throwable e) {
25          super(e);
26          this.statusCode = statusCode;
27          this.response = Optional.empty();
28      }
29  
30      public ServiceException(ResponseEntity<?> response) {
31          this(response.getStatusCodeValue(), response.getStatusCode().toString(), Optional.of(response));
32      }
33  
34      public int statusCode() {
35          return statusCode;
36      }
37  
38      public Optional<? extends ResponseEntity<?>> response() {
39          return response;
40      }
41  
42  }
