10  import org.springframework.stereotype.Component;
11  
12  import com.github.davidmoten.guavamini.Lists;
13  
14  import pet.store.path.FindPets200Response;
15  import pet.store.schema.Error;
16  import pet.store.schema.NewPet;
17  import pet.store.schema.Pet;
18  import pet.store.schema.PetId;
19  import pet.store.service.Service;
20  
21  @Component
22  @Primary
23  public class PetStoreService implements Service {
24  
25      private List<Pet> pets = Lists.of( //
26              new Pet(NewPet.builder().name("fido").tag("A0123").build(), //
27                      PetId.id(321L)));
28  
29      @Override
30      public FindPets200Response findPets(Optional<List<String>> tags, int limit) throws ServiceException {
31          return FindPets200Response.of(pets);
32      }
33  
34      @Override
35      public Pet addPet(NewPet a) throws ServiceException {
36          Pet p = new Pet(a, PetId.id(System.currentTimeMillis()));
37          pets.add(p);
38          return p;
39      }
40  
41      @Override
42      public Pet find_pet_by_id(long id) throws ServiceException {
43          Optional<Pet> pet = pets.stream().filter(p -> p.id() == id).findAny();
44          if (pet.isPresent()) {
45              return pet.get();
46          } else {
47              return response(ResponseEntity.status(HttpStatus.NOT_FOUND).body(Error.message("not found")));
48          }
49      }
50  
51      @Override
52      public void deletePet(long id) throws ServiceException {
53          Optional<Pet> pet = pets.stream().filter(p -> p.id() == id).findAny();
54          if (pet.isPresent()) {
55              pets.remove(pet.get());
56          } else {
57              response(ResponseEntity.status(HttpStatus.NOT_FOUND).body(Error.message("not found")));
58          }
59      }
60  
61      @Override
62      public Object errorResponseBody(int statusCode, Throwable e) {
63          return Error.message("an error happened");
64      }
65  
66  }
