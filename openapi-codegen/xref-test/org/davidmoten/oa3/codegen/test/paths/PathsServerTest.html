29      private static final ObjectMapper m = Globals.config().mapper();

42      @Test
43      public void testGet() {
44          Response2 r = Http.read(basePath() + "/item", HttpMethod.GET, Response2.class, m);
45          assertEquals("abcToken", r.token());
46      }

48      @Test
49      public void testDefaultError() {
50          DefaultError error = Http.readError(basePath() + "/item201", HttpMethod.GET, DefaultError.class, m);

52          assertEquals(500, error.statusCode());
53      }

55      @Test
56      public void testResponseMultiTypeWithAcceptHeader() {
57          HttpHeaders headers = new HttpHeaders();
58          headers.put(HttpHeaders.ACCEPT, Arrays.asList("application/json"));
59          Response1 response = Http.read(basePath() + "/responseMultiType?username=fred", HttpMethod.GET, Response1.class,
60                  headers, m);
61          assertEquals("fred", response.thing());
62      }

64      @Test

69      }
