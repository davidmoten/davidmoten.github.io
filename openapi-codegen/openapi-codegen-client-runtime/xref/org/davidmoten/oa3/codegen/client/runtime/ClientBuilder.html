1   package org.davidmoten.oa3.codegen.client.runtime;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.function.Function;
6   
7   import org.davidmoten.oa3.codegen.http.DefaultSerializer;
8   import org.davidmoten.oa3.codegen.http.Interceptor;
9   import org.davidmoten.oa3.codegen.http.Serializer;
10  import org.davidmoten.oa3.codegen.runtime.Config;
11  
12  public final class ClientBuilder<T> {
13  
14      private final Function<ClientBuilder<T>, T> creator;
15      private final String basePath;
16      private final List<Interceptor> interceptors;
17      private Serializer serializer;
18  
19      public ClientBuilder(Function<ClientBuilder<T>, T> creator, Config config, String basePath) {
20          this.creator = creator;
21          this.serializer = new DefaultSerializer(config.mapper());
22          this.interceptors = new ArrayList<>();
23          this.basePath = trimAndRemoveFinalSlash(basePath);
24      }
25  
26      public ClientBuilder<T> serializer(Serializer serializer) {
27          this.serializer = serializer;
28          return this;
29      }
30  
31      /**
32       * Enables modification of all http requests (url, method, headers, not the
33       * body). Particularly useful for defining authentication (which normally
34       * involves just the addition of an Authorization header).
35       * 
36       * @param interceptor modifies http requests
37       * @return this
38       */
39      public ClientBuilder<T> interceptor(Interceptor interceptor) {
40          this.interceptors.add(interceptor);
41          return this;
42      }
43      
44      public ClientBuilder<T> interceptors(Iterable<? extends Interceptor> list) {
45          list.forEach(x -> interceptor(x));
46          return this;
47      }
48  
49      public Serializer serializer() {
50          return serializer;
51      }
52  
53      public List<Interceptor> interceptors() {
54          return interceptors;
55      }
56  
57      public String basePath() {
58          return basePath;
59      }
60  
61      public T build() {
62          return creator.apply(this);
63      }
64  
65      private static String trimAndRemoveFinalSlash(String s) {
66          s = s.trim();
67          if (s.endsWith("/")) {
68              return s.substring(0, s.length() - 1);
69          } else {
70              return s;
71          }
72      }
73  
74  }
