60          Map<String, TypedObject> _parameters = ParameterMap
61              .put("quality", "microsoft.graph.teleconferenceDeviceQuality", quality)
62              .build();
63          return new ActionRequestNoReturn(this.contextPath.addSegment("microsoft.graph.logTeleconferenceDeviceQuality"), _parameters);
