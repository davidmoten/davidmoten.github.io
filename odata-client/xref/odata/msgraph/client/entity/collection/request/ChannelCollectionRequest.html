13  public class ChannelCollectionRequest extends CollectionPageEntityRequest<Channel, ChannelRequest>{
14  
15      protected ContextPath contextPath;
16  
17      public ChannelCollectionRequest(ContextPath contextPath) {
18          super(contextPath, Channel.class, cp -> new ChannelRequest(cp), SchemaInfo.INSTANCE);
19          this.contextPath = contextPath;
20      }
21  
22      public ChatMessageCollectionRequest messages() {
23          return new ChatMessageCollectionRequest(contextPath.addSegment("messages"));
24      }
25  
26      public ChatMessageRequest messages(String id) {
27          return new ChatMessageRequest(contextPath.addSegment("messages").addKeys(new NameValue(id.toString())));
28      }
29  
30      public TeamsTabCollectionRequest tabs() {
31          return new TeamsTabCollectionRequest(contextPath.addSegment("tabs"));
32      }
33  
34      public TeamsTabRequest tabs(String id) {
35          return new TeamsTabRequest(contextPath.addSegment("tabs").addKeys(new NameValue(id.toString())));
36      }
