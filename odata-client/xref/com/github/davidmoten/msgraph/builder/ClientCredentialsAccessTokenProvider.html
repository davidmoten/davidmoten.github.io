99      @Override
100     public synchronized String get() {
101         long now = System.currentTimeMillis();
102         if (accessToken != null && now < expiryTime - refreshBeforeExpiryMs) {
103             return accessToken;
104         } else {
105             return refreshAccessToken();
106         }
107     }

109     private String refreshAccessToken() {
110 
111         // post some parameters in json format to the access token url
112         // and record returned expiry information so they we know when we
113         // need to refresh the token
114         try {
115             log.debug("refreshing access token");
116             URL url = new URL(graphEndpoint + tenantName + OAUTH2_TOKEN_URL_SUFFIX);
117             final HttpsURLConnection con;
118             if (proxyHost.isPresent()) {
119                 InetSocketAddress proxyInet = new InetSocketAddress(proxyHost.get(),
120                         proxyPort.get());
121                 Proxy proxy = new Proxy(Proxy.Type.HTTP, proxyInet);
122                 con = (HttpsURLConnection) url.openConnection(proxy);
123                 if (proxyUsername.isPresent()) {
124                     String usernameAndPassword = proxyUsername.get() + ":" + proxyPassword.get();
125                     String authString = "Basic " + Base64.getEncoder()
126                             .encode(usernameAndPassword.getBytes(StandardCharsets.UTF_8));
127                     con.setRequestProperty("Proxy-Authorization", authString);
128                 }
129                 // TODO support NTLM?
130             } else {
131                 con = (HttpsURLConnection) url.openConnection();
132             }
133             con.setConnectTimeout((int) connectTimeoutMs);
134             con.setReadTimeout((int) readTimeoutMs);
135             con.setRequestMethod(POST);
136             con.setRequestProperty(REQUEST_HEADER, APPLICATION_JSON);
137             StringBuilder params = new StringBuilder();
138             add(params, PARAMETER_RESOURCE, RESOURCE_MS_GRAPH);
139             add(params, PARAMETER_CLIENT_ID, clientId);
140             add(params, PARAMETER_GRANT_TYPE, GRANT_TYPE_CLIENT_CREDENTIALS);
141             add(params, PARAMETER_CLIENT_SECRET, clientSecret);
142             add(params, PARAMETER_SCOPE, scope);
143             con.setDoOutput(true);
144             try (DataOutputStream dos = new DataOutputStream(con.getOutputStream())) {
145                 dos.writeBytes(params.toString());
146             }
147             int responseCode = con.getResponseCode();
148 
149             String json = Util.readString(con.getInputStream(), StandardCharsets.UTF_8);
150 
151             if (responseCode != OK) {
152                 throw new IOException("Response code=" + responseCode + ", output=" + json);
153             } else {
154                 ObjectMapper om = new ObjectMapper();
155                 JsonNode o = om.readTree(json);
156                 // update the cached values
157                 expiryTime = o.get("expires_on").asLong() * 1000;
158                 accessToken = o.get("access_token").asText();
159                 log.debug("refreshed access token");
160                 return accessToken;
161             }
162         } catch (IOException e) {
163             // reset stuff
164             expiryTime = 0;
165             accessToken = null;
166             throw new RuntimeException(e);
167         }
168     }
169 
170     private static void add(StringBuilder params, String key, String value) {
171         if (params.length() > 0) {
172             params.append("&");
173         }
174         params.append(key);
175         params.append("=");
176         try {
177             params.append(URLEncoder.encode(value, "UTF-8"));
178         } catch (UnsupportedEncodingException e) {
179             throw new RuntimeException(e);
180         }
181     }

234          * {@code get()} within {@code duration} of the expiry time then a refresh of

236          * token is refreshed on every call of {@code get()}.
