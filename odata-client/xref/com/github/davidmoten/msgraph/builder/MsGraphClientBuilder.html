364     private static CloseableHttpClient createHttpClient(long connectTimeoutMs, long readTimeoutMs,
365             Optional<String> proxyHost, Optional<Integer> proxyPort, Optional<String> proxyUsername,
366             Optional<String> proxyPassword,
367             Optional<Function<HttpClientBuilder, HttpClientBuilder>> httpClientBuilderExtras) {
368         RequestConfig config = RequestConfig.custom() //
369                 .setConnectTimeout((int) connectTimeoutMs) //
370                 .setSocketTimeout((int) readTimeoutMs) //
371                 .build();
372         
373         PoolingHttpClientConnectionManager cm = new PoolingHttpClientConnectionManager();
374         // Set soTimeout here to affect socketRead in the phase of ssl handshake. Note that
375         // the RequestConfig.setSocketTimeout will take effect only after the ssl handshake completed.
376         cm.setDefaultSocketConfig(SocketConfig.custom().setSoTimeout((int) readTimeoutMs).build());
377         
378         HttpClientBuilder b = HttpClientBuilder //
379                 .create() //
380                 .useSystemProperties() //
381                 .setDefaultRequestConfig(config) //
382                 .setConnectionManager(cm);
383         
384         if (proxyHost.isPresent()) {
385             HttpHost proxy = new HttpHost(proxyHost.get(), proxyPort.get());
386             if (proxyUsername.isPresent()) {
387                 Credentials credentials = new UsernamePasswordCredentials(proxyUsername.get(),
388                         proxyPassword.orElse(null));
389                 AuthScope authScope = new AuthScope(proxyHost.get(), proxyPort.get());
390                 CredentialsProvider credentialsProvider = new BasicCredentialsProvider();
391                 credentialsProvider.setCredentials(authScope, credentials);
392                 b = b.setDefaultCredentialsProvider(credentialsProvider);
393             }
394             b = b.setProxy(proxy);
395         }
396         if (httpClientBuilderExtras.isPresent()) {
397             b = httpClientBuilderExtras.get().apply(b);
398         }
399         return b.build();
400     }
