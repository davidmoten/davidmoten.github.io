3   import java.util.concurrent.TimeUnit;

5   import com.github.davidmoten.msgraph.builder.GraphExplorerHttpService;
6   import com.github.davidmoten.msgraph.builder.MsGraphClientBuilder;
7   import com.github.davidmoten.msgraph.builder.MsGraphClientBuilder.Builder3;
8   
9   import odata.msgraph.client.beta.container.GraphService;
10  
11  public final class MsGraph {
12  
13      private static final String MSGRAPH_BETA_BASE_URL = "https://graph.microsoft.com/beta";
14  
15      private MsGraph() {
16          // prevent instantiation
17      }
18  
19      public static MsGraphClientBuilder.Builder<GraphService> tenantName(String tenantName) {
20          return new MsGraphClientBuilder<GraphService>(MSGRAPH_BETA_BASE_URL,
21                  context -> new GraphService(context)).tenantName(tenantName);
22      }
23      
24      public static Builder3<GraphService> explorer() {
25          return MsGraph //
26                  .tenantName("unused") //
27                  .clientId("unused") //
28                  .clientSecret("unused") //
29                  .refreshBeforeExpiry(5, TimeUnit.MINUTES) //
30                  .accessTokenProvider(() -> "{token:https://graph.microsoft.com/}") //
31                  .httpServiceTransformer(s -> new GraphExplorerHttpService(s));
32      }
33  }
