3   import java.util.List;
4   import java.util.stream.Collectors;
5   
6   import org.oasisopen.odata.csdl.v4.TEntityKeyElement;
7   
8   import com.github.davidmoten.odata.client.generator.Names;
9   
10  public class KeyElement {
11  
12      private final TEntityKeyElement keyElement;
13      private final EntityType entityType;
14      private final Names names;
15  
16      public KeyElement(TEntityKeyElement keyElement, EntityType entityType, Names names) {
17          this.keyElement = keyElement;
18          this.entityType = entityType;
19          this.names = names;
20      }
21  
22      public List<PropertyRef> getPropertyRefs() {
23          return keyElement.getPropertyRef().stream().map(x -> new PropertyRef(x, entityType, names))
24                  .collect(Collectors.toList());
25      }

27  }
