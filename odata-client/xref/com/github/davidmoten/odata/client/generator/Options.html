5   public final class Options {
6   
7       private final String outputDirectory;
8   
9       private final List<SchemaOptions> schemaOptions;
10  
11      public Options(String outputDirectory, List<SchemaOptions> schemaOptions) {
12          this.outputDirectory = outputDirectory;
13          this.schemaOptions = schemaOptions;
14      }
15  
16      public String getOutputDirectory() {
17          return outputDirectory;
18      }
19  
20      public SchemaOptions getSchemaOptions(String namespace) {
21          return schemaOptions //
22                  .stream() //
23                  .filter(x -> namespace.equals(x.namespace)) //
24                  .findFirst() //
25                  .<IllegalArgumentException>orElseThrow(() -> {
26                      throw new IllegalArgumentException(
27                              "namespace not found in schemaOptions: " + namespace);
28                  });
29      }
