19  public final class Action implements Method {
20  
21      private TAction action;
22      private Names names;
23  
24      public Action(TAction action, Names names) {
25          this.action = action;
26          this.names = names;
27      }
28  
29      public String getSimpleClassNameActionRequest() {
30          return names.getSimpleClassNameActionRequest(schema(), action.getName());
31      }
32  
33      public Schema schema() {
34          return names.getSchema(action);
35      }
36  
37      public String getFullClassNameActionRequest() {
38          return names.getFullClassNameActionRequest(schema(), action.getName());
39      }
40  
41      public Object getPackageActionRequest() {
42          return names.getPackageActionRequest(schema());
43      }
44  
45      public String getFullClassNameActionReturnType() {
46          Optional<TActionFunctionReturnType> returnParameter = Util
47                  .filter(action.getParameterOrAnnotationOrReturnType(),
48                          TActionFunctionReturnType.class)
49                  .findFirst();
50          if (!returnParameter.isPresent()) {
51              return Void.class.getName();
52          } else {
53              return names.getFullClassNameFromTypeWithNamespace(
54                      names.getInnerType(returnParameter.get()));
55          }
56      }
57  
58      public File getClassFileActionRequest() {
59          return names.getClassFileActionRequest(schema(), action.getName());
60      }
61  
62      public Optional<String> getActionReturnType() {
63          return Util
64                  .filter(action.getParameterOrAnnotationOrReturnType(),
65                          TActionFunctionReturnType.class) //
66                  .map(x -> names.getInnerType(x)) //
67                  .findFirst();
68      }
69  
70      public Optional<String> getBoundTypeWithNamespace() {
71          if (!action.isIsBound()) {
72              return Optional.empty();
73          } else {
74              // TODO what does EntitySetPath mean?
75              return Util
76                      .filter(action.getParameterOrAnnotationOrReturnType(),
77                              TActionFunctionParameter.class)
78                      .map(x -> names.getInnerType(x)) //
79                      .findFirst();
80          }
81      }
82  
83      public String getName() {
84          return action.getName();
85      }
86  
87      public String getActionMethodName() {
88          return Names.getIdentifier(action.getName());
89      }
90  
91      public List<Parameter> getParametersUnbound(Imports imports) {
92          AtomicBoolean first = new AtomicBoolean(true);
93          return Util
94                  .filter(action.getParameterOrAnnotationOrReturnType(),
95                          TActionFunctionParameter.class) //
96                  .filter(x -> !action.isIsBound() || !first.getAndSet(false)) //
97                  .map(x -> new Parameter(x, names, imports)) //
98                  .collect(Collectors.toList());
99      }

101     public static final class Parameter implements HasNameJavaHasNullable {
102         public final String name;
103         private final String nameJava;
104         public final String importedFullClassName;
105 
106         public final boolean isCollection;
107         public final String typeWithNamespace;
108 
109         private final boolean isNullable;
110 
111         public Parameter(TActionFunctionParameter p, Names names, Imports imports) {
112             this.name = p.getName();
113             this.nameJava = Names.getIdentifier(p.getName());
114             this.importedFullClassName = names.toImportedFullClassName(p, imports);
115             this.isCollection = names.isCollection(p);
116             this.typeWithNamespace = p.getType().get(0);
117             this.isNullable = p.isNullable() == null ? true : p.isNullable();
118         }
119 
120         @Override
121         public String nameJava() {
122             return nameJava;
123         }
124 
125         @Override
126         public boolean isNullable() {
127             return isNullable;
128         }
129     }
130 
131     public static final class ReturnType {
132         public final String innerImportedFullClassName;
133         public final boolean isCollection;
134 
135         public final String schemaInfoFullClassName;
136 
137         public ReturnType(boolean isCollection, String innerImportedFullClassName,
138                 String schemaInfoFullClassName) {
139             this.isCollection = isCollection;
140             this.innerImportedFullClassName = innerImportedFullClassName;
141             this.schemaInfoFullClassName = schemaInfoFullClassName;
142         }
143     }
144 
145     public boolean hasReturnType() {
146         return Util
147                 .filter(action.getParameterOrAnnotationOrReturnType(),
148                         TActionFunctionReturnType.class) //
149                 .findFirst() //
150                 .isPresent();
151     }
152 
153     public ReturnType getReturnType(Imports imports) {
154         return Util
155                 .filter(action.getParameterOrAnnotationOrReturnType(),
156                         TActionFunctionReturnType.class) //
157                 .findFirst() //
158                 .map(x -> {
159                     String innerType = names.getInnerType(x);
160                     final String schemaInfoClassName;
161                     if (innerType.startsWith("Edm.")) {
162                         schemaInfoClassName = EdmSchemaInfo.INSTANCE.getClass().getCanonicalName();
163                     } else {
164                         schemaInfoClassName = names
165                                 .getFullClassNameSchemaInfo(names.getSchema(innerType));
166                     }
167 
168                     return new ReturnType( //
169                             names.isCollection(x), //
170                             names.toImportedTypeNonCollection(names.getInnerType(x), imports), //
171                             schemaInfoClassName);
172                 }) //
173                 .get();
174     }
175 
176     public String getFullType() {
177         return names.getFullTypeFromSimpleType(schema(), action.getName());
178     }
179 
180     public boolean isBoundToCollection() {
181         return getBoundType().map(x -> Names.isCollection(x)).orElse(false);
182     }
183 
184     public Optional<String> getBoundType() {
185         if (!action.isIsBound()) {
186             return Optional.empty();
187         } else {
188             TActionFunctionParameter p = Util.filter(action.getParameterOrAnnotationOrReturnType(),
189                     TActionFunctionParameter.class).findFirst().get();
190             return Optional.of(p.getType().get(0));
191         }
192     }
