1   package com.github.davidmoten.odata.client.generator.model;

3   import java.util.List;
4   import java.util.Optional;
5   import java.util.stream.Collectors;
6   
7   import org.oasisopen.odata.csdl.v4.TAnnotations;
8   
9   public final class Annotations {
10  
11      private final TAnnotations annotations;
12      private List<Annotation> list;
13  
14      public Annotations(TAnnotations annotations) {
15          this.annotations = annotations;
16          this.list = annotations //
17                  .getAnnotation() //
18                  .stream() //
19                  .map(x -> new Annotation(x)) //
20                  .collect(Collectors.toList());
21      }
22  
23      public TAnnotations value() {
24          return annotations;
25      }
26  
27      public Optional<String> getValue(String annotationType) {
28          return annotations //
29                  .getAnnotation() //
30                  .stream() //
31                  .filter(x -> annotationType.equals(x.getTerm())) //
32                  .map(x -> x.getString()) //
33                  .filter(x -> x != null) //
34                  .findFirst();
35      }
36  
37      public List<Annotation> getValues() {
38          return list;
39      }
40  
41  }
