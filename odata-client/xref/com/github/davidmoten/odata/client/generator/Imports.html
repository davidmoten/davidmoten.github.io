3   import java.util.HashMap;
4   import java.util.Map;
5   import java.util.function.Function;
6   import java.util.stream.Collectors;
7   
8   public final class Imports {
9   
10      private final String fullClassName;
11  
12      Imports(String fullClassName) {
13          this.fullClassName = fullClassName;
14          add(fullClassName);
15      }
16  
17      // simpleName to fullClassName
18      private final Map<String, String> map = new HashMap<>();
19  
20      public String add(Class<?> cls) {
21          return add(cls.getCanonicalName().replace("$", "."));
22      }
23  
24      public String add(String className) {
25          final String simpleName = simpleName(className);
26          String c = map.get(simpleName);
27          if (c == null) {
28              map.put(simpleName, className);
29              return simpleName;
30          } else if (c.equals(className)) {
31              return simpleName;
32          } else {
33              return className;
34          }
35      }
36  
37      private static String simpleName(String className) {
38          final String simpleName;
39          int i = className.lastIndexOf('.');
40          if (i == -1) {
41              simpleName = className;
42          } else {
43              simpleName = className.substring(i + 1, className.length());
44          }
45          return simpleName;
46      }
47  
48      @Override
49      public String toString() {
50          String pkgFullClassName = pkg(fullClassName);
51          String x = map //
52                  .values() //
53                  .stream() //
54                  .sorted() //
55                  .filter(c -> !c.startsWith("java.lang.")) //
56                  .filter(c -> !c.equals("boolean")) //
57                  .filter(c -> !c.equals("short")) //
58                  .filter(c -> !c.equals("float")) //
59                  .filter(c -> !c.equals("double")) //
60                  .filter(c -> !c.equals("int")) //
61                  .filter(c -> !c.equals("byte")) //
62                  .filter(c -> !c.equals(fullClassName))
63                  // ensure that if in same pkg as fullClassName that we don't need
64                  // to specify an import
65                  .filter(c -> !pkg(c).equals(pkgFullClassName)) //
66                  .map(new Function<String,String>() {
67  
68                      String previous;
69                      
70                      @Override
71                      public String apply(String c) {
72                          String firstSegment = firstSegment(c);
73                          boolean insertBlankLine  = previous != null && !firstSegment.equals(previous);
74                          previous = firstSegment;
75                          return (insertBlankLine? "\n" : "") + "import " + c + ";";
76                      }
77                  }) //
78                  .collect(Collectors.joining("\n"));
79          if (!x.isEmpty()) {
80              x = x + "\n\n";
81          }
82          return x;
83      }
84      
85      private static String pkg(String fullClassName) {
86          int i = fullClassName.lastIndexOf('.');
87          if (i == -1) {
88              return "";
89          } else {
90              return fullClassName.substring(0, i);
91          }
92      }
93      
94      private static String firstSegment(String s) {
95          int i = s.indexOf('.');
96          if (i == -1) {
97              return s;
98          } else {
99              return s.substring(0, i);
100         }
101     }
