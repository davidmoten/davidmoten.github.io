2   
3   import java.util.HashMap;
4   import java.util.List;
5   import java.util.Map;
6   import java.util.Optional;
7   
8   public final class CollectionRequestOptions implements RequestOptions {
9   
10      private final List<RequestHeader> requestHeaders;
11      private final Optional<String> search;
12      private final Optional<String> filter;
13      private final Optional<String> orderBy;
14      private final Optional<Long> skip;
15      private final Optional<Long> top;
16      private final Optional<String> select;
17      private final Optional<String> expand;
18      private final Optional<String> urlOverride;
19  
20      public CollectionRequestOptions(List<RequestHeader> requestHeaders, Optional<String> search,
21              Optional<String> filter, Optional<String> orderBy, Optional<Long> skip,
22              Optional<Long> top, Optional<String> select, Optional<String> expand, Optional<String> urlOverride) {
23          this.requestHeaders = requestHeaders;
24          this.search = search;
25          this.filter = filter;
26          this.orderBy = orderBy;
27          this.skip = skip;
28          this.top = top;
29          this.select = select;
30          this.expand = expand;
31          this.urlOverride = urlOverride;
32      }
33  
34      @Override
35      public List<RequestHeader> getRequestHeaders() {
36          return requestHeaders;
37      }
38  
39      @Override
40      public Map<String, String> getQueries() {
41          Map<String, String> map = new HashMap<>();
42          search.ifPresent(x -> map.put("$search", x));
43          filter.ifPresent(x -> map.put("$filter", x));
44          orderBy.ifPresent(x -> map.put("$orderBy", x));
45          skip.ifPresent(x -> map.put("$skip", String.valueOf(x)));
46          top.ifPresent(x -> map.put("$top", String.valueOf(x)));
47          select.ifPresent(x -> map.put("$select", x));
48          expand.ifPresent(x -> map.put("$expand", x));
49          return map;
50      }
51  
52      @Override
53      public Optional<String> getUrlOverride() {
54          return urlOverride;
55      }
56      
57  }
