3   import java.util.Collections;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   public final class Context {
8   
9       private final Serializer serializer;
10      private final HttpService service;
11  
12      // used to specify options that override standard odata behaviour to deal with
13      // service bugs etc. For example, MsGraph @odata.mediaEditLink is wrong for an
14      // fileAttachment and needs special editing to get working until Microsoft fix
15      // this bug.
16      private final Map<String, Object> properties;
17  
18      public Context(Serializer serializer, HttpService service, Map<String, Object> properties) {
19          this.serializer = serializer;
20          this.service = service;
21          this.properties = new HashMap<>(properties);
22      }
23  
24      public Context(Serializer serializer, HttpService service) {
25          this(serializer, service, Collections.emptyMap());
26      }
27  
28      public Serializer serializer() {
29          return serializer;
30      }
31  
32      public HttpService service() {
33          return service;
34      }
35  
36      public Object getProperty(String name) {
37          return properties.get(name);
38      }
39  
40      public Map<String, Object> getProperties() {
41          return new HashMap<>(properties);
42      }
43  
44  }
