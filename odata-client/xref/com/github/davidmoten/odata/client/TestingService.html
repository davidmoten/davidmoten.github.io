43      public static final class Response {
44          public String resource;
45          public final int statusCode;
46  
47          public Response(String resource, int statusCode) {
48              this.resource = resource;
49              this.statusCode = statusCode;
50          }
51          
52          public Response(String text) {
53              this(text, HttpURLConnection.HTTP_OK);
54          }
55  
56          @Override
57          public String toString() {
58              return "Response [resource=" + resource + ", statusCode=" + statusCode + "]";

62      public static abstract class BuilderBase<T extends BuilderBase<?, R>, R> {
63          Map<String, Response> responses = new HashMap<>();
64          Map<String, String> requests = new HashMap<>();
65  
66          String baseUrl = "https://testing.com";
67          PathStyle pathStyle = PathStyle.IDENTIFIERS_AS_SEGMENTS;
68  
69          @SuppressWarnings("unchecked")
70          public T baseUrl(String baseUrl) {
71              this.baseUrl = baseUrl;
72              return (T) this;
73          }
74  
75          @SuppressWarnings("unchecked")
76          public T pathStyle(PathStyle pathStyle) {
77              this.pathStyle = pathStyle;
78              return (T) this;
79          }
80  
81          public T expectResponse(String path, String responseResourceName,
82                  RequestHeader... requestHeaders) {
83              return expectResponse(path, responseResourceName, HttpMethod.GET, requestHeaders);
84          }
85  
86          @SuppressWarnings("unchecked")
87          public T expectResponse(String path, String responseResourceName, HttpMethod method, int statusCode,
88                  RequestHeader... requestHeaders) {
89              responses.put(toKey(method, baseUrl + path, asList(requestHeaders)),
90                      new Response(responseResourceName, statusCode));
91              return (T) this;
92          }
93          
94          public T expectResponse(String path, String responseResourceName, HttpMethod method,
95                  RequestHeader... requestHeaders) {
96              return expectResponse(path, responseResourceName, method, HttpURLConnection.HTTP_OK, requestHeaders);
97          }
98  
99          @SuppressWarnings("unchecked")
100         public T expectRequest(String path, String requestResourceName, HttpMethod method,
101                 RequestHeader... requestHeaders) {
102             Preconditions.checkArgument(method != HttpMethod.GET,
103                     "GET not expected for a request with content");
104             requests.put(toKey(method, baseUrl + path, asList(requestHeaders)),
105                     requestResourceName);
106             return (T) this;
107         }
108 
109         @SuppressWarnings("unchecked")
110         public T expectRequestAndResponse(String path, String requestResourceName,
111                 String responseResourceName, HttpMethod method, RequestHeader... requestHeaders) {
112             requests.put(toKey(method, baseUrl + path, asList(requestHeaders)),
113                     requestResourceName);
114             responses.put(toKey(method, baseUrl + path, asList(requestHeaders)),
115                     new Response(responseResourceName));
116             return (T) this;
117         }
118 
119         @SuppressWarnings("unchecked")
120         public T expectDelete(String path, RequestHeader... requestHeaders) {
121             requests.put(toKey(HttpMethod.DELETE, baseUrl + path, asList(requestHeaders)),
122                     "DELETE");
123             return (T) this;
124         }

126         private static String toKey(HttpMethod method, String url,
127                 Collection<RequestHeader> requestHeaders) {
128             return method + "\n  " + url + "\n  " + requestHeaders.stream()
129                     .map(x -> x.name() + "=" + x.value()).sorted().collect(Collectors.joining("|"));
130         }

132         private static final void log(Object o) {

136         protected HttpService createService() {
137             return new HttpService() {
138 
139                 @Override
140                 public HttpResponse get(String url, List<RequestHeader> requestHeaders) {
141                     log("Available responses:");
142                     responses.entrySet().forEach(r -> log(r.getKey() + "\n=>" + r.getValue()));
143                     String key = BuilderBase.toKey(HttpMethod.GET, url, requestHeaders);
144                     log("Getting:\n" + key);
145                     Response response = responses.get(key);
146                     String resourceName = response == null? null: response.resource;
147                     if (resourceName == null) {
148                         throw new RuntimeException("GET response not found for url=" + url
149                                 + ", headers=" + requestHeaders);
150                     }
151                     try {
152                         URL resource = TestingService.class.getResource(resourceName);
153                         if (resource == null) {
154                             throw new RuntimeException(
155                                     "resource not found on classpath: " + resourceName);
156                         }
157                         String text = new String(Files.readAllBytes(Paths.get(resource.toURI())));
158                         return new HttpResponse(response.statusCode, text);
159                     } catch (IOException | URISyntaxException e) {
160                         throw new RuntimeException(e);
161                     }
162                 }
163 
164                 @Override
165                 public HttpResponse patch(String url, List<RequestHeader> requestHeaders,
166                         InputStream content) {
167                     log("PATCH called at " + url);
168                     String text = Util.utf8(content);
169                     log(text);
170                     log("Available requests:");
171                     requests.entrySet().forEach(r -> log(r.getKey() + "\n=>" + r.getValue()));
172                     log("Calling:");
173                     String key = BuilderBase.toKey(HttpMethod.PATCH, url, requestHeaders);
174                     log(key);
175                     String resourceName = requests.get(key);
176                     if (resourceName == null) {
177                         throw new RuntimeException("PATCH response not found for url=" + url
178                                 + ", headers=" + requestHeaders);
179                     }
180                     try {
181                         String expected = new String(Files.readAllBytes(
182                                 Paths.get(TestingService.class.getResource(resourceName).toURI())));
183 
184                         if (Serializer.INSTANCE.matches(expected, text)) {
185                             return new HttpResponse(HttpURLConnection.HTTP_NO_CONTENT, null);
186                         } else {
187                             throw new RuntimeException(
188                                     "request does not match expected.\n==== Recieved ====\n" + text
189                                             + "\n==== Expected =====\n" + expected);
190                         }
191                     } catch (IOException | URISyntaxException e) {
192                         throw new RuntimeException(e);
193                     }
194                 }
195 
196                 @Override
197                 public HttpResponse put(String url, List<RequestHeader> h, InputStream content) {
198                     return patch(url, h, content);
199                 }
200 
201                 @Override
202                 public HttpResponse post(String url, List<RequestHeader> requestHeaders,
203                         InputStream content) {
204                     log("POST called at " + url);
205                     String text = Util.utf8(content);
206                     log(text);
207                     requests.entrySet().forEach(r -> log(r.getKey() + "\n=>" + r.getValue()));
208                     log("Calling:");
209                     String key = BuilderBase.toKey(HttpMethod.POST, url, requestHeaders);
210                     log(key);
211                     String requestResourceName = requests.get(key);
212                     if (requestResourceName == null) {
213                         throw new RuntimeException("POST request not expected for url=" + url
214                                 + ", headers=" + requestHeaders);
215                     }
216                     try {
217                         String requestExpected = readResource(url, requestResourceName);
218                         if (Serializer.INSTANCE.matches(requestExpected, text)) {
219                             String responseResourceName = responses
220                                     .get(BuilderBase.toKey(HttpMethod.POST, url, requestHeaders)).resource;
221                             String responseExpected = readResource(url, responseResourceName);
222                             int responseCode = url.contains("delta") ? HttpURLConnection.HTTP_OK
223                                     : HttpURLConnection.HTTP_CREATED;
224                             return new HttpResponse(responseCode, responseExpected);
225                         } else {
226                             throw new RuntimeException(
227                                     "request does not match expected.\n==== Received ====\n" + text
228                                             + "\n==== Expected =====\n" + requestExpected);
229                         }
230                     } catch (IOException | URISyntaxException e) {
231                         throw new RuntimeException(e);
232                     }
233                 }
234 
235                 @Override
236                 public HttpResponse delete(String url, List<RequestHeader> requestHeaders) {
237                     log("DELETE called at " + url);
238                     requests.entrySet().forEach(r -> log(r.getKey() + "\n=>" + r.getValue()));
239                     log("Calling:");
240                     String key = BuilderBase.toKey(HttpMethod.DELETE, url, requestHeaders);
241                     log(key);
242 
243                     String resourceName = requests.get(key);
244                     if (resourceName == null) {
245                         throw new RuntimeException("DELETE request not expected for url=" + url
246                                 + ", headers=" + requestHeaders);
247                     }
248                     return new HttpResponse(HttpURLConnection.HTTP_NO_CONTENT, null);
249                 }
250 
251                 @Override
252                 public Path getBasePath() {
253                     return new Path(baseUrl, pathStyle);
254                 }
255 
256                 @Override
257                 public void close() throws Exception {
258                     // do nothing
259                 }
260 
261                 @Override
262                 public InputStream getStream(String url, List<RequestHeader> requestHeaders) {
263                     HttpResponse h = get(url, requestHeaders);
264                     return new ByteArrayInputStream(h.getText().getBytes(StandardCharsets.UTF_8));
265                 }
266 
267             };

299     private static String readResource(String url, String resourceName)
300             throws IOException, URISyntaxException {
301         if (resourceName == null) {
302             throw new RuntimeException("resource not found for url=" + url);
303         }
304         return new String(Files
305                 .readAllBytes(Paths.get(TestingService.class.getResource(resourceName).toURI())));
