3   import java.util.Collections;
4   import java.util.LinkedHashMap;
5   import java.util.Map;
6   import java.util.stream.Collectors;

8   public final class ParameterMap {
9   
10      public static Builder put(String key, String typeWithNamespace, Object value) {
11          return new Builder().put(key, typeWithNamespace, value);
12      }
13  
14      public static Map<String, TypedObject> empty() {
15          return Collections.emptyMap();
16      }
17  
18      public static Map<String, TypedObject> build() {
19          return new Builder().build();
20      }
21  
22      public static final class Builder {
23          private final Map<String, TypedObject> map = new LinkedHashMap<>();
24  
25          public Builder put(String key, String typeWithNamespace, Object value) {
26              map.put(key, new TypedObject(typeWithNamespace, value));
27              return this;
28          }
29  
30          public Map<String, TypedObject> build() {
31              return map;
32          }
33  
34      }
35  
36      public static Map<String, Object> toMap(Map<String, TypedObject> map) {
37          return map //
38                  .entrySet() //
39                  .stream() //
40                  .collect(Collectors.toMap( //
41                          entry -> entry.getKey(), //
42                          entry -> entry.getValue().object()));
43      }

45  }
