48      /**
49       * Returns the json from an HTTP GET of the url built from the contextPath and
50       * options. In the case where the returned object is actually a sub-class of T
51       * we lookup the sub-class from schemaInfo based on the namespaced type of the
52       * return object.
53       * 
54       * @param <T>              return object type
55       * @param contextPath      context and current path
56       * @param returnCls        return class
57       * @param options          request options
58       * @param returnSchemaInfo schema to be used for lookup generated class of of
59       *                         the returned object from the namespaced type
60       * @return object hydrated from json
61       */
62      public static <T> T get(ContextPath contextPath, Class<T> returnCls, RequestOptions options,
63              SchemaInfo returnSchemaInfo) {
64          // build the url
65          ContextPath cp = contextPath.addQueries(options.getQueries());
66  
67          List<RequestHeader> h = cleanAndSupplementRequestHeaders(options, "minimal", false);
68  
69          // get the response
70          HttpResponse response = cp.context().service().get(cp.toUrl(), h);
71  
72          checkResponseCode(cp, response, HttpURLConnection.HTTP_OK);
73  
74          // deserialize
75          // Though cls might be Class<Attachment> we might actually want to return a
76          // sub-class like FileAttachment (which extends Attachment). This method returns
77          // the actual sub-class by inspecting the json response.
78          Class<? extends T> c = getSubClass(cp, returnSchemaInfo, returnCls, response.getText());
79          // check if we need to deserialize into a subclass of T (e.g. return a
80          // FileAttachment which is a subclass of Attachment)
81          return cp.context().serializer().deserialize(response.getText(), c, contextPath, false);
82      }

84      public static void checkResponseCode(String url, HttpResponse response,
85              int expectedResponseCodeMin, int expectedResponseCodeMax) {
86          if (response.getResponseCode() < expectedResponseCodeMin
87                  || response.getResponseCode() > expectedResponseCodeMax) {
88              throw new ClientException(response.getResponseCode(), "responseCode=" + response.getResponseCode() + " from url="
89                      + url + ", expectedResponseCode in [" + expectedResponseCodeMin + ", "
90                      + expectedResponseCodeMax + "], message=\n" + response.getText());
91          }
92      }

210     @SuppressWarnings("unused")
211     private static <T extends ODataEntityType> T patchOrPut(T entity, ContextPath contextPath,
212             RequestOptions options, HttpMethod method) {
213         Preconditions.checkArgument(method == HttpMethod.PUT || method == HttpMethod.PATCH);
214         final String json;
215         if (method == HttpMethod.PATCH) {
216             json = Serializer.INSTANCE.serializeChangesOnly(entity);
217         } else {
218             json = Serializer.INSTANCE.serialize(entity);
219         }
220 
221         // build the url
222         ContextPath cp = contextPath.addQueries(options.getQueries());
223 
224         List<RequestHeader> h = cleanAndSupplementRequestHeaders(options, "minimal", true);
225 
226         final String url;
227         String editLink = (String) entity.getUnmappedFields().get("@odata.editLink");
228         // TODO get patch working when editLink present (does not work with MsGraph)
229         if (editLink != null && false) {
230             if (editLink.startsWith(HTTPS) || editLink.startsWith("http://")) {
231                 url = editLink;
232             } else {
233                 // TOOD unit test relative url in editLink
234                 // from
235                 // http://docs.oasis-open.org/odata/odata-json-format/v4.01/cs01/odata-json-format-v4.01-cs01.html#_Toc499720582
236                 String context = (String) entity.getUnmappedFields().get("@odata.context");
237                 if (context != null) {
238                     try {
239                         URL u = new URL(context);
240                         String p = u.getPath();
241                         String basePath = p.substring(0, p.lastIndexOf('/'));
242                         url = basePath + "/" + editLink;
243                     } catch (MalformedURLException e) {
244                         throw new ClientException(e);
245                     }
246                 } else {
247                     url = cp.context().service().getBasePath().toUrl() + "/" + editLink;
248                 }
249             }
250         } else {
251             url = cp.toUrl();
252         }
253         // get the response
254         HttpService service = cp.context().service();
255         final HttpResponse response = service.submitWithContent(method, url, h, json);
256         checkResponseCode(cp, response, HTTP_OK_MIN, HTTP_OK_MAX);
257         // TODO is service returning the entity that we should use rather than the
258         // original?
259         return entity;
260     }

271     public static <T> Class<? extends T> getSubClass(ContextPath cp, SchemaInfo schemaInfo,
272             Class<T> cls, String json) {
273         Optional<String> namespacedType = cp.context().serializer().getODataType(json)
274                 .map(x -> x.substring(1));
275 
276         if (namespacedType.isPresent()) {
277             return (Class<? extends T>) schemaInfo
278                     .getClassFromTypeWithNamespace(namespacedType.get());
279         } else {
280             return cls;
281         }
282     }

284     public static List<RequestHeader> cleanAndSupplementRequestHeaders(
285             List<RequestHeader> requestHeaders, String contentTypeOdataMetadataValue,
286             boolean isWrite) {
287 
288         List<RequestHeader> list = new ArrayList<>();
289         list.add(RequestHeader.ODATA_VERSION);
290         if (isWrite) {
291             list.add(RequestHeader.contentTypeJsonWithMetadata(contentTypeOdataMetadataValue));
292         }
293         list.add(RequestHeader.ACCEPT_JSON);
294         list.addAll(requestHeaders);
295 
296         // remove duplicates
297         List<RequestHeader> list2 = new ArrayList<>();
298         Set<RequestHeader> set = new HashSet<>();
299         for (RequestHeader r : list) {
300             if (!set.contains(r)) {
301                 list2.add(r);
302             }
303             set.add(r);
304         }
305 
306         // remove overriden accept header
307         if (list2.contains(RequestHeader.ACCEPT_JSON) && list2.stream()
308                 .filter(x -> x.isAcceptJsonWithMetadata()).findFirst().isPresent()) {
309             list2.remove(RequestHeader.ACCEPT_JSON);
310         }
311 
312         // only use the last accept with metadata request header
313         Optional<RequestHeader> m = list2 //
314                 .stream() //
315                 .filter(x -> x.isAcceptJsonWithMetadata()) //
316                 .reduce((x, y) -> y);
317 
318         List<RequestHeader> list3 = list2.stream()
319                 .filter(x -> !x.isAcceptJsonWithMetadata() || !m.isPresent() || x.equals(m.get()))
320                 .collect(Collectors.toList());
321         return list3;
322 
323     }

344             ODataEntityType entity) {
345         String editLink = (String) entity.getUnmappedFields().get("@odata.mediaEditLink");
346         if (editLink == null) {
347             editLink = (String) entity.getUnmappedFields().get("@odata.editLink");
348         }
349         String contentType = (String) entity.getUnmappedFields().get("@odata.mediaContentType");
350         if (editLink == null && "false"
351                 .equals(contextPath.context().getProperty("attempt.stream.when.no.metadata"))) {
352             return Optional.empty();
353         } else {
354             if (contentType == null) {
355                 contentType = CONTENT_TYPE_APPLICATION_OCTET_STREAM;
356             }
357             // TODO support relative editLink?
358             Context context = contextPath.context();
359             if (editLink == null) {
360                 editLink = contextPath.toUrl();
361             }
362             if (!editLink.startsWith(HTTPS)) {
363                 // TODO should use the base path from @odata.context field?
364                 editLink = concatenate(contextPath.context().service().getBasePath().toUrl(),
365                         editLink);
366             }
367             if ("true".equals(contextPath.context().getProperty("modify.stream.edit.link"))) {
368                 // Bug fix for Microsoft Graph only?
369                 // When a collection is returned the editLink is terminated with the subclass if
370                 // the collection type has subclasses. For example when a collection of
371                 // Attachment (with full metadata) is requested the editLink of an individual
372                 // attachment may end in /itemAttachment to indicate the type of the attachment.
373                 // To get the $value download working we need to remove that type cast.
374                 int i = endsWith(editLink, "/" + entity.odataTypeName());
375                 if (i == -1) {
376                     i = endsWith(editLink, "/" + entity.odataTypeName() + "/$value");
377                 }
378                 if (i == -1) {
379                     i = endsWith(editLink, "/" + entity.odataTypeName() + "/%24value");
380                 }
381                 if (i != -1) {
382                     editLink = editLink.substring(0, i);
383                 }
384             }
385             Pathten/odata/client/Path.html#Path">Path path = new Path(editLink, contextPath.path().style());
386             if (!path.toUrl().endsWith("/$value")) {
387                 path = path.addSegment("$value");
388             }
389             return Optional.of(new StreamProvider( //
390                     new ContextPath(context, path), //
391                     RequestOptions.EMPTY, //
392                     contentType, //
393                     null));

397     private static int endsWith(String a, String b) {
398         if (a.endsWith(b)) {
399             return a.length() - b.length();
400         } else {
401             return -1;
402         }

405     // concatenate two url parts making sure there is a / delimiter
406     private static String concatenate(String a, String b) {
407         StringBuilder s = new StringBuilder();
408         s.append(a);
409         if (a.endsWith("/")) {
410             if (b.startsWith("/")) {
411                 s.append(b, 1, b.length());
412             } else {
413                 s.append(b);
414             }
415         } else {
416             if (!b.startsWith("/")) {
417                 s.append('/');
418             }
419             s.append(b);
420         }
421         return s.toString();
422     }

466     public static String createUploadSession(ContextPath contextPath,
467             List<RequestHeader> requestHeaders, String contentType) {
468         List<RequestHeader> h = cleanAndSupplementRequestHeaders(requestHeaders, contentType, true);
469         ContextPath cp = contextPath.addSegment("createUploadSession");
470         HttpResponse response = contextPath //
471                 .context() //
472                 .service() //
473                 .post(cp.toUrl(), h, new ByteArrayInputStream(new byte[] {}));
474         checkResponseCode(cp, response, HTTP_OK_MIN, HTTP_OK_MAX);
475         ObjectMapper m = new ObjectMapper();
476         try {
477             JsonNode tree = m.readTree(response.getText());
478             JsonNode v = tree.get("uploadUrl");
479             if (v == null) {
480                 throw new ClientException(
481                         "could not create upload session because response does not contain 'uploadUrl' field:\n"
482                                 + response.getText());
483             } else {
484                 return v.asText();
485             }
486         } catch (JsonProcessingException e) {
487             throw new ClientException(e);
488         }
489     }

491     public static void putChunk(HttpService service, String url, InputStream in,
492             List<RequestHeader> requestHeaders, long startByte, long finishByte, long size) {
493         List<RequestHeader> h = new ArrayList<RequestHeader>(requestHeaders);
494         h.add(RequestHeader.create("Content-Length", "" + (finishByte - startByte)));
495         h.add(RequestHeader.create("Content-Range",
496                 "bytes " + startByte + "-" + finishByte + "/" + size));
497         HttpResponse response = service.put(url, requestHeaders, in);
498         checkResponseCode(url, response, 200, 202);
499     }
