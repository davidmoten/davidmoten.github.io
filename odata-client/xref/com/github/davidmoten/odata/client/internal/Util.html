4   import java.io.IOException;

10      public static String readString(InputStream in, Charset charset) throws IOException {
11          ByteArrayOutputStream result = new ByteArrayOutputStream();
12          byte[] buffer = new byte[1024];
13          int length;
14          while ((length = in.read(buffer)) != -1) {
15              result.write(buffer, 0, length);
16          }
17          return new String(result.toByteArray(), charset);
18      }
19  
20      public static String trimTrailingSlash(String url) {
21          if (!url.isEmpty() && url.charAt(url.length() - 1) == '/') {
22              return url.substring(0, url.length() - 2);
23          } else {
24              return url;
25          }
26      }
