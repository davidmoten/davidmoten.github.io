11          GraphService client = MsGraph.explorer().build();
12  
13          client //
14                  .users() //
15                  .select("displayName") //
16                  .maxPageSize(10).stream() //
17                  .limit(10) //
18                  .map(user -> user.getDisplayName().orElse("?")) //
19                  .forEach(System.out::println);
20  
21          System.exit(0);
22  
23          String id = client.me().messages().select("id").stream().limit(1).findFirst().get().getId()
24                  .get();
25  
26          client //
27                  .me() //
28                  .messages(id) //
29                  .attachments() //
30                  .maxPageSize(2) //
31                  .select("name,size") //
32                  .stream() //
33                  .filter(att -> att instanceof FileAttachment) //
34                  .limit(2) //
35                  .map(att -> att.getName().orElse("") + " " + att.getSize().orElse(0) + "B") //
36                  .forEach(System.out::println);
37  
38          System.exit(0);
39  
40          client //
41                  .groups() //
42                  .top(3) //
43                  .select("id,displayName,groupTypes") //
44                  .stream() //
45                  .limit(3) //
46                  .filter(g -> g.getGroupTypes().toList().contains("Unified"))
47                  .peek(g -> System.out.println(g.getDisplayName().orElse("?"))) //
48                  .flatMap(group -> group //
49                          .getMembers() //
50                          .select("id,displayName,userType") //
51                          .filter(User.class) //
52                          .stream()) //
53                  .filter(user -> "Member".equalsIgnoreCase(user.getUserType().orElse("")))
54                  .forEach(user -> System.out.println("  " + user.getDisplayName().orElse("?")));
55      }
