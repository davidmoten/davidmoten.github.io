8       static byte[] read(InputStream in) throws IOException {
9           ByteArrayOutputStream bytes = new ByteArrayOutputStream();
10          byte[] buffer = new byte[8192];
11          int n;
12          while ((n = in.read(buffer)) != -1) {
13              bytes.write(buffer, 0, n);
14          }
15          return bytes.toByteArray();
16      }
