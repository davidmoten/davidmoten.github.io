33      public static byte[] getBytes(InputStream is) throws IOException {
34          ByteArrayOutputStream os = new ByteArrayOutputStream();
35          byte[] buffer = new byte[8192];
36          for (int len = is.read(buffer); len != -1; len = is.read(buffer)) {
37              os.write(buffer, 0, len);
38          }
39          return os.toByteArray();
40      }
