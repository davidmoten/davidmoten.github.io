122         public void wrap(Writer out) {
123             try {
124                 wordWrap(reader, out,newLine, maxWidth, stringWidth, wordChars );
125             } catch (IOException e) {
126                 throw new IORuntimeException(e);
127             } finally {
128                 if (closeReader) {

169     private static Set<Character> toSet(String chars) {
170         Set<Character> set = new HashSet<Character>();
171         for (int i = 0; i < chars.length(); i++) {
172             set.add(chars.charAt(i));
173         }
174         return set;
175     }

177     static void wordWrap(Reader in, Writer out, String newLine, Number maxWidth,
178             Function<? super CharSequence, ? extends Number> stringWidth, Set<Character> specialWordChars)
179             throws IOException {
180         StringBuilder line = new StringBuilder();
181         StringBuilder word = new StringBuilder();
182         double maxWidthDouble = maxWidth.doubleValue();
183         boolean broken = false;
184         boolean alphanumeric = false;
185         boolean previousWasPunctuation = false;
186         while (true) {
187             int c = in.read();
188             if (c == -1) {
189                 break;
190             }
191             char ch = (char) c;
192             alphanumeric = Character.isAlphabetic(ch) || specialWordChars.contains(ch);
193             if (ch == '\n') {
194                 line.append(word);
195                 out.write(line.toString());
196                 out.write(newLine);
197                 word.setLength(0);
198                 line.setLength(0);
199                 broken = false;
200             } else if (ch == '\r') {
201                 // ignore carriage return
202             } else if (alphanumeric && !previousWasPunctuation) {
203                 word.append(ch);
204                 if (broken && line.length() == 0) {
205                     leftTrim(word);
206                 }
207                 if (tooLong(stringWidth, line.toString() + word.toString(), maxWidthDouble)) {
208                     if (line.length() > 0) {
209                         writeLine(out, line, newLine);
210                         leftTrim(word);
211                         if (tooLong(stringWidth, word.toString(), maxWidthDouble)) {
212                             writeBrokenWord(out, word, newLine);
213                         } else {
214                             broken = true;
215                         }
216                     } else {
217                         writeBrokenWord(out, word, newLine);
218                     }
219                 }
220             } else {
221                 if (word.length() > 0 && !isWhitespace(word)) {
222                     appendWordToLine(line, word);
223                     if (broken) {
224                         leftTrim(line);
225                     }
226                 }
227                 word.append(ch);
228                 if (tooLong(stringWidth, line.toString() + word.toString(), maxWidthDouble)) {
229                     if (line.length() > 0) {
230                         if (!isWhitespace(line)) {
231                             writeLine(out, line, newLine);
232                         } else {
233                             line.setLength(0);
234                         }
235                         broken = true;
236                     } else {
237                         String w = word.substring(0, word.length() - 1);
238                         word.delete(0, word.length() - 1);
239                         if (broken) {
240                             w = leftTrim(w);
241                         }
242                         if (w.length() > 0) {
243                             out.write(w);
244                             out.write(newLine);
245                             broken = false;
246                         }
247                     }
248                 }
249             }
250             previousWasPunctuation = isPunctuation(ch) && !specialWordChars.contains(ch);
251         }
252         if (line.length() > 0) {
253             String s = line.toString() + word.toString();
254             if (broken) {
255                 s = leftTrim(s);
256             }
257             out.write(s);
258         } else {
259             if (broken) {
260                 leftTrim(word);
261             }
262             out.write(word.toString());
263         }
264     }

266     private static boolean isPunctuation(char ch) {
267         return Pattern.matches("\\p{Punct}", ch + "");
268     }

270     private static boolean tooLong(Function<? super CharSequence, ? extends Number> stringWidth, String s,
271             double maxWidthDouble) {
272         return stringWidth.apply(rightTrim(s)).doubleValue() > maxWidthDouble;
273     }

275     @VisibleForTesting
276     static String rightTrim(String s) {
277         int i = s.length();
278         while (i > 0) {
279             if (Character.isWhitespace(s.charAt(i - 1))) {
280                 i--;
281             } else {
282                 break;
283             }
284         }
285         if (i != s.length()) {
286             return s.substring(0, i);
287         } else {
288             return s;
289         }
290     }

292     static boolean isWhitespace(CharSequence s) {
293         for (int i = 0; i < s.length(); i++) {
294             if (!Character.isWhitespace(s.charAt(i))) {
295                 return false;
296             }
297         }
298         return true;
299     }

301     @VisibleForTesting
302     static void leftTrim(StringBuilder word) {
303         // trim leading spaces on the word
304         // because we have inserted a new line
305         int i;
306         for (i = 0; i < word.length(); i++) {
307             if (!Character.isWhitespace(word.charAt(i))) {
308                 break;
309             }
310         }
311         if (i < word.length() && i > 0) {
312             word.delete(0, i);
313         }
314     }
315 
316     private static String leftTrim(String s) {
317         StringBuilder b = new StringBuilder(s);
318         leftTrim(b);
319         return b.toString();
320     }

322     private static void appendWordToLine(StringBuilder line, StringBuilder word) {
323         line.append(word.toString());
324         word.setLength(0);

327     private static void writeBrokenWord(Writer out, StringBuilder word, String newLine) throws IOException {
328         out.write(word.substring(0, word.length() - 1));
329         out.write(newLine);
330         word.delete(0, word.length() - 1);
331     }

333     private static void writeLine(Writer out, StringBuilder line, String newLine) throws IOException {
334         out.write(line.toString());
