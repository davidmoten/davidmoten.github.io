12  public final class ActorRefUnsynchronizedDrain<T> extends ActorRefImpl<T> {
13  
14      boolean running;
15  
16      protected ActorRefUnsynchronizedDrain(String name, Supplier<? extends Actor<T>> factory, Scheduler scheduler,
17              Context context, Supervisor supervisor, ActorRef<?> parent, MailboxFactory mailboxFactory) {
18          super(name, factory, scheduler, context, supervisor, parent, mailboxFactory);
19      }
20      
21      @Override
22      public void run() {
23          // protect against recursion
24          if (!running) {
25              running = true;
26              drain();
27              running = false;
28          }
29      }
