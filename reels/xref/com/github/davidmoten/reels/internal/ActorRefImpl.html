49      protected final AtomicInteger state = new AtomicInteger(); // ACTIVE

52      private static final int ACTIVE = 0;
53      private static final int STOPPING = 1;
54      private static final int STOPPED = 2;
55      private static final int STOPPING_NOW = 3;
56      protected static final int RESTART = 4;
57      private static final int PAUSED = 5;

59      public static <T> ActorRefImpl<T> create(String name, Supplier<? extends Actor<T>> factory, Scheduler scheduler,
60              Context context, Supervisor supervisor, ActorRef<?> parent, MailboxFactory mailboxFactory) {
61          final ActorRefImpl<T> a;
62          if (scheduler.requiresDrainSynchronization()) {
63              a = new ActorRefSynchronizedDrain<T>(name, factory, scheduler, context, supervisor, parent, mailboxFactory);
64          } else {
65              a = new ActorRefUnsynchronizedDrain<T>(name, factory, scheduler, context, supervisor, parent, mailboxFactory);
66          }
67          if (parent != null) {
68              ((ActorRefImpl<?>) parent).addChild(a);
69          }
70          return a;
71      }
72  
73      protected ActorRefImpl(String name, Supplier<? extends Actor<T>> factory, Scheduler scheduler, Context context,
74              Supervisor supervisor, ActorRef<?> parent, MailboxFactory mailboxFactory) {
75          this.name = name;
76          this.factory = factory;
77          this.context = context;
78          this.supervisor = supervisor;
79          this.mailbox = mailboxFactory.create();
80          this.worker = scheduler.createWorker();
81          this.parent = parent;
82          this.scheduler = scheduler;
83          this.children = new ConcurrentHashMap<>();
84          createActor();
85      }

87      private void addChild(ActorRef<?> actor) {
88          int s = state.get();
89          if (s == STOPPING_NOW || s == STOPPING || s == STOPPED) {
90              actor.stopNow();
91          } else {
92              children.put(actor.name(), actor);
93          }
94      }

96      @Override
97      public void stop() {
98          tell(PoisonPill.instance(), parent);
99      }

101     @Override
102     public void stopNow() {
103         if (debug)
104             log("stopNow");
105         // use a stack rather than recursion to avoid
106         // stack overflow on deeply nested hierarchies
107         Deque<ActorRef<?>> stack = new ArrayDeque<>();
108         stack.offer(this);
109         ActorRef<?> a;
110         while ((a = stack.poll()) != null) {
111             ((ActorRefImpl<?>) a).stopNowThis();
112             stack.addAll(children.values());
113         }
114     }
115 
116     public void stopNowThis() {
117 
118         while (true) {
119             final int s = state.get();
120             if (s == STOPPING_NOW || s == STOPPING || s == STOPPED) {
121                 break;
122             } else if (state.compareAndSet(s, STOPPING_NOW)) {
123                 if (debug) {
124                     log("removing from parent and calling stop");
125                 }
126                 stop();
127                 break;
128             }
129         }
130     }
131 
132     private boolean setState(int value) {
133         while (true) {
134             int s = state.get();
135             if (s == STOPPED || s == STOPPING && value == STOPPING_NOW) {
136                 return false;
137             }
138             if (state.compareAndSet(s, value)) {
139                 return true;
140             }
141         }
142     }

160     private void scheduleDrain() {
161         worker.schedule(this);
162     }

164     private void handleTerminationMessage(Message<T> message) {
165         children.remove(message.sender().name());
166         if (children.isEmpty()) {
167             runOnStop(message);
168         }
169     }
170 
171     private void sendToDeadLetter(Message<T> message) {
172         if (context.deadLetterActor() != this && !systemMessagesOnly) {
173             context.deadLetterActor().tell(new DeadLetter(message), this);
174         }
175     }
176 
177     @SuppressWarnings("unchecked")
178     private void runOnStop(Message<T> message) {
179         if (setState(STOPPED)) {
180             if (debug) {
181                 log(this + ": running onStop with message=" + message);
182             }
183             try {
184                 actor.onStop(this);
185             } catch (Throwable e) {
186                 supervisor.processFailure(message, this, new OnStopException(e));
187             }
188             complete();
189             ActorRef<?> p = parent;
190             if (p == null) {
191                 // is root actor (which is the only actor without a parent)
192                 mailbox.offer(new Message<T>((T) Terminated.INSTANCE, this, this));
193             } else {
194                 p.<Object>recast().tell(Terminated.INSTANCE, this);
195             }
196         }
197     }

208     @Override
209     public boolean restart() {
210         while (true) {
211             int s = state.get();
212             if (s == ACTIVE || s == PAUSED) {
213                 if (state.compareAndSet(s, RESTART)) {
214                     run();
215                     return true;
216                 }
217             } else {
218                 return false;
219             }
220         }
221     }
222 
223     @Override
224     public boolean pause(long duration, TimeUnit unit) {
225         if (state.compareAndSet(ACTIVE, PAUSED)) {
226             scheduler.schedule(() -> state.compareAndSet(PAUSED, ACTIVE), duration, unit);
227             return true;
228         } else {
229             return false;
230         }
231     }
232 
233     @Override
234     public boolean pauseAndRestart(long delay, TimeUnit unit) {
235         if (pause(delay, unit)) {
236             scheduler.schedule(this::restart, delay, unit);
237             return true;
238         } else {
239             return false;
240         }
241     }

300     @SuppressWarnings("unchecked")
301     protected void drain() {
302         Message<T> message;
303         int s;
304         while ((s = state.get()) != PAUSED && (message = (Message<T>) mailbox.poll()) != null) {
305             if (debug) {
306                 log("message polled=" + message.content() + " from " + message.sender() + ", state=" + s);
307             }
308             if (s == RESTART) {
309                 actor.onStop(this);
310                 createActor();
311                 if (setState(ACTIVE)) {
312                     s = ACTIVE;
313                 } else {
314                     s = state.get();
315                 }
316             }
317             if (s == STOPPING_NOW) {
318                 systemMessagesOnly = true;
319             }
320             if (s == STOPPING) {
321                 if (message.content() == Terminated.INSTANCE) {
322                     handleTerminationMessage(message);
323                 } else {
324                     sendToDeadLetter(message);
325                 }
326             } else if (s == STOPPED) {
327                 if (message.content() != PoisonPill.instance()) {
328                     sendToDeadLetter(message);
329                 }
330             } else if (message.content() == PoisonPill.instance()) {
331                 if (setState(STOPPING)) {
332                     boolean isEmpty = true;
333                     ActorRef<Object> ch;
334                     for (ActorRef<?> child : children.values()) {
335                         ch = (ActorRef<Object>) child;
336                         if (systemMessagesOnly) {
337                             ch.stopNow();
338                         } else {
339                             ch.stop();
340                         }
341                         isEmpty = false;
342                     }
343                     if (isEmpty) {
344                         // no children, run onStop and send Terminated to parent
345                         runOnStop(message);
346                     }
347                 }
348             } else if (message.content() == Terminated.INSTANCE) {
349                 handleTerminationMessage(message);
350             } else if (systemMessagesOnly) {
351                 sendToDeadLetter(message);
352             } else {
353                 if (!preStartHasBeenRun) {
354                     runPreStart(message);
355                 }
356                 try {
357                     actor.onMessage(message);
358                 } catch (Throwable e) {
359                     // if the line below throws then the actor will no longer process messages
360                     // (because wip will be != 0)
361                     supervisor.processFailure(message, this, e);
362                 }
363             }
364         }
365     }
366 
367     private void runPreStart(Message<T> message) {
368         try {
369             actor.preStart(this);
370             preStartHasBeenRun = true;
371         } catch (Throwable e) {
372             // if the line below throws then the actor will no longer process messages
373             // (because wip will be != 0)
374             supervisor.processFailure(message, this, new PreStartException(e));
375         }
376     }
