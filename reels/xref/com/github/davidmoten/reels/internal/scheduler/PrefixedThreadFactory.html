3   import java.util.concurrent.ThreadFactory;
4   import java.util.concurrent.atomic.AtomicInteger;
5   
6   public final class PrefixedThreadFactory implements ThreadFactory {
7   
8       private final String prefix;
9       private final AtomicInteger count = new AtomicInteger();
10  
11      public PrefixedThreadFactory(String prefix) {
12          this.prefix = prefix;
13      }
14  
15      @Override
16      public Thread newThread(Runnable r) {
17          String name = prefix + "-" + count.incrementAndGet();
18          Thread t = new Thread(r, name);
19          t.setPriority(Thread.NORM_PRIORITY);
20          t.setDaemon(true);
21          return t;
22      }
23  }
