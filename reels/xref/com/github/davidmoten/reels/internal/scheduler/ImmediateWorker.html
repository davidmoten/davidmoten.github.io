1   package com.github.davidmoten.reels.internal.scheduler;

3   import java.util.concurrent.TimeUnit;
4   
5   import com.github.davidmoten.reels.Disposable;
6   import com.github.davidmoten.reels.Worker;
7   
8   public final class ImmediateWorker extends AbstractCanScheduleDisposable implements Worker {
9   
10      private volatile boolean disposed;
11  
12      @Override
13      public void dispose() {
14          disposed = true;
15      }
16  
17      @Override
18      public boolean isDisposed() {
19          return disposed;
20      }
21      
22      @Override
23      public Disposable _schedule(Runnable run) {
24          run.run();
25          return Disposable.disposed();
26      }
27  
28      @Override
29      public Disposable _schedule(Runnable run, long delay, TimeUnit unit) {
30          try {
31              unit.sleep(delay);
32          } catch (InterruptedException e) {
33              return Disposable.disposed();
34          }
35          if (!disposed) {
36              run.run();
37          }
38          return Disposable.disposed();
39      }
40  
41      @Override
42      public Disposable _schedulePeriodically(Runnable run, long initialDelay, long period, TimeUnit unit) {
43          throw new UnsupportedOperationException("cannot schedule periodically with the immediate scheduler");
44      }
45  }
