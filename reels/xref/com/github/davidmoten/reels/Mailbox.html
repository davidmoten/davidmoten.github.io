3   public interface Mailbox<T> {
4   
5       /**
6        * If mailbox empty returns null otherwise returns first message available on
7        * the the queue and removes it. Should only be called from one thread (that is
8        * serially).
9        * 
10       * @return first message available or null if mailbox empty
11       */
12      Message<T> poll();
13  
14      /**
15       * Adds the message to the mailbox. In general this method is thread-safe (can
16       * be called concurrently).
17       * 
18       * @param message message to add to the mailbox
19       * @return true if added to the mailbox
20       */
21      boolean offer(Message<T> message);
22  
23      /**
24       * Places the last message polled back on the queue so that the next poll will
25       * return that message again. Should be called from the same thread as poll or
26       * more formally there must always be a formal happens-before relationship
27       * between poll and this method.
28       */
29      void retryLatest();
30  }
