3   import java.util.List;
4   import java.util.Map;
5   import java.util.concurrent.TimeUnit;
6   import java.util.stream.Collectors;
7   
8   import com.github.davidmoten.aws.lw.client.xml.XmlElement;
9   
10  public final class ClientMain {
11  
12      public static void main(String[] args) throws InterruptedException {
13          String regionName = "ap-southeast-2";
14          String accessKey = System.getProperty("accessKey");
15          String secretKey = System.getProperty("secretKey");
16  
17          Credentials credentials = Credentials.of(accessKey, secretKey);
18          Client sqs = Client //
19                  .sqs() //
20                  .regionName(regionName) //
21                  .credentials(credentials) //
22                  .build();
23          Client s3 = Client.s3().from(sqs).build();
24          {
25              // create bucket
26              String bucketName = "temp-bucket-" + System.currentTimeMillis();
27              String createXml = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
28                      + "<CreateBucketConfiguration xmlns=\"http://s3.amazonaws.com/doc/2006-03-01/\">\n"
29                      + "   <LocationConstraint>" + regionName + "</LocationConstraint>\n"
30                      + "</CreateBucketConfiguration>";
31              s3.path(bucketName) //
32                      .method(HttpMethod.PUT) //
33                      .requestBody(createXml) //
34                      .execute();
35  
36              String objectName = "ExampleObject.txt";
37              Map<String, List<String>> h = s3 //
38                      .path(bucketName, objectName) //
39                      .method(HttpMethod.PUT) //
40                      .requestBody("hi there") //
41                      .metadata("category", "something") //
42                      .response() //
43                      .headers();
44              System.out.println("put object completed, headers:");
45              h.entrySet().stream().forEach(x -> System.out.println("  " + x));
46  
47              try {
48                  s3.path(bucketName + "/" + "not-there") //
49                          .responseAsUtf8();
50              } catch (Throwable e) {
51                  e.printStackTrace(System.out);
52              }
53  
54              try {
55                  Client.s3() //
56                          .regionName(regionName) //
57                          .credentials(credentials) //
58                          .exception(r -> !r.isOk(), r -> new NoSuchKeyException(r.contentUtf8()))
59                          .build() //
60                          .path(bucketName + "/" + "not-there") //
61                          .responseAsUtf8();
62              } catch (NoSuchKeyException e) {
63                  e.printStackTrace(System.out);
64              }
65  
66              {
67                  Response r = s3.path(bucketName , "notThere").response();
68                  System.out.println("ok=" + r.isOk() + ", statusCode=" + r.statusCode()
69                          + ", message=" + r.contentUtf8());
70              }
71              {
72                  // read bucket object
73                  String text = s3.path(bucketName, objectName).responseAsUtf8();
74                  System.out.println(text);
75                  System.out.println("presignedUrl="
76                          + s3.path("amsa-xml-in" + "/" + objectName).presignedUrl(1, TimeUnit.DAYS));
77              }
78              {
79                  // read bucket object with metadata
80                  Response r = s3 //
81                          .path(bucketName, objectName) //
82                          .response();
83                  System.out.println(r.content().length + " chars read");
84                  r //
85                          .metadata() //
86                          .entrySet() //
87                          .stream() //
88                          .map(x -> x.getKey() + "=" + x.getValue()) //
89                          .forEach(System.out::println);
90  
91                  System.out.println("category[0]=" + r.metadata("category").orElse(""));
92              }
93  
94              List<String> keys = s3 //
95                      .url("https://" + bucketName + ".s3." + regionName + ".amazonaws.com") //
96                      .query("list-type", "2") //
97                      .responseAsXml() //
98                      .childrenWithName("Contents") //
99                      .stream() //
100                     .map(x -> x.content("Key")) //
101                     .collect(Collectors.toList());
102 
103             System.out.println(keys);
104 
105             // delete object
106             s3.path(bucketName, objectName) //
107                     .method(HttpMethod.DELETE) //
108                     .execute();
109 
110             // delete bucket
111             s3.path(bucketName) //
112                     .method(HttpMethod.DELETE) //
113                     .execute();
114             System.out.println("bucket deleted");
115 
116             System.out.println("all actions complete on s3");
117         }
118 
119         {
120             String queueName = "MyQueue-" + System.currentTimeMillis();
121 
122             sqs.query("Action", "CreateQueue") //
123                     .query("QueueName", queueName) //
124                     .execute();
125 
126             String queueUrl = sqs //
127                     .query("Action", "GetQueueUrl") //
128                     .query("QueueName", queueName) //
129                     .responseAsXml() //
130                     .content("GetQueueUrlResult", "QueueUrl");
131 
132             for (int i = 1; i <= 3; i++) {
133                 sqs.url(queueUrl) //
134                         .query("Action", "SendMessage") //
135                         .query("MessageBody", "hi there --> " + i) //
136                         .execute();
137             }
138 
139             // read all messages, print to console and delete them
140             List<XmlElement> list;
141             Request request = sqs //
142                     .url(queueUrl) //
143                     .query("Action", "ReceiveMessage");
144             do {
145                 list = request //
146                         .responseAsXml() //
147                         .child("ReceiveMessageResult") //
148                         .children();
149 
150                 list.forEach(x -> {
151                     String msg = x.child("Body").content();
152                     System.out.println(msg);
153                     // mark message as read
154                     sqs.url(queueUrl) //
155                             .query("Action", "DeleteMessage") //
156                             .query("ReceiptHandle", x.child("ReceiptHandle").content()) //
157                             .execute();
158                 });
159             } while (!list.isEmpty());
160 
161             sqs.url(queueUrl) //
162                     .query("Action", "DeleteQueue") //
163                     .execute();
164 
165             System.out.println("all actions complete on " + queueUrl);
166         }
167     }
168 }
