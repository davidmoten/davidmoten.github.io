17  public final class HttpClientDefault implements HttpClient {
18  
19      public static final HttpClientDefault INSTANCE = new HttpClientDefault();
20  
21      private HttpClientDefault() {
22      }

24      @Override
25      public Response request(URL endpointUrl, String httpMethod, Map<String, String> headers,
26              byte[] requestBody, int connectTimeoutMs, int readTimeoutMs) {
27          HttpURLConnection connection = Util.createHttpConnection(endpointUrl, httpMethod, headers,
28                  connectTimeoutMs, readTimeoutMs);
29          try {
30              if (requestBody != null) {
31                  OutputStream out = connection.getOutputStream();
32                  out.write(requestBody);
33                  out.flush();
34              }
35              Map<String, List<String>> responseHeaders = connection.getHeaderFields();
36              int responseCode = connection.getResponseCode();
37              boolean ok = isOk(responseCode);
38              InputStream is;
39              if (ok) {
40                  is = connection.getInputStream();
41              } else {
42                  is = connection.getErrorStream();
43              }
44              final byte[] bytes;
45              if (is == null) {
46                  bytes = new byte[0];
47              } else {
48                  bytes = readBytes(is);
49              }
50              return new Response(responseHeaders, bytes, responseCode);
51          } catch (IOException e) {
52              throw new UncheckedIOException(e);
53          } finally {
54              if (connection != null) {
55                  connection.disconnect();
56              }
57          }
58      }
59  
60      private static boolean isOk(int responseCode) {
61          return responseCode >= 200 && responseCode <= 299;
62      }
63  
64      private static byte[] readBytes(InputStream in) throws IOException {
65          byte[] buffer = new byte[8192];
66          int n;
67          ByteArrayOutputStream bytes = new ByteArrayOutputStream();
68          while ((n = in.read(buffer)) != -1) {
69              bytes.write(buffer, 0, n);
70          }
71          return bytes.toByteArray();
72      }
