11  public final class Response {
12  
13      private final Map<String, List<String>> headers;
14      private final byte[] content;
15      private final int statusCode;
16  
17      public Response(Map<String, List<String>> headers, byte[] content, int statusCode) {
18          this.headers = headers;
19          this.content = content;
20          this.statusCode = statusCode;
21      }
22  
23      public Map<String, List<String>> headers() {
24          return headers;
25      }
26  
27      /**
28       * Returns those headers that start with {@code x-amz-meta-} (and removes that
29       * prefix).
30       * 
31       * @return headers that start with {@code x-amz-meta-} (and removes that prefix)
32       */
33      public Metadata metadata() {
34          return new Metadata(headers //
35                  .entrySet() //
36                  .stream() //
37                  .filter(x -> x.getKey() != null) //
38                  .filter(x -> x.getKey().startsWith("x-amz-meta-")) //
39                  .collect(Collectors.toMap( //
40                          x -> x.getKey().substring(11), //
41                          x -> x.getValue().get(0))));
42      }
43  
44      public Optional<String> metadata(String name) {
45          Preconditions.checkNotNull(name);
46          return metadata().value(name);
47      }
48  
49      public byte[] content() {
50          return content;
51      }
52  
53      public String contentUtf8() {
54          return new String(content, StandardCharsets.UTF_8);
55      }
56  
57      public int statusCode() {
58          return statusCode;
59      }
60  
61      public boolean isOk() {
62          return statusCode >= 200 && statusCode <= 299;
63      }
