26      static void put(Map<String, List<String>> map, String name, String value) {
27          Preconditions.checkNotNull(map);
28          Preconditions.checkNotNull(name);
29          Preconditions.checkNotNull(value);
30          List<String> list = map.get(name);
31          if (list == null) {
32              list = new ArrayList<>();
33              map.put(name, list);
34          }
35          list.add(value);
36      }

38      static Map<String, String> combineHeaders(Map<String, List<String>> headers) {
39          Preconditions.checkNotNull(headers);
40          return headers.entrySet().stream().collect(Collectors.toMap(x -> x.getKey(),
41                  x -> x.getValue().stream().collect(Collectors.joining(","))));
42      }

44      static String presignedUrl(Clock clock, String url, String method, Map<String, String> headers,
45              byte[] requestBody, String serviceName, String regionName, Credentials credentials,
46              int connectTimeoutMs, int readTimeoutMs, long expirySeconds) {
47  
48          // the region-specific endpoint to the target object expressed in path style
49          URL endpointUrl = Util.toUrl(url);
50  
51          Map<String, String> h = new HashMap<>(headers);
52          final String contentHashString;
53          if (isEmpty(requestBody)) {
54              contentHashString = Aws4SignerBase.UNSIGNED_PAYLOAD;
55              h.put("x-amz-content-sha256", "");
56          } else {
57              // compute hash of the body content
58              byte[] contentHash = Util.sha256(requestBody);
59              contentHashString = Util.toHex(contentHash);
60              h.put("content-length", "" + requestBody.length);
61              h.put("x-amz-content-sha256", contentHashString);
62          }
63  
64  //        if (credentials.sessionToken().isPresent()) {
65  //            h.put("x-amz-security-token", credentials.sessionToken().get());
66  //        }
67  
68          List<Parameter> parameters = extractQueryParameters(endpointUrl);
69          Map<String, String> q = parameters.stream()
70                  .collect(Collectors.toMap(p -> p.name, p -> p.value));
71          // construct the query parameter string to accompany the url
72  
73          // for SignatureV4, the max expiry for a presigned url is 7 days,
74          // expressed in seconds
75          q.put("X-Amz-Expires", "" + expirySeconds);
76  
77          Aws4SignerForQueryParameterAuth signer = new Aws4SignerForQueryParameterAuth(endpointUrl,
78                  method, serviceName, regionName);
79          String authorizationQueryParameters = signer.computeSignature(clock, h, q,
80                  contentHashString, credentials.accessKey(), credentials.secretKey());
81  
82          // build the presigned url to incorporate the authorization elements as query
83          // parameters
84          String u = endpointUrl.toString();
85          final String presignedUrl;
86          if (u.contains("?")) {
87              presignedUrl = u + "&" + authorizationQueryParameters;
88          } else {
89              presignedUrl = u + "?" + authorizationQueryParameters;
90          }
91          return presignedUrl;
92      }

94      static Response request(Clock clock, HttpClient httpClient, String url, String method,
95              Map<String, String> headers, byte[] requestBody, String serviceName, String regionName,
96              Credentials credentials, int connectTimeoutMs, int readTimeoutMs) {
97  
98          // the region-specific endpoint to the target object expressed in path style
99          URL endpointUrl = Util.toUrl(url);
100 
101         Map<String, String> h = new HashMap<>(headers);
102         final String contentHashString;
103         if (isEmpty(requestBody)) {
104             contentHashString = Aws4SignerBase.EMPTY_BODY_SHA256;
105         } else {
106             // compute hash of the body content
107             byte[] contentHash = Util.sha256(requestBody);
108             contentHashString = Util.toHex(contentHash);
109             h.put("content-length", "" + requestBody.length);
110         }
111         h.put("x-amz-content-sha256", contentHashString);
112         if (credentials.sessionToken().isPresent()) {
113             h.put("x-amz-security-token", credentials.sessionToken().get());
114         }
115 
116         List<Parameter> parameters = extractQueryParameters(endpointUrl);
117         Map<String, String> q = parameters.stream()
118                 .collect(Collectors.toMap(p -> p.name, p -> p.value));
119         Aws4SignerForAuthorizationHeader signer = new Aws4SignerForAuthorizationHeader(endpointUrl,
120                 method, serviceName, regionName);
121         String authorization = signer.computeSignature(clock, h, q, contentHashString,
122                 credentials.accessKey(), credentials.secretKey());
123 
124         // place the computed signature into a formatted 'Authorization' header
125         // and call S3
126         h.put("Authorization", authorization);
127         return httpClient.request(endpointUrl, method, h, requestBody, connectTimeoutMs,
128                 readTimeoutMs);
129     }

131     private static List<Parameter> extractQueryParameters(URL endpointUrl) {
132         String query = endpointUrl.getQuery();
133         if (query == null) {
134             return Collections.emptyList();
135         } else {
136             return extractQueryParameters(query);
137         }
138     }

140     private static final char QUERY_PARAMETER_SEPARATOR = '&';
141     private static final char QUERY_PARAMETER_VALUE_SEPARATOR = '=';
142 
143     /**
144      * Extract parameters from a query string, preserving encoding.
145      * <p>
146      * We can't use Apache HTTP Client's URLEncodedUtils.parse, mainly because we
147      * don't want to decode names/values.
148      *
149      * @param rawQuery the query to parse
150      * @return The list of parameters, in the order they were found.
151      */
152     // VisibleForTesting
153     static List<Parameter> extractQueryParameters(String rawQuery) {
154         List<Parameter> results = new ArrayList<>();
155         int endIndex = rawQuery.length() - 1;
156         int index = 0;
157         while (0 <= index && index <= endIndex) {
158             /*
159              * Ideally we should first look for '&', then look for '=' before the '&', but
160              * obviously that's not how AWS understand query parsing; see the test
161              * "post-vanilla-query-nonunreserved" in the test suite. A string such as
162              * "?foo&bar=qux" will be understood as one parameter with name "foo&bar" and
163              * value "qux". Don't ask me why.
164              */
165             String name;
166             String value;
167             int nameValueSeparatorIndex = rawQuery.indexOf(QUERY_PARAMETER_VALUE_SEPARATOR, index);
168             if (nameValueSeparatorIndex < 0) {
169                 // No value
170                 name = rawQuery.substring(index);
171                 value = null;
172 
173                 index = endIndex + 1;
174             } else {
175                 int parameterSeparatorIndex = rawQuery.indexOf(QUERY_PARAMETER_SEPARATOR,
176                         nameValueSeparatorIndex);
177                 if (parameterSeparatorIndex < 0) {
178                     parameterSeparatorIndex = endIndex + 1;
179                 }
180                 name = rawQuery.substring(index, nameValueSeparatorIndex);
181                 value = rawQuery.substring(nameValueSeparatorIndex + 1, parameterSeparatorIndex);
182 
183                 index = parameterSeparatorIndex + 1;
184             }
185             if (value != null) {
186                 try {
187                     results.add(new Parameter(URLDecoder.decode(name, "UTF-8"),
188                             URLDecoder.decode(value, "UTF-8")));
189                 } catch (UnsupportedEncodingException e) {
190                     throw new RuntimeException(e);
191                 }
192             }
193         }
194         return results;
195     }
