70      @SuppressWarnings("unused")
71      @Benchmark
72      @BenchmarkMode(Mode.SampleTime)
73      @OutputTimeUnit(TimeUnit.MILLISECONDS)
74      @Warmup(iterations = 5, time = ShakespearePlaysScrabble.DURATION_SECONDS)
75      @Measurement(iterations = 5, time = ShakespearePlaysScrabble.DURATION_SECONDS)
76      @Fork(1)
77      public List<Entry<Integer, List<String>>> measureThroughput() throws InterruptedException {
78  
79          // Function to compute the score of a given word
80          Function<Integer, Stream<Integer>> scoreOfALetter = letter -> Stream.of(letterScores[letter - 'a']);
81  
82          // score of the same letters in a word
83          Function<Entry<Integer, LongWrapper>, Stream<Integer>> letterScore = entry -> Stream
84                  .of(letterScores[entry.getKey() - 'a']
85                          * Integer.min((int) entry.getValue().get(), scrabbleAvailableLetters[entry.getKey() - 'a']));
86  
87          Function<String, Stream<Integer>> toIntegerStream = string -> Stream
88                  .fromIterable(IterableSpliterator.of(string.chars().boxed().spliterator()));
89  
90          // Histogram of the letters in a given word
91          Function<String, Single<HashMap<Integer, LongWrapper>>> histoOfLetters = word -> toIntegerStream.apply(word) //
92                  .collect(() -> new HashMap<>(), (HashMap<Integer, LongWrapper> map, Integer value) -> {
93                      LongWrapper newValue = map.get(value);
94                      if (newValue == null) {
95                          newValue = () -> 0L;
96                      }
97                      map.put(value, newValue.incAndSet());
98                  });
99  
100         // number of blanks for a given letter
101         Function<Entry<Integer, LongWrapper>, Stream<Long>> blank = entry -> Stream
102                 .of(Long.max(0L, entry.getValue().get() - scrabbleAvailableLetters[entry.getKey() - 'a']));
103 
104         // number of blanks for a given word
105         Function<String, Stream<Long>> nBlanks = word -> histoOfLetters.apply(word)
106                 .flatMap(map -> Stream.fromIterable(map.entrySet())).flatMap(blank).reduce(Long::sum).toStream();
107 
108         // can a word be written with 2 blanks?
109         Function<String, Stream<Boolean>> checkBlanks = word -> nBlanks.apply(word).flatMap(l -> Stream.of(l <= 2L));
110 
111         // score taking blanks into account letterScore1
112         Function<String, Stream<Integer>> score2 = word -> histoOfLetters.apply(word)
113                 .flatMap(map -> Stream.fromIterable(map.entrySet())).flatMap(letterScore).reduce(Integer::sum)
114                 .toStream();
115 
116         // Placing the word on the board
117         // Building the streams of first and last letters
118         Function<String, Stream<Integer>> first3 = word -> Stream
119                 .fromIterable(IterableSpliterator.of(word.chars().boxed().limit(3).spliterator()));
120         Function<String, Stream<Integer>> last3 = word -> Stream
121                 .fromIterable(IterableSpliterator.of(word.chars().boxed().skip(3).spliterator()));
122 
123         // Stream to be maxed
124         Function<String, Stream<Integer>> toBeMaxed = word -> Stream.of(first3.apply(word), last3.apply(word))
125                 .flatMap(stream -> stream);
126 
127         // Bonus for double letter
128         Function<String, Stream<Integer>> bonusForDoubleLetter = word -> toBeMaxed.apply(word).flatMap(scoreOfALetter)
129                 .reduce(Integer::max).toStream();
130 
131         // score of the word put on the board
132         Function<String, Stream<Integer>> score3 = word -> Stream
133                 .of(score2.apply(word), score2.apply(word), bonusForDoubleLetter.apply(word),
134                         bonusForDoubleLetter.apply(word), Stream.of(word.length() == 7 ? 50 : 0))
135                 .flatMap(stream -> stream).reduce(Integer::sum).toStream();
136 
137         Function<Function<String, Stream<Integer>>, Single<TreeMap<Integer, List<String>>>> buildHistoOnScore = score -> Stream
138                 .fromIterable(shakespeareWords) //
139                 .filter(scrabbleWords::contains) //
140                 .filter(word -> checkBlanks.apply(word).first().get())
141                 .collect(() -> new TreeMap<Integer, List<String>>(Comparator.reverseOrder()),
142                         (TreeMap<Integer, List<String>> map, String word) -> {
143                             Integer key = score.apply(word).first().get();
144                             List<String> list = map.get(key);
145                             if (list == null) {
146                                 list = new ArrayList<>();
147                                 map.put(key, list);
148                             }
149                             list.add(word);
150                         });
151 
152         // best key / value pairs
153         List<Entry<Integer, List<String>>> finalList2 = buildHistoOnScore //
154                 .apply(score3) //
155                 .flatMap(map -> Stream.fromIterable(map.entrySet())) //
156                 .take(3) //
157                 .toList();
158 
159         return finalList2;
160     }

162     public static void main(String[] args) throws Exception {

164         s.init();
165         System.out.println(s.measureThroughput());
166     }
