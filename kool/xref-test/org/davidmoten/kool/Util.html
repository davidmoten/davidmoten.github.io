19  package org.davidmoten.kool;
20  
21  import java.io.File;
22  import java.io.IOException;
23  import java.nio.file.Files;
24  import java.util.HashSet;
25  import java.util.Set;
26  import java.util.stream.Collectors;
27  
28  public final class Util {
29  
30      private Util() {
31      }
32  
33      public static Set<String> readScrabbleWords() {
34          Set<String> scrabbleWords = new HashSet<>();
35          try (java.util.stream.Stream<String> scrabbleWordsStream = Files
36                  .lines(new File("src/test/resources/ospd.txt").toPath())) {
37              scrabbleWords.addAll(
38                      scrabbleWordsStream.map(String::toLowerCase).collect(Collectors.toSet()));
39          } catch (IOException e) {
40              throw new RuntimeException(e);
41          }
42          return scrabbleWords;
43      }
44  
45      public static Set<String> readShakespeareWords() {
46          Set<String> shakespeareWords = new HashSet<>();
47          try (java.util.stream.Stream<String> shakespeareWordsStream = Files
48                  .lines(new File("src/test/resources/words.shakespeare.txt").toPath())) {
49              shakespeareWords.addAll(
50                      shakespeareWordsStream.map(String::toLowerCase).collect(Collectors.toSet()));
51          } catch (IOException e) {
52              throw new RuntimeException(e);
53          }
54  
55          return shakespeareWords;
56      }
57  
58  }
