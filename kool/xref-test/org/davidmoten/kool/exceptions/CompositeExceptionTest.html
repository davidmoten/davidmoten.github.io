163     /**
164      * This hijacks the Throwable.printStackTrace() output and puts it in a string, where we can look for
165      * "CIRCULAR REFERENCE" (a String added by Throwable.printEnclosedStackTrace)
166      */
167     private static void assertNoCircularReferences(Throwable ex) {
168         ByteArrayOutputStream baos = new ByteArrayOutputStream();
169         PrintStream printStream = new PrintStream(baos);
170         ex.printStackTrace(printStream);
171         assertFalse(baos.toString().contains("CIRCULAR REFERENCE"));
172     }
173 
174     private static Throwable getRootCause(Throwable ex) {
175         Throwable root = ex.getCause();
176         if (root == null) {
177             return null;
178         } else {
179             while (true) {
180                 if (root.getCause() == null) {
181                     return root;
182                 } else {
183                     root = root.getCause();
184                 }
185             }
186         }
187     }

263     @Test
264     public void complexCauses() {
265         Throwable e1 = new Throwable("1");
266         Throwable e2 = new Throwable("2");
267         e1.initCause(e2);
268 
269         Throwable e3 = new Throwable("3");
270         Throwable e4 = new Throwable("4");
271         e3.initCause(e4);
272 
273         Throwable e5 = new Throwable("5");
274         Throwable e6 = new Throwable("6");
275         e5.initCause(e6);
276 
277         CompositeException compositeException = new CompositeException(e1, e3, e5);
278         assertTrue(compositeException.getCause() instanceof CompositeExceptionCausalChain);
279         assertEquals(3, compositeException.size());
280         
281         List<Throwable> causeChain = new ArrayList<Throwable>();
282         Throwable cause = compositeException.getCause().getCause();
283         while (cause != null) {
284             causeChain.add(cause);
285             cause = cause.getCause();
286         }
287         // The original relations
288         //
289         // e1 -> e2
290         // e3 -> e4
291         // e5 -> e6
292         //
293         // will be set to
294         //
295         // e1 -> e2 -> e3 -> e4 -> e5 -> e6
296         assertEquals(Arrays.asList(e1, e2, e3, e4, e5, e6), causeChain);
297     }
