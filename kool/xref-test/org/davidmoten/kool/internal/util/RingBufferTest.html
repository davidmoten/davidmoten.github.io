14      @Test
15      public void test() {
16          RingBuffer<Integer> r = new RingBuffer<>(MAX_SIZE);
17          assertEquals(0, r.size());
18          assertTrue(r.isEmpty());
19          assertNull(r.poll());
20          r.add(1).add(2);
21          assertFalse(r.isEmpty());
22          assertEquals(2, r.size());
23          assertEquals(1, (int) r.poll());
24          assertEquals(2, (int) r.poll());
25          assertTrue(r.isEmpty());
26          r.replay(1);
27          assertEquals(2, (int) r.poll());
28          r.replay(1);
29          assertEquals(2, (int) r.poll());
30          r.replay(2);
31          assertEquals(1, (int) r.poll());
32          assertEquals(2, (int) r.poll());
33          assertTrue(r.isEmpty());
34          r.add(3);
35          r.replay(1);
36          assertEquals(2, r.size());
37          assertEquals(2, (int) r.poll());
38          assertEquals(3, (int) r.poll());
39      }

41      @Test
42      public void testSize() {
43          RingBuffer<Integer> r = new RingBuffer<>(4);
44          r.add(1).add(2).add(3).add(4);
45          assertEquals(4, r.size());
46          assertEquals(1, (int) r.poll());
47          assertEquals(2, (int) r.poll());
48          assertEquals(3, (int) r.poll());
49          assertEquals(4, (int) r.poll());
50          assertEquals(0, r.size());
51          r.add(5).add(6).add(7);
52          assertEquals(3, r.size());
53      }
