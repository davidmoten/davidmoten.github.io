33      @Test
34      public void testStreamInputStream() throws IOException {
35          InputStream is = Stream.inputStream(
36                  Stream.bytes(() -> new ByteArrayInputStream("hello there".getBytes(StandardCharsets.UTF_8))));
37          try (BufferedReader br = new BufferedReader(new InputStreamReader(is, StandardCharsets.UTF_8))) {
38              assertEquals("hello there", br.readLine());
39              assertNull(br.readLine());
40          }
41      }
