12  public final class JsonArray {
13  
14      private final Stream<JsonParser> stream;
15      private final ObjectMapper mapper;
16  
17      JsonArray(Stream<JsonParser> flowable, ObjectMapper mapper) {
18          this.stream = flowable;
19          this.mapper = mapper;
20      }
21  
22      public <T> Stream<LazyTreeNode> nodes() {
23          return nodes_().map(p -> Util.MAPPER.readTree(p));
24      }
25  
26      public Stream<LazyObjectNode> objectNodes() {
27          return nodes_().map(p -> new LazyObjectNode(p));
28      }
29  
30      public Stream<LazyValueNode> valueNodes() {
31          return nodes_().map(p -> new LazyValueNode(p));
32      }
33  
34      public Stream<LazyArrayNode> arrayNodes() {
35          return nodes_().map(p -> new LazyArrayNode(p, mapper));
36      }
37  
38      public Stream<JsonNode> field(String name) {
39          return objectNodes().map(on -> on.get().get(name));
40      }
41  
42      private Stream<JsonParser> nodes_() {
43          return Stream.defer(() -> {
44              int[] depth = new int[1];
45              return stream //
46                      .doOnNext(p -> {
47                          JsonToken t = p.currentToken();
48                          if (t == JsonToken.START_OBJECT || t == JsonToken.START_ARRAY) {
49                              depth[0]++;
50                          } else if (t == JsonToken.END_OBJECT || t == JsonToken.END_ARRAY) {
51                              depth[0]--;
52                          }
53                      }) //
54                      .skipWhile(p -> p.currentToken() == JsonToken.FIELD_NAME
55                              || p.currentToken() == JsonToken.START_ARRAY) //
56                      .takeUntil(p -> depth[0] == 0) //
57                      .filter(p -> p.currentToken() != JsonToken.END_ARRAY
58                              && p.currentToken() != JsonToken.END_OBJECT); //
59          });
60      }

62  }
