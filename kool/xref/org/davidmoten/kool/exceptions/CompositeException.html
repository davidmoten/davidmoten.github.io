13  /**
14   * Represents an exception that is a composite of one or more other exceptions. A {@code CompositeException}
15   * does not modify the structure of any exception it wraps, but at print-time it iterates through the list of
16   * Throwables contained in the composite in order to print them all.
17   *
18   * Its invariant is to contain an immutable, ordered (by insertion order), unique list of non-composite
19   * exceptions. You can retrieve individual exceptions in this list with {@link #getExceptions()}.
20   *
21   * The {@link #printStackTrace()} implementation handles the StackTrace in a customized way instead of using
22   * {@code getCause()} so that it can avoid circular references.
23   *
24   * If you invoke {@link #getCause()}, it will lazily create the causal chain but will stop if it finds any
25   * Throwable in the chain that it has already seen.
26   */
27  public final class CompositeException extends RuntimeException {
28  
29      private static final long serialVersionUID = 3026362227162912146L;
30  
31      private final List<Throwable> exceptions;
32      private final String message;
33      private Throwable cause;
34  
35      /**
36       * Constructs a CompositeException with the given array of Throwables as the
37       * list of suppressed exceptions.
38       * @param exceptions the Throwables to have as initially suppressed exceptions
39       *
40       * @throws IllegalArgumentException if <code>exceptions</code> is empty.
41       */
42      public CompositeException(Throwable... exceptions) {
43          this(exceptions == null ?
44                  Collections.singletonList(new NullPointerException("exceptions was null")) : Arrays.asList(exceptions));
45      }
46  
47      /**
48       * Constructs a CompositeException with the given array of Throwables as the
49       * list of suppressed exceptions.
50       * @param errors the Throwables to have as initially suppressed exceptions
51       *
52       * @throws IllegalArgumentException if <code>errors</code> is empty.
53       */
54      public CompositeException(Iterable<? extends Throwable> errors) {
55          Set<Throwable> deDupedExceptions = new LinkedHashSet<Throwable>();
56          List<Throwable> localExceptions = new ArrayList<Throwable>();
57          if (errors != null) {
58              for (Throwable ex : errors) {
59                  if (ex instanceof CompositeException) {
60                      deDupedExceptions.addAll(((CompositeException) ex).getExceptions());
61                  } else
62                  if (ex != null) {
63                      deDupedExceptions.add(ex);
64                  } else {
65                      deDupedExceptions.add(new NullPointerException("Throwable was null!"));
66                  }
67              }
68          } else {
69              deDupedExceptions.add(new NullPointerException("errors was null"));
70          }
71          if (deDupedExceptions.isEmpty()) {
72              throw new IllegalArgumentException("errors is empty");
73          }
74          localExceptions.addAll(deDupedExceptions);
75          this.exceptions = Collections.unmodifiableList(localExceptions);
76          this.message = exceptions.size() + " exceptions occurred. ";
77      }
78  
79      /**
80       * Retrieves the list of exceptions that make up the {@code CompositeException}.
81       *
82       * @return the exceptions that make up the {@code CompositeException}, as a {@link List} of {@link Throwable}s
83       */
84      public List<Throwable> getExceptions() {
85          return exceptions;
86      }

88      @Override
89      public String getMessage() {
90          return message;

93      @Override
94      public synchronized Throwable getCause() { // NOPMD
95          if (cause == null) {
96              // we lazily generate this causal chain if this is called
97              CompositeExceptionCausalChain localCause = new CompositeExceptionCausalChain();
98              Set<Throwable> seenCauses = new HashSet<Throwable>();
99  
100             Throwable chain = localCause;
101             for (Throwable e : exceptions) {
102                 if (seenCauses.contains(e)) {
103                     // already seen this outer Throwable so skip
104                     continue;
105                 }
106                 seenCauses.add(e);
107 
108                 List<Throwable> listOfCauses = getListOfCauses(e);
109                 // check if any of them have been seen before
110                 for (Throwable child : listOfCauses) {
111                     if (seenCauses.contains(child)) {
112                         // already seen this outer Throwable so skip
113                         e = new RuntimeException("Duplicate found in causal chain so cropping to prevent loop ...");
114                         continue;
115                     }
116                     seenCauses.add(child);
117                 }
118 
119                 // we now have 'e' as the last in the chain
120                 try {
121                     chain.initCause(e);
122                 } catch (Throwable t) { // NOPMD
123                     // ignore
124                     // the JavaDocs say that some Throwables (depending on how they're made) will never
125                     // let me call initCause without blowing up even if it returns null
126                 }
127                 chain = getRootCause(chain);
128             }
129             cause = localCause;
130         }
131         return cause;
132     }

134     /**
135      * All of the following {@code printStackTrace} functionality is derived from JDK {@link Throwable}
136      * {@code printStackTrace}. In particular, the {@code PrintStreamOrWriter} abstraction is copied wholesale.
137      *
138      * Changes from the official JDK implementation:<ul>
139      * <li>no infinite loop detection</li>
140      * <li>smaller critical section holding {@link PrintStream} lock</li>
141      * <li>explicit knowledge about the exceptions {@link List} that this loops through</li>
142      * </ul>
143      */
144     @Override
145     public void printStackTrace() {
146         printStackTrace(System.err);
147     }

159     /**
160      * Special handling for printing out a {@code CompositeException}.
161      * Loops through all inner exceptions and prints them out.
162      *
163      * @param s
164      *            stream to print to
165      */
166     private void printStackTrace(PrintStreamOrWriter s) {
167         StringBuilder b = new StringBuilder(128);
168         b.append(this).append('\n');
169         for (StackTraceElement myStackElement : getStackTrace()) {
170             b.append("\tat ").append(myStackElement).append('\n');
171         }
172         int i = 1;
173         for (Throwable ex : exceptions) {
174             b.append("  ComposedException ").append(i).append(" :\n");
175             appendStackTrace(b, ex, "\t");
176             i++;
177         }
178         s.println(b.toString());
179     }
180 
181     private void appendStackTrace(StringBuilder b, Throwable ex, String prefix) {
182         b.append(prefix).append(ex).append('\n');
183         for (StackTraceElement stackElement : ex.getStackTrace()) {
184             b.append("\t\tat ").append(stackElement).append('\n');
185         }
186         if (ex.getCause() != null) {
187             b.append("\tCaused by: ");
188             appendStackTrace(b, ex.getCause(), "");
189         }
190     }

226     static final class CompositeExceptionCausalChain extends RuntimeException {
227         private static final long serialVersionUID = 3875212506787802066L;
228         /* package-private */static final String MESSAGE = "Chain of Causes for CompositeException In Order Received =>";
229 
230         @Override
231         public String getMessage() {
232             return MESSAGE;
233         }

236     private List<Throwable> getListOfCauses(Throwable ex) {
237         List<Throwable> list = new ArrayList<Throwable>();
238         Throwable root = ex.getCause();
239         if (root == null || root == ex) {
240             return list;
241         } else {
242             while (true) {
243                 list.add(root);
244                 Throwable cause = root.getCause();
245                 if (cause == null || cause == root) {
246                     return list;
247                 } else {
248                     root = cause;
249                 }
250             }
251         }
252     }

262     /**
263      * Returns the root cause of {@code e}. If {@code e.getCause()} returns {@code null} or {@code e}, just return {@code e} itself.
264      *
265      * @param e the {@link Throwable} {@code e}.
266      * @return The root cause of {@code e}. If {@code e.getCause()} returns {@code null} or {@code e}, just return {@code e} itself.
267      */
268     /*private */Throwable getRootCause(Throwable e) {
269         Throwable root = e.getCause();
270         if (root == null || cause == root) {
271             return e;
272         }
273         while (true) {
274             Throwable cause = root.getCause();
275             if (cause == null || cause == root) {
276                 return root;
277             }
278             root = cause;
279         }
280     }
