3   import java.util.Iterator;
4   
5   import org.davidmoten.kool.Stream;
6   import org.davidmoten.kool.StreamIterator;
7   
8   public final class StreamImpl<T> implements Stream<T> {
9   
10      private final Iterable<? extends T> source;
11  
12      public StreamImpl(Iterable<? extends T> source) {
13          this.source = source;
14      }
15  
16      @Override
17      public StreamIterator<T> iterator() {
18          return new StreamIterator<T>() {
19  
20              Iterator<? extends T> it = source.iterator();
21  
22              @Override
23              public boolean hasNext() {
24                  return it.hasNext();
25              }
26  
27              @Override
28              public T next() {
29                  return it.next();
30              }
31  
32              @Override
33              public void dispose() {
34                  // do nothing
35              }
36  
37          };
38      }
39  
40  }
