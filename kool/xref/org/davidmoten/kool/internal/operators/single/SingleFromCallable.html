3   import java.util.concurrent.Callable;
4   
5   import org.davidmoten.kool.Single;
6   import org.davidmoten.kool.internal.util.Exceptions;
7   
8   import com.github.davidmoten.guavamini.Preconditions;
9   
10  public final class SingleFromCallable<T> implements Single<T> {
11  
12      private final Callable<? extends T> callable;
13  
14      public SingleFromCallable(Callable<? extends T> callable) {
15          this.callable = callable;
16      }
17  
18      @Override
19      public T get() {
20          try {
21              return Preconditions.checkNotNull(callable.call(), "callable returned null!");
22          } catch (Exception e) {
23              return Exceptions.rethrow(e);
24          }
25      }
26  
27  }
