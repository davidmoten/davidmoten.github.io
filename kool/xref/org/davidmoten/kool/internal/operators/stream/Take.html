3   import java.util.NoSuchElementException;
4   
5   import org.davidmoten.kool.Stream;
6   import org.davidmoten.kool.StreamIterable;
7   import org.davidmoten.kool.StreamIterator;
8   
9   import com.github.davidmoten.guavamini.Preconditions;
10  
11  public final class Take<T> implements Stream<T> {
12  
13      private final long n;
14      private final StreamIterable<T> source;
15  
16      public Take(long n, StreamIterable<T> source) {
17          this.n = n;
18          this.source = source;
19      }
20  
21      @Override
22      public StreamIterator<T> iterator() {
23          return new StreamIterator<T>() {
24  
25              final StreamIterator<T> it = source.iteratorNullChecked();
26              int count;
27  
28              @Override
29              public boolean hasNext() {
30                  return count < n && Preconditions.checkNotNull(it.hasNext());
31              }
32  
33              @Override
34              public T next() {
35                  if (count == n || !it.hasNext()) {
36                      it.dispose();
37                      throw new NoSuchElementException();
38                  } else {
39                      count++;
40                      return it.next();
41                  }
42              }
43  
44              @Override
45              public void dispose() {
46                  it.dispose();
47              }
48  
49          };
50      }
51  
52  }
