3   import java.util.NoSuchElementException;
4   import java.util.concurrent.TimeUnit;
5   
6   import org.davidmoten.kool.Single;
7   import org.davidmoten.kool.Stream;
8   import org.davidmoten.kool.StreamIterator;
9   import org.davidmoten.kool.function.Function;
10  import org.davidmoten.kool.function.Predicate;
11  import org.davidmoten.kool.internal.util.Exceptions;
12  
13  public final class RetryWhen<T> implements Stream<T> {
14  
15      private final Stream<T> stream;
16      private final Function<? super Throwable, ? extends Single<?>> function;
17  
18      public RetryWhen(Stream<T> stream, Function<? super Throwable, ? extends Single<?>> function) {
19          this.stream = stream;
20          this.function = function;
21      }
22  
23      @Override
24      public StreamIterator<T> iterator() {
25          return new StreamIterator<T>() {
26              StreamIterator<T> it;
27              T next;
28              boolean finished;
29  
30              @Override
31              public boolean hasNext() {
32                  load();
33                  return next != null;
34              }
35  
36              @Override
37              public T next() {
38                  load();
39                  if (next == null) {
40                      throw new NoSuchElementException();
41                  } else {
42                      T v = next;
43                      next = null;
44                      return v;
45                  }
46              }
47  
48              public void load() {
49                  if (!finished && next == null) {
50                      while (true) {
51                          try {
52                              if (it == null) {
53                                  it = stream.iteratorNullChecked();
54                              }
55                              if (it.hasNext()) {
56                                  next = it.nextNullChecked();
57                              } else {
58                                  finished = true;
59                              }
60                              break;
61                          } catch (Throwable e) {
62                              // will rethrow if no more retries
63                              function.applyUnchecked(e).get();
64                              try {
65                                  dispose();
66                              } catch (Throwable t) {
67                                  function.applyUnchecked(e).get();
68                              }
69                          }
70                      }
71                  }
72              }
73  
74              @Override
75              public void dispose() {
76                  if (it != null) {
77                      it.dispose();
78                      it = null;
79                      next = null;
80                  }
81              }
82  
83          };
84      }
85  
86      public static <T> Stream<T> build(Stream<T> stream, Stream<Long> delays, int maxRetries,
87              Predicate<? super Throwable> predicate) {
88          return Stream.defer(() -> {
89              int[] retryNumber = new int[1];
90              StreamIterator<Long> delaysIt = delays == null ? null : delays.iteratorNullChecked();
91              Function<Throwable, Single<?>> function = e -> {
92                  retryNumber[0]++;
93                  if (maxRetries > 0) {
94                      if (retryNumber[0] > maxRetries) {
95                          return Exceptions.rethrow(e);
96                      }
97                  }
98                  if (predicate != null) {
99                      if (!predicate.test(e)) {
100                         return Exceptions.rethrow(e);
101                     }
102                 }
103                 if (delaysIt != null) {
104                     if (delaysIt.hasNext()) {
105                         long ms = delaysIt.nextNullChecked();
106                         return Single.timer(ms, TimeUnit.MILLISECONDS);
107                     } else {
108                         return Exceptions.rethrow(e);
109                     }
110                 }
111                 return Single.of(1);
112             };
113             return new RetryWhen<T>(stream, function);
114         });
115     }
