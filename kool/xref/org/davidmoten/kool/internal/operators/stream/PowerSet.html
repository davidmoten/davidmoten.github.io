4   import java.util.Set;

6   import org.davidmoten.kool.Stream;

9   import com.github.davidmoten.guavamini.Sets;
10  
11  public class PowerSet implements Stream<Set<Integer>> {
12  
13      private final int n;
14  
15      public PowerSet(int n) {
16          this.n = n;
17      }
18  
19      @Override
20      public StreamIterator<Set<Integer>> iterator() {
21          return new StreamIterator<Set<Integer>>() {
22  
23              boolean[] b = new boolean[n];
24  
25              @Override
26              public boolean hasNext() {
27                  return b != null;
28              }
29  
30              @Override
31              public Set<Integer> next() {
32                  if (b == null) {
33                      throw new NoSuchElementException();
34                  }
35                  Set<Integer> set = Sets.newHashSet();
36                  for (int i = 0; i < b.length; i++) {
37                      if (b[i]) {
38                          set.add(i + 1);
39                      }
40                  }
41                  int i = 0;
42                  while (true) {
43                      if (i == b.length) {
44                          break;
45                      }
46                      boolean x = b[i];
47                      b[i] = !b[i];
48                      if (!x) {
49                          break;
50                      }
51                      i++;
52                  }
53                  if (i == b.length) {
54                      b = null;
55                  }
56                  return set;
57              }
58  
59              @Override
60              public void dispose() {
61                  // do nothing
62              }
63          };
64      }
65  
66  }
