1   package org.davidmoten.kool.internal.operators.stream;
2   
3   import java.util.Comparator;
4   import java.util.Optional;
5   
6   import org.davidmoten.kool.Maybe;
7   import org.davidmoten.kool.Stream;
8   import org.davidmoten.kool.StreamIterator;
9   
10  public class Max<T> implements Maybe<T> {
11  
12      private final Stream<T> stream;
13      private final Comparator<? super T> comparator;
14      private final boolean not;
15  
16      public Max(Stream<T> stream, Comparator<? super T> comparator, boolean not) {
17          this.stream = stream;
18          this.comparator = comparator;
19          this.not = not;
20      }
21  
22      @Override
23      public Optional<T> get() {
24          StreamIterator<T> it = stream.iteratorNullChecked();
25          try {
26              T max = null;
27              while (it.hasNext()) {
28                  T v = it.nextNullChecked();
29                  if (max == null) {
30                      max = v;
31                  } else {
32                      boolean comp = comparator.compare(max, v) < 0;
33                      if (comp ^ not) {
34                          max = v;
35                      }
36                  }
37              }
38              return Optional.ofNullable(max);
39          } finally {
40              it.dispose();
41          }
42      }
43  
44  }
