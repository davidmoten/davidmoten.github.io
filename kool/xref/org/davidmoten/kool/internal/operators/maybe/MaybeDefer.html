3   import java.util.Optional;
4   import java.util.concurrent.Callable;
5   
6   import org.davidmoten.kool.Maybe;
7   import org.davidmoten.kool.internal.util.Exceptions;
8   
9   public final class MaybeDefer<T> implements Maybe<T> {
10  
11      private final Callable<? extends Maybe<? extends T>> factory;
12  
13      public MaybeDefer(Callable<? extends Maybe<? extends T>> factory) {
14          this.factory = factory;
15      }
16  
17      @SuppressWarnings("unchecked")
18      @Override
19      public Optional<T> get() {
20          try {
21              return (Optional<T>) factory.call().get();
22          } catch (Throwable e) {
23              return Exceptions.rethrow(e);
24          }
25      }
26  
27  }
