433     /**
434      * Emits the integers 0, 1, 2, .... The first element is emitted immediately (0)
435      * and then the current thread is blocked with {@code Thread.sleep} for the
436      * given duration between further emissions.
437      * 
438      * If you don't want the stream to start with 0 immediately then call
439      * {@code interval(...).skip(1)}.
440      * 
441      * @param duration sleep duration
442      * @param unit     unit of sleep duration
443      * @return stream with interval wait between emissions
444      */
445     public static Stream<Integer> interval(long duration, TimeUnit unit) {
446         return range(1, Integer.MAX_VALUE).doOnNext(x -> unit.sleep(duration)).prepend(0);

527     public default <A, R> Single<R> collect(java.util.stream.Collector<T, A, R> collector) {
528 
529         Supplier<A> supplier = collector.supplier();
530         java.util.function.BiConsumer<A, T> accumulator = collector.accumulator();
531         Function<A, R> finisher = new Function<A, R>() {
532             final java.util.function.Function<A, R> finisher = collector.finisher();
533 
534             @Override
535             public R apply(A a) throws Exception {
536                 return finisher.apply(a);
537             }
538         };
539         return collect(new Collector<T, A>() {
540 
541             @Override
542             public A call() throws Exception {
543                 return supplier.get();
544             }
545 
546             @Override
547             public A apply(A a, T t) throws Exception {
548                 accumulator.accept(a, t);
549                 return a;
550             }
551 
552         }).map(finisher);
553     }

555     public default <M extends java.util.Map<K, D>, K, V, D extends Collection<V>> Single<M> groupBy( //
556             Callable<M> mapFactory, //
557             Function<? super T, ? extends K> keySelector, //
558             Function<? super T, ? extends V> valueSelector, //
559             Callable<D> collectionFactory) {
560         return collect(mapFactory, (map, t) -> {
561             K k = keySelector.apply(t);
562             V v = valueSelector.apply(t);
563             D x = map.get(k);
564             if (x == null) {
565                 try {
566                     x = collectionFactory.call();
567                     map.put(k, x);
568                 } catch (Exception e) {
569                     Exceptions.rethrow(e);
570                 }
571             }
572             x.add(v);
573         });
574     }

1064     public static <T> Stream<List<Integer>> permutations(int size) {
1065         List<Integer> indexes = new ArrayList<Integer>(size);
1066         for (int i = 0; i < size; i++) {
1067             indexes.add(i);
1068         }
1069         return Stream.from(Permutations.iterable(indexes)) //
1070                 .scan(indexes, (a, swap) -> {
1071                     List<Integer> b = new ArrayList<Integer>(a);
1072                     b.set(swap.left(), a.get(swap.right()));
1073                     b.set(swap.right(), a.get(swap.left()));
1074                     return b;
1075                 }) //
1076                 .prepend(new ArrayList<>(indexes));
1077     }
