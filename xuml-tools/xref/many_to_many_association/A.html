339     public static Optional<A> find(String id) {
340         if (Context.em()!=null) {
341             return Optional.fromNullable(Context.em().find(A.class,id));
342         } else {
343             EntityManager em = Context.createEntityManager();
344             try {
345                 A result = em.find(A.class,id);
346                 return Optional.fromNullable(result);

348                 em.close();
