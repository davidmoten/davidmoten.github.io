152     @Column(name="description", nullable=false,length=4096)
153     private String description = new String("");
154 
155     /**
156      * Validates description against type constraints.
157      */
158     private void validateDescription() {
159         if (description == null || description.length() < 1)
160             throw new ValidationException("min length constraint not met");
161          if (description == null || !description.startsWith(""))
162             throw new ValidationException("prefix constraint not met");
163         if (description == null || !description.endsWith(""))
164             throw new ValidationException("suffix constraint not met");
165         if (description == null || !Pattern.matches(".*", description))
166             throw new ValidationException("validation pattern constraint not met");
167     }

341     private void validateSpecializationR12() {
342         int count = 0;
343         if (stockedProduct_R12 != null)
344             count++;
345         if (specialOrderProduct_R12 != null)
346             count++;
347         if (count == 0)
348             throw new RelationshipNotEstablishedException("wrong number of specializations = " + count);
349         if (count != 1)
350             throw new TooManySpecializationsException("wrong number of specializations = " + count);
351         }
