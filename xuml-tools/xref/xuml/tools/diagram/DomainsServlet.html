3   import java.io.IOException;
4   import java.io.PrintWriter;
5   
6   import javax.servlet.ServletException;
7   import javax.servlet.http.HttpServlet;
8   import javax.servlet.http.HttpServletRequest;
9   import javax.servlet.http.HttpServletResponse;
10  import javax.xml.bind.JAXBElement;
11  
12  
13  import org.apache.commons.io.IOUtils;
14  
15  import xuml.tools.miuml.metamodel.jaxb.Domain;
16  import xuml.tools.miuml.metamodel.jaxb.Domains;
17  import xuml.tools.miuml.metamodel.jaxb.Marshaller;
18  import xuml.tools.miuml.metamodel.jaxb.ModeledDomain;
19  import xuml.tools.miuml.metamodel.jaxb.Subsystem;
20  import xuml.tools.util.xml.TaggedString;
21  
22  public class DomainsServlet extends HttpServlet {
23  
24  	private static final long serialVersionUID = 2511746331522695068L;
25  
26  	@Override
27  	protected void doGet(HttpServletRequest req, HttpServletResponse resp)
28  			throws ServletException, IOException {
29  		String id = req.getParameter("id");
30  		String xml = Context.instance().getDatastore()
31  				.get("diagram", id + "-model", "model");
32  		Domains domains = new Marshaller()
33  				.unmarshal(IOUtils.toInputStream(xml));
34  
35  		TaggedString t = new TaggedString();
36  		for (JAXBElement<? extends Domain> domain : domains.getDomain()) {
37  			t.startTag("h2");
38  			t.append(domain.getValue().getName());
39  			t.closeTag();
40  			if (domain.getValue() instanceof ModeledDomain) {
41  				ModeledDomain d = (ModeledDomain) domain.getValue();
42  				int ssNumber = 0;
43  				for (Subsystem ss : d.getSubsystem()) {
44  					ssNumber++;
45  					t.startTag("a");
46  					t.addAttribute("href", "cd?id=" + id + "&ss=" + ssNumber);
47  					t.append(ss.getName());
48  					t.closeTag();
49  					t.startTag("br");
50  					t.closeTag();
51  				}
52  			}
53  		}
54  		t.close();
55  
56  		resp.setContentType("text/html");
57  		PrintWriter out = resp.getWriter();
58  		out.print(t);
59  	}
60  }
