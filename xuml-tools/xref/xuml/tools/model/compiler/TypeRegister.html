1   package xuml.tools.model.compiler;
2   
3   import java.util.TreeSet;
4   
5   import com.google.common.collect.BiMap;
6   import com.google.common.collect.HashBiMap;
7   
8   public class TypeRegister {
9   
10  	/**
11  	 * Full type name -> abbr (if possible)
12  	 */
13  	private final BiMap<String, String> types = HashBiMap.create();
14  
15  	public String addType(java.lang.Class<?> clsWithoutGenerics) {
16  		return addType(new Type(clsWithoutGenerics.getName()));
17  	}
18  
19  	public String addType(Type type) {
20  		StringBuilder result = new StringBuilder(addType(type.getBase()));
21  		StringBuilder typeParams = new StringBuilder();
22  		for (Type t : type.getGenerics()) {
23  			String typeParameter = addType(t);
24  			if (typeParams.length() > 0)
25  				typeParams.append(",");
26  			typeParams.append(typeParameter);
27  		}
28  		if (typeParams.length() > 0) {
29  			result.append("<");
30  			result.append(typeParams);
31  			result.append(">");
32  		}
33  		return result.toString();
34  	}
35  
36  	String addType(String type) {
37  		String abbr = types.get(type);
38  		if (abbr != null)
39  			return abbr;
40  		else {
41  			int i = type.lastIndexOf(".");
42  			if (i >= 0) {
43  				String last = type.substring(i + 1);
44  				if (types.inverse().get(last) != null)
45  					return type;
46  				else {
47  					types.put(type, last);
48  					return last;
49  				}
50  			} else
51  				return type;
52  		}
53  	}
54  
55  	public String getImports(String relativeToClass) {
56  		TreeSet<String> set = new TreeSet<String>(types.keySet());
57  		StringBuilder s = new StringBuilder();
58  		for (String t : set) {
59  			boolean isImmediateChildOfRelativeClass = t
60  					.startsWith(relativeToClass)
61  					&& t.length() > relativeToClass.length()
62  					&& t.indexOf('.', relativeToClass.length() + 1) == -1;
63  			if (!isImmediateChildOfRelativeClass)
64  				s.append("import " + t + ";\n");
65  		}
66  		return s.toString();
67  	}
68  
69  	public void addTypes(java.lang.Class<?>... classes) {
70  		for (Class<?> cls : classes) {
71  			addType(cls);
72  		}
73  	}
74  }
