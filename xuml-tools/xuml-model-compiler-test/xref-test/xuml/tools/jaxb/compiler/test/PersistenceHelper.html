11  import com.google.common.collect.Maps;

13  public class PersistenceHelper {
14  
15  	public static enum DatabaseType {
16  		DERBY, H2, HSQLDB, OBJECT_DB;
17  	}
18  
19  	public static EntityManagerFactory createEmf(String name,
20  			final String schema, DatabaseType type) {
21  		return Persistence.createEntityManagerFactory(name,
22  				getProperties(name, schema, type));
23  	}
24  
25  	public static EntityManagerFactory createEmf(String name, String schema) {
26  		return createEmf(name, schema, DatabaseType.H2);
27  	}
28  
29  	public static EntityManagerFactory createEmf(String name) {
30  		return createEmf(name, name.replace("-", "_"));
31  	}
32  
33  	private static Map<String, String> getProperties(String name,
34  			String schema, DatabaseType type) {
35  		Map<String, String> map = Maps.newHashMap();
36  		map.put("hibernate.hbm2ddl.auto", "create-drop");
37  		map.put("hibernate.connection.pool_size", "5");
38  		map.put("hibernate.show_sql", "true");
39  		map.put("hibernate.format_sql", "true");
40  		if (type == DatabaseType.DERBY) {
41  			String dialect = "org.hibernate.dialect.DerbyDialect";
42  			String driver = "org.apache.derby.jdbc.EmbeddedDriver";
43  			String url = "jdbc:derby:memory:" + name + "-db;create=true";
44  			insertDatabaseSpecificProperties(map, dialect, driver, url);
45  		} else if (type == DatabaseType.H2) {
46  			String dialect = "org.hibernate.dialect.H2Dialect";
47  			String driver = "org.h2.Driver";
48  			String url = "jdbc:h2:mem:target/" + name
49  					+ "-db;INIT=CREATE SCHEMA IF NOT EXISTS " + schema;
50  			insertDatabaseSpecificProperties(map, dialect, driver, url);
51  		} else if (type == DatabaseType.HSQLDB) {
52  			String dialect = "org.hibernate.dialect.HSQLDialect";
53  			String driver = "org.hsqldb.jdbcDriver";
54  			String url = "jdbc:hsqldb:file:target/" + name + "-db";
55  			createSchema(url, schema);
56  			insertDatabaseSpecificProperties(map, dialect, driver, url);
57  		}
58  		return map;
59  	}
60  
61  	private static void createSchema(String url, String schema) {
62  		try {
63  			Connection con = DriverManager.getConnection(url);
64  			con.createStatement().execute(
65  					"create schema " + schema + " authorization sa");
66  			con.close();
67  		} catch (SQLException e) {
68  			throw new RuntimeException(e);
69  		}
70  
71  	}
72  
73  	private static void insertDatabaseSpecificProperties(
74  			Map<String, String> map, String dialect, String driver, String url) {
75  		map.put("hibernate.dialect", dialect);
76  		map.put("hibernate.connection.url", url);
77  		map.put("hibernate.connection.driver_class", driver);
78  	}
