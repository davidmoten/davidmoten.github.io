112     /**
113      * Validates name against type constraints.
114      */
115     private void validateName() {
116         if (name == null || name.length() < 1)
117             throw new ValidationException("min length constraint not met");
118          if (name == null || !name.startsWith(""))
119             throw new ValidationException("prefix constraint not met");
120         if (name == null || !name.endsWith(""))
121             throw new ValidationException("suffix constraint not met");
122         if (name == null || !Pattern.matches(".*", name))
123             throw new ValidationException("validation pattern constraint not met");
124     }

160     @Column(name="purchases_made", nullable=false)
161     private Integer purchasesMade = new Integer("0");
162 
163     private static final BigDecimal PURCHASESMADE_UPPER_LIMIT=new BigDecimal("999999999999");
164     private static final BigDecimal PURCHASESMADE_LOWER_LIMIT=new BigDecimal("0");
165 
166     /**
167      * Validates purchasesMade against type constraints.
168      */
169     private void validatePurchasesMade() {
170         if (PURCHASESMADE_UPPER_LIMIT.doubleValue() < purchasesMade) 
171             throw new ValidationException("upper limit of 999999999999 failed");
172         if (PURCHASESMADE_LOWER_LIMIT.doubleValue() > purchasesMade)
173              throw new ValidationException("lower limit of 0 failed");
174     }

422     public static Optional<Customer> find(String id) {
423         if (Context.em()!=null) {
424             return Optional.fromNullable(Context.em().find(Customer.class,id));
425         } else {
426             EntityManager em = Context.createEntityManager();
427             try {
428                 Customer result = em.find(Customer.class,id);
429                 return Optional.fromNullable(result);
430             } finally {
431                 em.close();
432             }
433         }
434     }
