1   package xuml.tools.model.compiler.runtime;

3   import xuml.tools.model.compiler.runtime.SignalProcessorListener;
4   import xuml.tools.model.compiler.runtime.actor.EntityActor;
5   import xuml.tools.model.compiler.runtime.message.Signal;
6   
7   public class SignalProcessorListenerStopsAllSignalProcessingOnFailure implements
8   		SignalProcessorListener {
9   
10  	private static boolean failureOccured = false;
11  
12  	@Override
13  	public void beforeProcessing(Signal<?> signal, EntityActor actor) {
14  		if (failureOccured)
15  			// Note that this throw will also prompt a call to the failure
16  			// method below.
17  			throw new FailedProcessingException(
18  					"failure has occurred, no further processing");
19  	}
20  
21  	@Override
22  	public void afterProcessing(Signal<?> signal, EntityActor actor) {
23  		// do nothing
24  	}
25  
26  	@Override
27  	public void failure(Signal<?> signal, Exception e, EntityActor actor) {
28  		if (!(e instanceof FailedProcessingException))
29  			failureOccured = true;
30  	}
31  
32  	public static class FailedProcessingException extends RuntimeException {
33  
34  		private static final long serialVersionUID = 1L;
35  
36  		public FailedProcessingException(String message) {
37  			super(message);
38  		}
39  
40  	}
41  
42  }
