36      /**
37       * Demonstrates the major aspects of using entities generated by
38       * xuml-model-compiler including reloading signals not fully processed at
39       * time of last shutdown, creation of entities and asynchronous signalling.
40       * 
41       * @throws InterruptedException
42       */
43      @Test
44      public void testCreateEntityManagerFactoryAndCreateAndSignalEntities()
45              throws InterruptedException {
46  
47          // Note that the classes Context and A are generated from xml using
48          // the xuml-tools-maven-plugin
49  
50          // see the setup() method below for how the Context is initialized.
51  
52          // send any signals not processed from last shutdown
53          Context.sendSignalsInQueue();
54  
55          // create some entities (this happens synchronously)
56          A a1 = A.create(new A.Events.Create("value1.1", "value2.1", "1234"));
57          assertEquals("value1.1", a1.getId().getAOne());
58          A a2 = A.create(new A.Events.Create("value1.2", "value2.2", "1234"));
59          // use Builder pattern for this one (all Events and Id classes have
60          // builders)
61          A a3 = A.create(A.Events.Create.builder().aOne("value1.3").aTwo("value2.3")
62                  .accountNumber("1234").build());
63  
64          // send asynchronous signals to a1 and a2
65          a1.signal(new A.Events.SomethingDone(11));
66          a2.signal(new A.Events.SomethingDone(12));
67          // send asynchronous signal to a3 after a tiny delay (5ms)
68          a3.signal(new A.Events.SomethingDone(13), Duration.create(5, TimeUnit.MILLISECONDS));
69  
70          // wait a bit for all signals to be processed
71          Thread.sleep(2000);
72  
73          // Check the signals were processed
74  
75          // The load method reloads the entity using a fresh entity manager and
76          // then closes the entity manager. As a consequence only non-proxied
77          // fields of the entity will be retrievable using the load method.
78          assertEquals("11", a1.load().getAThree());
79          assertEquals("12", a2.load().getAThree());
80          assertEquals("13", a3.load().getAThree());
81  
82          // Notice that all the above could be done without explicitly creating
83          // EntityManagers at all. Nice!
84  
85          // Refresh the entities from the database using the
86          // load(em) method and check the signals were processed
87          EntityManager em = Context.createEntityManager();
88          // note that the load method below does an em merge and refresh and
89          // returns a new entity for use within the current entity manager
90          assertEquals("11", a1.load(em).getAThree());
91          assertEquals("12", a2.load(em).getAThree());
92          assertEquals("13", a3.load(em).getAThree());
93  
94          // demonstrate select statements
95          {
96              List<A> list = A.select().many(em);
97              assertTrue(list.size() >= 3);
98          }
99  
100         {
101             List<A> list = A.select(aOne.eq("value1.1")).many(em);
102             assertEquals(1, list.size());
103             assertFalse(A.select(aOne.eq("zz")).any(em).isPresent());
104             assertFalse(A.select(aOne.eq("zz")).one(em).isPresent());
105         }
106 
107         em.close();
108 
109     }

145     /**
146      * Tests signal persistence.
147      * 
148      * @throws InterruptedException
149      */
150     @Test
151     public void testEntitiesAndSignalsArePersisted() throws InterruptedException {
152 
153         A a = A.create(new A.Events.Create("value1.4", "value2.4", "1234"));
154         // check that the entity was persisted
155         {
156             EntityManager em = Context.createEntityManager();
157             Assert.assertNotNull(
158                     em.find(A.class, AId.builder().aOne("value1.4").aTwo("value2.4").build()));
159             em.close();
160         }
161 
162         // test the reloading of a persisted signal to a1
163         Context.persistSignal("fromSomeId", a.getId(), A.class, new A.Events.SomethingDone(14),
164                 System.currentTimeMillis(), Optional.<Long> absent(), a.uniqueId());
165         assertEquals(1, Context.sendSignalsInQueue());
166         // wait a bit for all signals to be processed
167         Thread.sleep(2000);
168 
169         // Refresh the entities from the database using the
170         // load method and check the signals were processed
171         EntityManager em = Context.createEntityManager();
172         // note that the load method below does an em merge and refresh and
173         // returns a new entity for use within the current entity manager
174         assertEquals("14", a.load(em).getAThree());
175         em.close();
176 
177     }

187      * Behaviour for an A entity.

189      * @author dxm
190      * 
191      */
192     public static class ABehaviour implements A.Behaviour {
193 
194         private final A self;
195 
196         public ABehaviour(A self) {
197             this.self = self;
198         }
199 
200         @Override
201         public void onEntryHasStarted(Create event) {
202             self.setId(AId.builder().aOne(event.getAOne()).aTwo(event.getATwo()).build());
203             self.setAThree(event.getAccountNumber());
204             System.out.println("created");
205         }
206 
207         @Override
208         public void onEntryDoneSomething(StateSignature_DoneSomething event) {
209             // use the method chaining version of setAThree
210             // (underscore appended)
211             System.out.println(self.setAThree_(event.getTheCount() + "").getId());
212             System.out.println("setting A.athree=" + self.getAThree() + " for " + self.getId());
213             // demonstrate/unit test getting access to the current
214             // entity manager when needed
215             Object count = Context.em().createQuery("select count(b) from B b").getResultList();
216             System.out.println("counted " + count + " B entities");
217 
218             // also demonstrate select many using a SelectBuilder
219             List<A> list = A.select(aOne.eq("value1.1")).many();
220             System.out.println("list size = " + list.size());
221         }

238     /**
239      * This listener logs activity using the Akka actor system logger.
240      * 
241      * @return
242      */
243     private static SignalProcessorListener createSignalProcessorListener() {
244         return new SignalProcessorListener() {
245 
246             private int processed = 0;
247 
248             @Override
249             public void beforeProcessing(Signal<?> signal, EntityActor actor) {
250                 LoggingAdapter log = Logging.getLogger(actor.getContext().system(), this);
251                 log.info("before processing");
252             }
253 
254             @Override
255             public void afterProcessing(Signal<?> signal, EntityActor actor) {
256                 // count the number processed and log it
257                 processed++;
258                 LoggingAdapter log = Logging.getLogger(actor.getContext().system(), this);
259                 log.info("after processing " + processed);
260             }
261 
262             @Override
263             public void failure(Signal<?> signal, Exception e, EntityActor actor) {
264                 LoggingAdapter log = Logging.getLogger(actor.getContext().system(), this);
265                 log.error(e, e.getMessage());
266             }
267         };
