30      @Test
31      public void testSignalPersistence() throws ClassNotFoundException {
32  
33          EntityManager em = Context.createEntityManager();
34          em.getTransaction().begin();
35          A.create(new A.AId("hello2", "there2")).persist(em);
36          em.getTransaction().commit();
37  
38          // now test can find using id
39          em.getTransaction().begin();
40          assertNotNull(em.find(A.class, new A.AId("hello2", "there2")));
41          em.getTransaction().commit();
42  
43          // now test can find using id and Class.forName
44          em.getTransaction().begin();
45          assertNotNull(em.find(Class.forName(A.class.getName()), new A.AId("hello2", "there2")));
46          em.getTransaction().commit();
47  
48          em.close();
49  
50          // now test can find using new em, Class.forName and reconstituted id
51          em = Context.createEntityManager();
52          em.getTransaction().begin();
53          AId id = new A.AId("hello2", "there2");
54          A.AId id2 = Util.toObject(Util.toBytes(id), A.AId.class);
55          assertNotNull(em.find(Class.forName(A.class.getName()), id2));
56          em.getTransaction().commit();
57  
58          em.close();
59      }
