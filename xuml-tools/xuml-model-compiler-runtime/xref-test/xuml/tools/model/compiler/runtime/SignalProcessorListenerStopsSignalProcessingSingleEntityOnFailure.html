1   package xuml.tools.model.compiler.runtime;

3   import java.util.Set;
4   
5   import com.google.common.collect.Sets;
6   
7   import xuml.tools.model.compiler.runtime.actor.EntityActor;
8   import xuml.tools.model.compiler.runtime.message.Signal;
9   
10  public class SignalProcessorListenerStopsSignalProcessingSingleEntityOnFailure
11          implements SignalProcessorListener {
12  
13      private static Set<String> stoppedEntities = Sets.newHashSet();
14  
15      @Override
16      public synchronized void beforeProcessing(Signal<?> signal, EntityActor actor) {
17          if (stoppedEntities.contains(signal.getEntityUniqueId()))
18              throw new SignalProcessingStoppedException(signal.getEntityUniqueId());
19      }
20  
21      @Override
22      public void afterProcessing(Signal<?> signal, EntityActor actor) {
23          // do nothing
24      }
25  
26      @Override
27      public synchronized void failure(Signal<?> signal, Exception e, EntityActor actor) {
28          if (!(e instanceof SignalProcessingStoppedException))
29              stoppedEntities.add(signal.getEntityUniqueId());
30      }
31  
32      public static class SignalProcessingStoppedException extends RuntimeException {
33  
34          private static final long serialVersionUID = 4537196557689877257L;
35  
36          public SignalProcessingStoppedException(String message) {
37              super(message);
38          }
39  
40      }
41  
42  }
