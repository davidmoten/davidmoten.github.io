1   package xuml.tools.model.compiler.runtime;

3   import java.util.Set;
4   
5   import xuml.tools.model.compiler.runtime.SignalProcessorListener;
6   import xuml.tools.model.compiler.runtime.actor.EntityActor;
7   import xuml.tools.model.compiler.runtime.message.Signal;
8   
9   import com.google.common.collect.Sets;
10  
11  public class SignalProcessorListenerStopsSignalProcessingSingleEntityOnFailure
12  		implements SignalProcessorListener {
13  
14  	private static Set<String> stoppedEntities = Sets.newHashSet();
15  
16  	@Override
17  	public synchronized void beforeProcessing(Signal<?> signal,
18  			EntityActor actor) {
19  		if (stoppedEntities.contains(signal.getEntity().uniqueId()))
20  			throw new SignalProcessingStoppedException(signal.getEntity()
21  					.uniqueId());
22  	}
23  
24  	@Override
25  	public void afterProcessing(Signal<?> signal, EntityActor actor) {
26  		// do nothing
27  	}
28  
29  	@Override
30  	public synchronized void failure(Signal<?> signal, Exception e,
31  			EntityActor actor) {
32  		if (!(e instanceof SignalProcessingStoppedException))
33  			stoppedEntities.add(signal.getEntity().uniqueId());
34  	}
35  
36  	public static class SignalProcessingStoppedException extends
37  			RuntimeException {
38  
39  		private static final long serialVersionUID = 4537196557689877257L;
40  
41  		public SignalProcessingStoppedException(String message) {
42  			super(message);
43  		}
44  
45  	}
46  
47  }
