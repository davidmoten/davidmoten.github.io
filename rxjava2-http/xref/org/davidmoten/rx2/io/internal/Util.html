31      public static long toLong(byte[] b) {
32          return (((long) b[0] << 56) //
33                  + ((long) (b[1] & 255) << 48) //
34                  + ((long) (b[2] & 255) << 40) //
35                  + ((long) (b[3] & 255) << 32) //
36                  + ((long) (b[4] & 255) << 24) //
37                  + ((b[5] & 255) << 16) //
38                  + ((b[6] & 255) << 8) //
39                  + ((b[7] & 255) << 0));

42      public static byte[] toBytes(long v) {
43          byte[] b = new byte[8];
44          b[0] = (byte) (v >>> 56);
45          b[1] = (byte) (v >>> 48);
46          b[2] = (byte) (v >>> 40);
47          b[3] = (byte) (v >>> 32);
48          b[4] = (byte) (v >>> 24);
49          b[5] = (byte) (v >>> 16);
50          b[6] = (byte) (v >>> 8);
51          b[7] = (byte) (v >>> 0);
52          return b;

55      public static byte[] toBytes(int v) throws IOException {
56          byte[] b = new byte[4];
57          b[0] = (byte) ((v >>> 24) & 0xFF);
58          b[1] = (byte) ((v >>> 16) & 0xFF);
59          b[2] = (byte) ((v >>> 8) & 0xFF);
60          b[3] = (byte) ((v >>> 0) & 0xFF);
61          return b;

74      // copied from DataInputStream so don't need to instantiate one
75      public static void readFully(InputStream in, byte b[], int off, int len) throws IOException {
76          if (len < 0) {
77              throw new IndexOutOfBoundsException();
78          }
79          int n = 0;
80          while (n < len) {
81              int count = in.read(b, off + n, len - n);
82              if (count < 0) {
83                  throw new EOFException();
84              }
85              n += count;
86          }
87      }

89      public static long readLong(InputStream in) throws IOException {
90          byte[] b = new byte[8];
91          readFully(in, b, 0, 8);
92          return (((long) b[0] << 56) + ((long) (b[1] & 255) << 48) + ((long) (b[2] & 255) << 40)
93                  + ((long) (b[3] & 255) << 32) + ((long) (b[4] & 255) << 24) + ((b[5] & 255) << 16)
94                  + ((b[6] & 255) << 8) + ((b[7] & 255) << 0));
95      }

97      public static int readInt(InputStream in) throws IOException {
98          int ch1 = in.read();
99          int ch2 = in.read();
100         int ch3 = in.read();
101         int ch4 = in.read();
102         if ((ch1 | ch2 | ch3 | ch4) < 0)
103             throw new EOFException();
104         return ((ch1 << 24) + (ch2 << 16) + (ch3 << 8) + (ch4 << 0));
105     }

107     public static byte[] toBytes(ByteBuffer bb) {
108         int p = bb.position();
109         byte[] bytes = new byte[bb.remaining()];
110         bb.get(bytes);
111         bb.position(p);
112         return bytes;
113     }
