3   import java.util.concurrent.atomic.AtomicLong;
4   import java.util.concurrent.atomic.AtomicReference;

6   import org.reactivestreams.Publisher;
7   import org.reactivestreams.Subscriber;
8   import org.reactivestreams.Subscription;
9   
10  import io.reactivex.Flowable;
11  import io.reactivex.FlowableSubscriber;
12  import io.reactivex.SingleObserver;
13  import io.reactivex.SingleSource;
14  import io.reactivex.disposables.Disposable;
15  import io.reactivex.functions.Function;
16  import io.reactivex.internal.subscriptions.SubscriptionHelper;
17  
18  public final class FlowableSingleFlatMapPublisher<S, T> extends Flowable<T> {
19  
20      private final SingleSource<S> source;
21      private final Function<? super S, ? extends Publisher<? extends T>> mapper;
22  
23      public FlowableSingleFlatMapPublisher(SingleSource<S> source,
24              Function<? super S, ? extends Flowable<? extends T>> mapper) {
25          this.source = source;
26          this.mapper = mapper;
27      }
28  
29      @Override
30      protected void subscribeActual(Subscriber<? super T> child) {
31          SingleFlatMapPublisherObserver<S, T> observer = new SingleFlatMapPublisherObserver<S, T>(child, mapper);
32          source.subscribe(observer);
33      }
34  
35      static final class SingleFlatMapPublisherObserver<S, T> extends AtomicLong
36              implements SingleObserver<S>, FlowableSubscriber<T>, Subscription {
37  
38          private static final long serialVersionUID = 7759721921468635667L;
39  
40          private final Subscriber<? super T> child;
41          private final Function<? super S, ? extends Publisher<? extends T>> mapper;
42          private Disposable disposable;
43          private final AtomicReference<Subscription> parent;
44  
45          SingleFlatMapPublisherObserver(Subscriber<? super T> child,
46                  Function<? super S, ? extends Publisher<? extends T>> mapper) {
47              this.child = child;
48              this.mapper = mapper;
49              this.parent = new AtomicReference<>();
50          }
51  
52          @Override
53          public void onSubscribe(Disposable d) {
54              this.disposable = d;
55              child.onSubscribe(this);
56          }
57  
58          @Override
59          public void onSuccess(S value) {
60              Publisher<? extends T> f;
61              try {
62                  f = mapper.apply(value);
63              } catch (Throwable e) {
64                  child.onError(e);
65                  return;
66              }
67              f.subscribe(this);
68          }
69  
70          @Override
71          public void onSubscribe(Subscription s) {
72              SubscriptionHelper.deferredSetOnce(parent, this, s);
73          }
74  
75          @Override
76          public void onNext(T t) {
77              child.onNext(t);
78          }
79  
80          @Override
81          public void onComplete() {
82              child.onComplete();
83          }
84  
85          @Override
86          public void onError(Throwable e) {
87              child.onError(e);
88          }
89  
90          @Override
91          public void request(long n) {
92              SubscriptionHelper.deferredRequest(parent, this, n);
93          }
94  
95          @Override
96          public void cancel() {
97              disposable.dispose();
98              SubscriptionHelper.cancel(parent);
99          }
100     }
101 
102 }
