3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.nio.ByteBuffer;
6   
7   public final class ByteBufferInputStream extends InputStream {
8   
9       private final ByteBuffer bb;
10  
11      public ByteBufferInputStream(ByteBuffer bb) {
12          this.bb = bb;
13      }
14  
15      @Override
16      public int read() throws IOException {
17          if (!bb.hasRemaining()) {
18              return -1;
19          }
20          return bb.get() & 0xFF;
21      }
22  
23      @Override
24      public int read(byte[] bytes, int offset, int length) throws IOException {
25          if (length == 0) {
26              return 0;
27          }
28          int n = Math.min(bb.remaining(), length);
29          if (n == 0) {
30              return -1;
31          }
32          bb.get(bytes, offset, n);
33          return n;
34      }
35  
36      @Override
37      public int available() throws IOException {
38          return bb.remaining();
39      }
40  }
