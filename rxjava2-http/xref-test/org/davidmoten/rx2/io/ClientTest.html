403     @Test
404     public void testBackpressure() throws Exception {
405         System.out.println("testBackpressure");
406         List<Long> requests = new CopyOnWriteArrayList<>();
407         AtomicBoolean cancelled = new AtomicBoolean();
408         Flowable<ByteBuffer> flowable = Flowable.range(1, 1000) //
409                 .map(Serializer.javaIo()::serialize) //
410                 .doOnRequest(n -> requests.add(n)) //
411                 .doOnCancel(() -> cancelled.set(true));
412         Server server = createServerAsync(flowable);
413         try {
414             TestSubscriber<Integer> ts = get(server) //
415                     .<Integer>deserialized() //
416                     .test(0);
417             Thread.sleep(300);
418             ts.assertNoValues() //
419                     .assertNotTerminated();
420             // 16 pre-request and 16 from flatmap
421             assertEquals(Arrays.asList(), requests);
422             ts.requestMore(1);
423             Thread.sleep(300);
424             ts.assertValue(1);
425             assertEquals(Arrays.asList(1L), requests);
426             ts.requestMore(2);
427             Thread.sleep(300);
428             ts.assertValues(1, 2, 3);
429             ts.requestMore(2);
430             Thread.sleep(300);
431             ts.assertValues(1, 2, 3, 4, 5);
432             ts.cancel();
433             Thread.sleep(300);
434             assertTrue(cancelled.get());
435         } finally {
436             // Stop Server
437             server.stop();
438         }
439     }

706     private static BiConsumer<Long, Long> createRequester(int port) {

708 
709             @Override
710             public void accept(Long id, Long request) throws Exception {
