9   import java.security.UnrecoverableKeyException;
10  import java.security.cert.CertificateException;
11  
12  import javax.net.ssl.KeyManager;
13  import javax.net.ssl.KeyManagerFactory;
14  import javax.net.ssl.SSLContext;
15  import javax.net.ssl.TrustManager;
16  
17  public class Ssl {
18  
19      private static final String PASSWORD = "password";
20  
21      public static KeyManager[] getKeyManagers() throws KeyStoreException, IOException, NoSuchAlgorithmException,
22              CertificateException, UnrecoverableKeyException {
23          InputStream keyStore = Ssl.class.getResourceAsStream("/keyStore.jks");
24          KeyStore ks = KeyStore.getInstance("JKS");
25          ks.load(keyStore, PASSWORD.toCharArray());
26          KeyManagerFactory kmf = KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
27          kmf.init(ks, PASSWORD.toCharArray());
28          KeyManager[] keyManagers = kmf.getKeyManagers();
29          return keyManagers;
30      }
31  
32      public static TrustManager[] getTrustManagers() {
33          InputStream trustStore = Ssl.class.getResourceAsStream("/trustStore.jks");
34          TrustManager trustManager = new ExtendedTrustManager(trustStore, PASSWORD.toCharArray(), false);
35          TrustManager[] trustManagers = new TrustManager[] { trustManager };
36          return trustManagers;
37      }
38  
39      public static SSLContext createTlsSslContext(KeyManager[] keyManagers, TrustManager[] trustManagers)
40              throws NoSuchAlgorithmException, KeyManagementException {
41          SSLContext sslContext = SSLContext.getInstance("TLS");
42          sslContext.init(keyManagers, trustManagers, new java.security.SecureRandom());
43          return sslContext;
44      }
45      
46      public static SSLContext createTlsSslContext(TrustManager[] trustManagers)
47              throws NoSuchAlgorithmException, KeyManagementException {
48          return createTlsSslContext(null, trustManagers);
49      }
50  
51  }
