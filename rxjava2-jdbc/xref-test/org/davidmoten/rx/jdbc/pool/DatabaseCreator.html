51      private static Database createDerby(int maxSize, boolean withStoredProcs) {
52          return Database.from(Pools.nonBlocking() //
53                  .connectionProvider(connectionProviderDerby(nextUrlDerby(), withStoredProcs)) //
54                  .maxPoolSize(maxSize) //
55                  .scheduler(Schedulers.from(Executors.newFixedThreadPool(maxSize))) //
56                  .build());
57      }

59      private static ConnectionProvider connectionProviderDerby(String url, boolean withStoredProcs) {
60          Connection c;
61          try {
62              c = DriverManager.getConnection(url);
63              createDatabaseDerby(c);
64              if (withStoredProcs) {
65                  addStoredProcs(c);
66              }
67          } catch (SQLException e) {
68              throw new SQLRuntimeException(e);
69          }
70          return new ConnectionProvider() {
71  
72              @Override
73              public Connection get() {
74                  try {
75                      return DriverManager.getConnection(url);
76                  } catch (SQLException e) {
77                      throw new SQLRuntimeException(e);
78                  }
79              }
80  
81              @Override
82              public void close() {
83                  //
84              }
85          };
86      }

252     public static Database create(int maxSize, boolean big, Scheduler scheduler) {
253         NonBlockingConnectionPool pool = Pools.nonBlocking() //
254                 .connectionProvider(connectionProvider(nextUrl(), big)) //
255                 .maxPoolSize(maxSize) //
256                 .scheduler(scheduler) //
257                 .build();
258         return Database.from(pool, () -> {
259             pool.close();
260             scheduler.shutdown();
261         });
262     }

272     private static ConnectionProvider connectionProvider(String url, boolean big) {
273         return new ConnectionProvider() {
274 
275             private final AtomicBoolean once = new AtomicBoolean(false);
276             private final CountDownLatch latch = new CountDownLatch(1);
277 
278             @Override
279             public Connection get() {
280                 try {
281                     if (once.compareAndSet(false, true)) {
282                         try {
283                             Connection c = DriverManager.getConnection(url);
284                             createDatabase(c, big);
285                             c.setAutoCommit(false);
286                             return c;
287                         } finally {
288                             latch.countDown();
289                         }
290                     } else {
291                         if (latch.await(30, TimeUnit.SECONDS)) {
292                             return DriverManager.getConnection(url);
293                         } else {
294                             throw new TimeoutException("big database timed out on creation");
295                         }
296                     }
297                 } catch (SQLException | InterruptedException | TimeoutException e) {
298                     throw new SQLRuntimeException(e);
299                 }
300             }
301 
302             @Override
303             public void close() {
304                 //
305             }
306         };
307     }

317     private static void createDatabase(Connection c, boolean big) {
318         try {
319             c.setAutoCommit(true);
320             c.prepareStatement(
321                     "create table person (name varchar(50) primary key, score int not null, date_of_birth date, registered timestamp)")
322                     .execute();
323             if (big) {
324                 List<String> lines = IOUtils.readLines(DatabaseCreator.class.getResourceAsStream("/big.txt"),
325                         StandardCharsets.UTF_8);
326                 lines.stream().map(line -> line.split("\t")).forEach(items -> {
327                     try {
328                         c.prepareStatement("insert into person(name,score) values('" + items[0] + "',"
329                                 + Integer.parseInt(items[1]) + ")").execute();
330                     } catch (SQLException e) {
331                         throw new SQLRuntimeException(e);
332                     }
333                 });
334             } else {
335                 exec(c, "insert into person(name,score,registered) values('FRED',21, {ts '2015-09-17 18:47:52.69Z'})");
336                 exec(c, "insert into person(name,score) values('JOSEPH',34)");
337                 exec(c, "insert into person(name,score) values('MARMADUKE',25)");
338             }
339 
340             exec(c, "create table person_clob (name varchar(50) not null,  document clob)");
341             exec(c, "create table person_blob (name varchar(50) not null, document blob)");
342 
343             exec(c, "create table address (address_id int primary key, full_address varchar(255) not null)");
344             exec(c, "insert into address(address_id, full_address) values(1,'57 Something St, El Barrio, Big Place')");
345             exec(c, "create table note(id bigint auto_increment primary key, text varchar(255))");
346         } catch (SQLException e) {
347             throw new SQLRuntimeException(e);
348         } catch (IOException e) {
349             throw new RuntimeException(e);
350         }
351     }
