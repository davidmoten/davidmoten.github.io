48      /**
49       * Returns an {@link Flowable} that uses given given Flowable to push modified
50       * events to an Flowable that reads and reports new sequences of bytes to a
51       * subscriber. The NIO {@link WatchService} MODIFY and OVERFLOW events are
52       * sampled according to <code>sampleTimeMs</code> so that lots of discrete
53       * activity on a file (for example a log file with very frequent entries) does
54       * not prompt an inordinate number of file reads to pick up changes. File create
55       * events are not sampled and are always passed through.
56       * 
57       * @param file
58       *            the file to tail
59       * @param startPosition
60       *            start tailing file at position in bytes
61       * @param pollingIntervalMs
62       *            polling time in millis for MODIFY and OVERFLOW events and half of
63       *            sample time for overflow
64       * @param chunkSize
65       *            max array size of each element emitted by the Flowable. Is also
66       *            used as the buffer size for reading from the file. Try
67       *            {@link FileFlowable#DEFAULT_MAX_BYTES_PER_EMISSION} if you don't
68       *            know what to put here.
69       * @param events
70       *            trigger a check for file changes. Use
71       *            {@link Flowable#interval(long, TimeUnit)} for example.
72       * @return Flowable of byte arrays
73       */
74      private static Flowable<byte[]> tailBytes(File file, long startPosition, long pollingIntervalMs, int chunkSize,
75              Flowable<?> events) {
76          Preconditions.checkNotNull(file);
77          return sampleModifyOrOverflowEventsOnly(events, pollingIntervalMs * 2)
78                  // tail file triggered by events
79                  .compose(x -> eventsToBytes(events, file, startPosition, chunkSize));
80      }

82      /**
83       * Returns an {@link Flowable} of String that uses the given events stream to
84       * trigger checks on file change so that new lines can be read and emitted.
85       * 
86       * @param file
87       *            the file to tail, cannot be null
88       * @param startPosition
89       *            start tailing file at position in bytes
90       * @param chunkSize
91       *            max array size of each element emitted by the Flowable. Is also
92       *            used as the buffer size for reading from the file. Try
93       *            {@link FileFlowable#DEFAULT_MAX_BYTES_PER_EMISSION} if you don't
94       *            know what to put here.
95       * @param charset
96       *            the character set to use to decode the bytes to a string
97       * @param events
98       *            trigger a check for file changes. Use
99       *            {@link Flowable#interval(long, TimeUnit)} for example.
100      * @return Flowable of strings
101      */
102     private static Flowable<String> tailLines(File file, long startPosition, int chunkSize, Charset charset,
103             Flowable<?> events) {
104         Preconditions.checkNotNull(file);
105         Preconditions.checkNotNull(charset);
106         Preconditions.checkNotNull(events);
107         return toLines(events.compose(x -> eventsToBytes(x, file, startPosition, chunkSize)), charset);
108     }

110     private static Flowable<WatchEvent<?>> events(WatchService watchService, Scheduler scheduler, long intervalMs) {
111         Preconditions.checkNotNull(watchService, "watchService cannot be null");
112         Preconditions.checkNotNull(scheduler, "scheduler cannot be null");
113         Preconditions.checkArgument(intervalMs > 0, "intervalMs must be positive");
114         return Flowable.interval(intervalMs, TimeUnit.MILLISECONDS, scheduler) //
115                 .flatMap(x -> {
116                     WatchKey key = watchService.poll();
117                     if (key != null) {
118                         Flowable<WatchEvent<?>> r = Flowable.fromIterable(key.pollEvents());
119                         key.reset();
120                         return r;
121                     } else {
122                         return Flowable.empty();
123                     }
124                 });
125     }

127     /**
128      * If file does not exist at subscribe time then is assumed to not be a
129      * directory. If the file is not a directory (bearing in mind the aforesaid
130      * assumption) then a {@link WatchService} is set up on its parent and
131      * {@link WatchEvent}s of the given kinds are filtered to concern the file in
132      * question. If the file is a directory then a {@link WatchService} is set up on
133      * the directory and all events are passed through of the given kinds.
134      * 
135      * @param file
136      *            file to generate watch events from
137      * @param onWatchStarted
138      *            called when WatchService is created
139      * @param kinds
140      *            kinds of watch events to register for
141      * @return Flowable of watch events
142      */
143     private static Flowable<WatchEvent<?>> events(File file, Scheduler scheduler, long pollingIntervalMs,
144             List<Kind<?>> kinds, List<Modifier> modifiers) {
145         return Flowable.using(() -> watchService(file, kinds, modifiers), //
146                 ws -> events(ws, scheduler, pollingIntervalMs)
147                         // restrict to events related to the file
148                         .filter(onlyRelatedTo(file)), //
149                 ws -> ws.close(), true);
150     }

152     /**
153      * Creates a {@link WatchService} on subscribe for the given file and event
154      * kinds.
155      * 
156      * @param file
157      *            the file to watch
158      * @param kinds
159      *            event kinds to watch for
160      * @return Flowable of watch events
161      * @throws IOException
162      */
163     private static WatchService watchService(File file, List<Kind<?>> kinds, List<Modifier> modifiers)
164             throws IOException {
165         final Path path = getBasePath(file);
166         WatchService watchService = path.getFileSystem().newWatchService();
167         path.register(watchService, kinds.toArray(new Kind<?>[] {}), modifiers.toArray(new Modifier[] {}));
168         return watchService;
169     }

171     /**
172      * Returns true if and only if the path corresponding to a WatchEvent represents
173      * the given file. This will be the case for Create, Modify, Delete events.
174      * 
175      * @param file
176      *            the file to restrict events to
177      * @return predicate
178      */
179     private final static Predicate<WatchEvent<?>> onlyRelatedTo(final File file) {
180         return new Predicate<WatchEvent<?>>() {
181 
182             @Override
183             public boolean test(WatchEvent<?> event) {
184 
185                 final boolean ok;
186                 if (file.isDirectory())
187                     ok = true;
188                 else if (StandardWatchEventKinds.OVERFLOW.equals(event.kind()))
189                     ok = true;
190                 else {
191                     Object context = event.context();
192                     if (context != null && context instanceof Path) {
193                         Path p = (Path) context;
194                         Path basePath = getBasePath(file);
195                         File pFile = new File(basePath.toFile(), p.toString());
196                         ok = pFile.getAbsolutePath().equals(file.getAbsolutePath());
197                     } else
198                         ok = false;
199                 }
200                 return ok;
201             }
202         };
203     }

209     private static Flowable<Object> sampleModifyOrOverflowEventsOnly(Flowable<?> events, final long sampleTimeMs) {
210         return events
211                 // group by true if is modify or overflow, false otherwise
212                 .groupBy(IS_MODIFY_OR_OVERFLOW)
213                 // only sample if is modify or overflow
214                 .flatMap(sampleIfTrue(sampleTimeMs));
215     }
216 
217     private static Function<GroupedFlowable<Boolean, ?>, Flowable<?>> sampleIfTrue(final long sampleTimeMs) {
218         return group -> { // if is modify or overflow WatchEvent
219             if (group.getKey())
220                 return group.sample(sampleTimeMs, TimeUnit.MILLISECONDS);
221             else
222                 return group;
223         };
224     }
225 
226     private static Function<Object, Boolean> IS_MODIFY_OR_OVERFLOW = event -> {
227         if (event instanceof WatchEvent) {
228             WatchEvent<?> w = (WatchEvent<?>) event;
229             String kind = w.kind().name();
230             if (kind.equals(StandardWatchEventKinds.ENTRY_MODIFY.name())
231                     || kind.equals(StandardWatchEventKinds.OVERFLOW.name())) {
232                 return true;
233             } else
234                 return false;
235         } else
236             return false;
237     };

297         public Flowable<WatchEvent<?>> build() {
298             List<Kind<?>> kindsCopy = new ArrayList<>(kinds);
299             if (kindsCopy.isEmpty()) {
300                 kindsCopy.add(StandardWatchEventKinds.ENTRY_CREATE);
301                 kindsCopy.add(StandardWatchEventKinds.ENTRY_DELETE);
302                 kindsCopy.add(StandardWatchEventKinds.ENTRY_MODIFY);
303                 kindsCopy.add(StandardWatchEventKinds.OVERFLOW);
304             }
305             return Flowable.using( //
306                     () -> watchService(file, kindsCopy, modifiers), //
307                     ws -> Files.events(ws, scheduler.orElse(Schedulers.io()), pollIntervalUnit.toMillis(pollInterval)), //
308                     ws -> ws.close(), //
309                     true);
310         }

503     private static final class State {
504         long position;
505     }

507     private static Flowable<byte[]> eventsToBytes(Flowable<?> events, File file, long startPosition, int chunkSize) {
508         return Flowable.defer(() -> {
509             State state = new State();
510             state.position = startPosition;
511             return events.flatMap(event -> eventToBytes(event, file, state, chunkSize));
512         });
513     }
514 
515     private static Flowable<byte[]> eventToBytes(Object event, File file, State state, int chunkSize) {
516         if (event instanceof WatchEvent) {
517             WatchEvent<?> w = (WatchEvent<?>) event;
518             String kind = w.kind().name();
519             if (kind.equals(StandardWatchEventKinds.ENTRY_CREATE.name())) {
520                 // if file has just been created then start from the start of the new file
521                 state.position = 0;
522             } else if (kind.equals(StandardWatchEventKinds.ENTRY_DELETE.name())) {
523                 return Flowable.error(new IOException("file has been deleted"));
524             }
525             // we hope that ENTRY_CREATE and ENTRY_DELETE events never get wrapped up into
526             // ENTRY_OVERFLOW!
527         }
528         long length = file.length();
529         if (length > state.position) {
530             // apply using method to ensure fis is closed on
531             // termination or unsubscription
532             return Flowable.using( //
533                     () -> new FileInputStream(file), //
534                     fis -> {
535                         fis.skip(state.position);
536                         return Bytes.from(fis, chunkSize) //
537                                 .doOnNext(x -> state.position += x.length);
538                     }, //
539                     fis -> fis.close(), //
540                     true);
541         } else {
542             return Flowable.empty();
543         }
544     }

546     private final static Path getBasePath(final File file) {
547         final Path path;
548         if (file.exists() && file.isDirectory())
549             path = Paths.get(file.toURI());
550         else
551             path = Paths.get(file.getParentFile().toURI());
552         return path;
553     }
