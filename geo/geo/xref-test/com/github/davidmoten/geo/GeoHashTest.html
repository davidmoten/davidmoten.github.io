304                 HARTFORD_LAT, HARTFORD_LON, 4).getHashes();
305 
306         // check schenectady hash
307         assertEquals("dre7", encodeHash(SCHENECTADY_LAT, SCHENECTADY_LON, 4));
308         // check hartford hash
309         assertEquals("drkq", encodeHash(HARTFORD_LAT, HARTFORD_LON, 4));
310         assertEquals("drs7", encodeHash(SCHENECTADY_LAT, HARTFORD_LON, 4));
311         assertEquals("dr7q", encodeHash(HARTFORD_LAT, SCHENECTADY_LON, 4));
312 
313         // check neighbours
314         assertEquals("drs", adjacentHash("dre", Direction.RIGHT));
315         assertEquals("dr7", adjacentHash("dre", Direction.BOTTOM));
316         assertEquals("drk", adjacentHash("drs", Direction.BOTTOM));
317 
318         System.out.println(gridAsString("dreb", 5, hashes));
319 
320         // check corners are in
321         assertTrue(hashes.contains("dre7"));
322         assertTrue(hashes.contains("drkq"));
323         assertTrue(hashes.contains("drs7"));
324         assertTrue(hashes.contains("dr7q"));
325         String expected = "dre7,dree,dreg,drs5,drs7,dre6,dred,dref,drs4,drs6,dre3,dre9,drec,drs1,drs3,dre2,dre8,dreb,drs0,drs2,dr7r,dr7x,dr7z,drkp,drkr,dr7q,dr7w,dr7y,drkn,drkq";
326         TreeSet<String> ex = Sets
327                 .newTreeSet(Arrays.asList(expected.split(",")));
328         System.out.println(ex);
329         System.out.println(hashes);
330         assertEquals(ex, hashes);
