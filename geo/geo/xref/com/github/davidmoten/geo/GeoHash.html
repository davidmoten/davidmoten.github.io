126     /**
127      * Returns the adjacent hash in given {@link Direction}. Based on
128      * https://github.com/davetroy/geohash-js/blob/master/geohash.js. This
129      * method is an improvement on the original js method because it works at
130      * borders too (at the poles and the -180,180 longitude boundaries).
131      * 
132      * @param hash
133      *            string hash relative to which the adjacent is returned
134      * @param direction
135      *            direction relative to {@code hash} for which the adjacent is
136      *            returned
137      * @return hash of adjacent hash
138      */
139     public static String adjacentHash(String hash, Direction direction) {
140         checkHash(hash);
141         Preconditions.checkArgument(hash.length() > 0,
142                 "adjacent has no meaning for a zero length hash that covers the whole world");
143 
144         String adjacentHashAtBorder = adjacentHashAtBorder(hash, direction);
145         if (adjacentHashAtBorder != null)
146             return adjacentHashAtBorder;
147 
148         String source = hash.toLowerCase(Locale.ENGLISH);
149         char lastChar = source.charAt(source.length() - 1);
150         Parity parity = (source.length() % 2 == 0) ? Parity.EVEN : Parity.ODD;
151         String base = source.substring(0, source.length() - 1);
152         if (BORDERS.get(direction).get(parity).indexOf(lastChar) != -1)
153             base = adjacentHash(base, direction);
154         return base + BASE32.charAt(NEIGHBOURS.get(direction).get(parity).indexOf(lastChar));
155     }

157     private static String adjacentHashAtBorder(String hash, Direction direction) {
158         // check if hash is on edge and direction would push us over the edge
159         // if so, wrap round to the other limit for longitude
160         // or if at latitude boundary (a pole) then spin longitude around 180
161         // degrees.
162         LatLong centre = decodeHash(hash);
163 
164         // if rightmost hash
165         if (Direction.RIGHT.equals(direction)) {
166             if (Math.abs(centre.getLon() + widthDegrees(hash.length()) / 2 - 180) < PRECISION) {
167                 return encodeHash(centre.getLat(), -180, hash.length());
168             }
169         }
170         // if leftmost hash
171         else if (Direction.LEFT.equals(direction)) {
172             if (Math.abs(centre.getLon() - widthDegrees(hash.length()) / 2 + 180) < PRECISION) {
173                 return encodeHash(centre.getLat(), 180, hash.length());
174             }
175         }
176         // if topmost hash
177         else if (Direction.TOP.equals(direction)) {
178             if (Math.abs(centre.getLat() + widthDegrees(hash.length()) / 2 - 90) < PRECISION) {
179                 return encodeHash(centre.getLat(), centre.getLon() + 180, hash.length());
180             }
181         }
182         // if bottommost hash
183         else {
184             if (Math.abs(centre.getLat() - widthDegrees(hash.length()) / 2 + 90) < PRECISION) {
185                 return encodeHash(centre.getLat(), centre.getLon() + 180, hash.length());
186             }
187         }
188 
189         return null;
190     }

247     /**
248      * Returns the adjacent hash N steps in the given {@link Direction}. A
249      * negative N will use the opposite {@link Direction}.
250      * 
251      * @param hash
252      *            origin hash
253      * @param direction
254      *            to desired hash
255      * @param steps
256      *            number of hashes distance to desired hash
257      * @return hash at position in direction a number of hashes away (steps)
258      */
259     public static String adjacentHash(String hash, Direction direction, int steps) {
260         if (steps < 0)
261             return adjacentHash(hash, direction.opposite(), Math.abs(steps));
262         else {
263             String h = hash;
264             for (int i = 0; i < steps; i++)
265                 h = adjacentHash(h, direction);
266             return h;
267         }
268     }

272      * left,right,top,bottom,left-top,left-bottom,right-top,right-bottom.
273      * 
274      * @param hash
275      *            source
276      * @return a list of neighbour hashes

278     public static List<String> neighbours(String hash) {
279         List<String> list = new ArrayList<String>();
280         String left = adjacentHash(hash, Direction.LEFT);
281         String right = adjacentHash(hash, Direction.RIGHT);
282         list.add(left);
283         list.add(right);
284         list.add(adjacentHash(hash, Direction.TOP));
285         list.add(adjacentHash(hash, Direction.BOTTOM));
286         list.add(adjacentHash(left, Direction.TOP));
287         list.add(adjacentHash(left, Direction.BOTTOM));
288         list.add(adjacentHash(right, Direction.TOP));
289         list.add(adjacentHash(right, Direction.BOTTOM));
290         return list;
291     }

333      * Returns a geohash of given length for the given WGS84 point
334      * (latitude,longitude). If latitude is not between -90 and 90 throws an
335      * {@link IllegalArgumentException}.
336      * 
337      * @param latitude
338      *            in decimal degrees (WGS84)

345     // Translated to java from:
346     // geohash.js
347     // Geohash library for Javascript
348     // (c) 2008 David Troy

350     public static String encodeHash(double latitude, double longitude, int length) {
351         Preconditions.checkArgument(length > 0 && length <=12, "length must be between 1 and 12");
352         Preconditions.checkArgument(latitude >= -90 && latitude <= 90,
353                 "latitude must be between -90 and 90 inclusive");
354         return fromLongToString(encodeHashToLong(latitude, to180(longitude), length));
355     }

357     /**
358      * Takes a hash represented as a long and returns it as a string.
359      *
360      * @param hash
361      *            the hash, with the length encoded in the 4 least significant
362      *            bits
363      * @return the string encoded geohash
364      */
365     static String fromLongToString(long hash) {
366         int length = (int) (hash & 0xf);
367         if (length > 12 || length < 1)
368             throw new IllegalArgumentException("invalid long geohash " + hash);
369         char[] geohash = new char[length];
370         for (int pos = 0; pos < length; pos++) {
371             geohash[pos] = BASE32.charAt((int) (hash >>> 59));
372             hash <<= 5;
373         }
374         return new String(geohash);
375     }

377     static long encodeHashToLong(double latitude, double longitude, int length) {
378         boolean isEven = true;
379         double minLat = -90.0, maxLat = 90;
380         double minLon = -180.0, maxLon = 180.0;
381         long bit = 0x8000000000000000L;
382         long g = 0;
383 
384         long target = 0x8000000000000000L >>> (5 * length);
385         while (bit != target) {
386             if (isEven) {
387                 double mid = (minLon + maxLon) / 2;
388                 if (longitude >= mid) {
389                     g |= bit;
390                     minLon = mid;
391                 } else
392                     maxLon = mid;
393             } else {
394                 double mid = (minLat + maxLat) / 2;
395                 if (latitude >= mid) {
396                     g |= bit;
397                     minLat = mid;
398                 } else
399                     maxLat = mid;
400             }
401 
402             isEven = !isEven;
403             bit >>>= 1;
404         }
405         return g | length;
406     }

408     /**
409      * Returns a latitude,longitude pair as the centre of the given geohash.
410      * Latitude will be between -90 and 90 and longitude between -180 and 180.
411      * 
412      * @param geohash
413      *            hash to decode
414      * @return lat long point
415      */
416     // Translated to java from:
417     // geohash.js
418     // Geohash library for Javascript
419     // (c) 2008 David Troy
420     // Distributed under the MIT License
421     public static LatLong decodeHash(String geohash) {
422         Preconditions.checkNotNull(geohash, "geohash cannot be null");
423         boolean isEven = true;
424         double[] lat = new double[2];
425         double[] lon = new double[2];
426         lat[0] = -90.0;
427         lat[1] = 90.0;
428         lon[0] = -180.0;
429         lon[1] = 180.0;
430 
431         for (int i = 0; i < geohash.length(); i++) {
432             char c = geohash.charAt(i);
433             int cd = BASE32.indexOf(c);
434             for (int j = 0; j < 5; j++) {
435                 int mask = BITS[j];
436                 if (isEven) {
437                     refineInterval(lon, cd, mask);
438                 } else {
439                     refineInterval(lat, cd, mask);
440                 }
441                 isEven = !isEven;
442             }
443         }
444         double resultLat = (lat[0] + lat[1]) / 2;
445         double resultLon = (lon[0] + lon[1]) / 2;
446 
447         return new LatLong(resultLat, resultLon);
448     }

450     /**
451      * Refines interval by a factor or 2 in either the 0 or 1 ordinate.
452      * 
453      * @param interval
454      *            two entry array of double values
455      * @param cd
456      *            used with mask
457      * @param mask
458      *            used with cd
459      */
460     private static void refineInterval(double[] interval, int cd, int mask) {
461         if ((cd & mask) != 0)
462             interval[0] = (interval[0] + interval[1]) / 2;
463         else
464             interval[1] = (interval[0] + interval[1]) / 2;
465     }

467     /**
468      * Returns the maximum length of hash that covers the bounding box. If no
469      * hash can enclose the bounding box then 0 is returned.
470      * 
471      * @param topLeftLat
472      *            latitude of top left point (north west)
473      * @param topLeftLon
474      *            longitude of top left point (north west)
475      * @param bottomRightLat
476      *            latitude of bottom right point (south east)
477      * @param bottomRightLon
478      *            longitude of bottom right point (south east)
479      * @return length of the hash
480      */
481     public static int hashLengthToCoverBoundingBox(double topLeftLat, double topLeftLon,
482             double bottomRightLat, double bottomRightLon) {
483         boolean isEven = true;
484         double minLat = -90.0, maxLat = 90;
485         double minLon = -180.0, maxLon = 180.0;
486 
487         for (int bits = 0; bits < MAX_HASH_LENGTH * 5; bits++) {
488             if (isEven) {
489                 double mid = (minLon + maxLon) / 2;
490                 if (topLeftLon >= mid) {
491                     if (bottomRightLon < mid)
492                         return bits / 5;
493                     minLon = mid;
494                 } else {
495                     if (bottomRightLon >= mid)
496                         return bits / 5;
497                     maxLon = mid;
498                 }
499             } else {
500                 double mid = (minLat + maxLat) / 2;
501                 if (topLeftLat >= mid) {
502                     if (bottomRightLat < mid)
503                         return bits / 5;
504                     minLat = mid;
505                 } else {
506                     if (bottomRightLat >= mid)
507                         return bits / 5;
508                     maxLat = mid;
509                 }
510             }
511 
512             isEven = !isEven;
513         }
514         return MAX_HASH_LENGTH;
515     }

518      * Returns true if and only if the bounding box corresponding to the hash
519      * contains the given lat and long.
520      * 
521      * @param hash
522      *            hash to test containment in
523      * @param lat
524      *            latitude
525      * @param lon
526      *            longitude
527      * @return true if and only if the hash contains the given lat and long

529     public static boolean hashContains(String hash, double lat, double lon) {
530         LatLong centre = decodeHash(hash);
531         return Math.abs(centre.getLat() - lat) <= heightDegrees(hash.length()) / 2
532                 && Math.abs(to180(centre.getLon() - lon)) <= widthDegrees(hash.length()) / 2;
533     }

556     /**
557      * Returns the hashes that are required to cover the given bounding box. The
558      * maximum length of hash is selected that satisfies the number of hashes
559      * returned is less than <code>maxHashes</code>. Returns null if hashes
560      * cannot be found satisfying that condition. Maximum hash length returned
561      * will be {@link GeoHash}.MAX_HASH_LENGTH.
562      * 
563      * @param topLeftLat
564      *            latitude of top left point (north west)
565      * @param topLeftLon
566      *            longitude of top left point (north west)
567      * @param bottomRightLat
568      *            latitude of bottom right point (south east)
569      * @param bottomRightLon
570      *            longitude of bottom right point (south east)
571      * @param maxHashes
572      *            maximum number of hashes to use to cover the box
573      * @return coverage
574      */
575     public static Coverage coverBoundingBoxMaxHashes(double topLeftLat, final double topLeftLon,
576             final double bottomRightLat, final double bottomRightLon, int maxHashes) {
577         CoverageLongs coverage = null;
578         int startLength = hashLengthToCoverBoundingBox(topLeftLat, topLeftLon, bottomRightLat,
579                 bottomRightLon);
580         if (startLength == 0)
581             startLength = 1;
582         for (int length = startLength; length <= MAX_HASH_LENGTH; length++) {
583             CoverageLongs c = coverBoundingBoxLongs(topLeftLat, topLeftLon, bottomRightLat,
584                     bottomRightLon, length);
585             if (c.getCount() > maxHashes)
586                 return coverage == null ? null : new Coverage(coverage);
587             else
588                 coverage = c;
589         }
590         // note coverage can never be null
591         return new Coverage(coverage);
592     }

617     private static class LongSet {
618         int count = 0;
619         private int cap = 16;
620         long[] array = new long[cap];
621 
622         void add(long l) {
623             for (int i = 0; i < count; i++)
624                 if (array[i] == l)
625                     return;
626             if (count == cap) {
627                 long[] newArray = new long[cap *= 2];
628                 System.arraycopy(array, 0, newArray, 0, count);
629                 array = newArray;
630             }
631             array[count++] = l;
632         }
633     }

635     static CoverageLongs coverBoundingBoxLongs(double topLeftLat, double topLeftLon,
636             double bottomRightLat, double bottomRightLon, final int length) {
637         Preconditions.checkArgument(topLeftLat >= bottomRightLat,
638                 "topLeftLat must be >= bottomRighLat");
639         Preconditions.checkArgument(length > 0, "length must be greater than zero");
640         final double actualWidthDegreesPerHash = widthDegrees(length);
641         final double actualHeightDegreesPerHash = heightDegrees(length);
642 
643         LongSet hashes = new LongSet();
644 
645         double diff = bottomRightLon - topLeftLon;
646         if (diff < 0) {
647             // case where bottomRightLon cross the antimeridian
648             bottomRightLon += 360;
649             diff = bottomRightLon - topLeftLon;
650         } else if (diff > 360) {
651             // case where this bounding box displays more than one copy of the world
652             topLeftLon = -180;
653             bottomRightLon = 180;
654             diff = 360;
655         }
656 
657         for (double lat = bottomRightLat; lat <= topLeftLat; lat += actualHeightDegreesPerHash) {
658             for (double lon = topLeftLon; lon <= bottomRightLon; lon += actualWidthDegreesPerHash) {
659                 hashes.add(encodeHashToLong(lat, to180(lon), length));
660             }
661         }
662         // ensure have the borders covered
663         for (double lat = bottomRightLat; lat <= topLeftLat; lat += actualHeightDegreesPerHash) {
664             hashes.add(encodeHashToLong(lat, to180(bottomRightLon), length));
665         }
666         for (double lon = topLeftLon; lon <= bottomRightLon; lon += actualWidthDegreesPerHash) {
667             hashes.add(encodeHashToLong(topLeftLat, to180(lon), length));
668         }
669         // ensure that the topRight corner is covered
670         hashes.add(encodeHashToLong(topLeftLat, to180(bottomRightLon), length));
671 
672         double areaDegrees = diff * (topLeftLat - bottomRightLat);
673         double coverageAreaDegrees = hashes.count * widthDegrees(length) * heightDegrees(length);
674         double ratio = coverageAreaDegrees / areaDegrees;
675         return new CoverageLongs(hashes.array, hashes.count, ratio);
676     }

815     /**
816      * <p>Returns a String of lines of hashes to represent the relative positions
817      * of hashes on a map. Highlighted hashes are displayed in upper case. For
818      * example, gridToString("dr",-1,-1,1,1,Sets.newHashSet("f2","f8")) returns:
819      * </p>
820      * 
821      * <pre>
822      * f0 F2 F8 
823      * dp dr dx 
824      * dn dq dw
825      * </pre>
826      * 
827      * @param hash reference hash
828      * @param fromRight
829      *            top left of the grid in hashes to the right (can be negative).
830      * @param fromBottom
831      *            top left of the grid in hashes to the bottom (can be
832      *            negative).
833      * @param toRight
834      *            bottom righth of the grid in hashes to the bottom (can be
835      *            negative).
836      * @param toBottom
837      *            bottom right of the grid in hashes to the bottom (can be
838      *            negative).
839      * @param highlightThese hashes to highlight
840      * @return String representation of grid
841      */
842     public static String gridAsString(String hash, int fromRight, int fromBottom, int toRight,
843             int toBottom, Set<String> highlightThese) {
844         StringBuilder s = new StringBuilder();
845         for (int bottom = fromBottom; bottom <= toBottom; bottom++) {
846             for (int right = fromRight; right <= toRight; right++) {
847                 String h = adjacentHash(hash, Direction.RIGHT, right);
848                 h = adjacentHash(h, Direction.BOTTOM, bottom);
849                 if (highlightThese.contains(h))
850                     h = h.toUpperCase(Locale.ENGLISH);
851                 s.append(h).append(" ");
852             }
853             s.append("\n");
854         }
855         return s.toString();
856     }

858     /**
859      * Converts an angle in degrees to range -180< x <= 180.
860      * 
861      * @param d angle in degrees
862      * @return converted angle in degrees
863      */
864     private static double to180(double d) {
865         if (d < 0)
866             return -to180(Math.abs(d));
867         else {
868             if (d > 180) {
869                 long n = Math.round(Math.floor((d + 180) / 360.0));
870                 return d - n * 360;
871             } else
872                 return d;
873         }
874     }
