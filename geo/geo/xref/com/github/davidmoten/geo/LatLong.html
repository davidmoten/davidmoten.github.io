1   package com.github.davidmoten.geo;
2   
3   /**
4    * A lat, long pair (WGS84). Immutable.
5    * 
6    */
7   public class LatLong {
8   
9       private final double lat;
10      private final double lon;
11  
12      /**
13       * Constructor.
14       * 
15       * @param lat latitude in decimal degrees
16       * @param lon longitude in decimal degrees
17       */
18      public LatLong(double lat, double lon) {
19          this.lat = lat;
20          this.lon = lon;
21      }
22  
23      /**
24       * Returns the latitude in decimal degrees.
25       * 
26       * @return latitude in decimal degrees
27       */
28      public double getLat() {
29          return lat;
30      }
31  
32      /**
33       * Returns the longitude in decimal degrees.
34       * 
35       * @return longitude in decimal degrees
36       */
37      public double getLon() {
38          return lon;
39      }
40  
41      /**
42       * Returns a new {@link LatLong} object with lat, lon increased by deltaLat,
43       * deltaLon.
44       * 
45       * @param deltaLat change in latitude
46       * @param deltaLon change in longitude
47       * @return latitude and longitude
48       */
49      public LatLong add(double deltaLat, double deltaLon) {
50          return new LatLong(lat + deltaLat, lon + deltaLon);
51      }
52  
53      @Override
54      public String toString() {
55          StringBuilder builder = new StringBuilder();
56          builder.append("LatLong [lat=");
57          builder.append(lat);
58          builder.append(", lon=");
59          builder.append(lon);
60          builder.append("]");
61          return builder.toString();
62      }
63  
64  }
