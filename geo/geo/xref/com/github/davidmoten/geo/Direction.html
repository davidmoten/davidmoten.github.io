3   /**
4    * Directions on a WGS84 projection. top = north, bottom = south, left = west,
5    * right = east.
6    * 
7    * @author dave
8    * 
9    */
10  public enum Direction {
11      BOTTOM, TOP, LEFT, RIGHT;
12  
13      /**
14       * Returns the opposite direction. For example LEFT.opposite() == RIGHT.
15       * 
16       * @return the opposite direction
17       */
18      public Direction opposite() {
19          if (this == BOTTOM)
20              return TOP;
21          else if (this == TOP)
22              return BOTTOM;
23          else if (this == LEFT)
24              return RIGHT;
25          else
26              return LEFT;
27      }
28  }
