3   import java.sql.Connection;
4   import java.sql.SQLException;
5   import java.util.concurrent.atomic.AtomicInteger;
6   
7   public class DatabaseCreator {
8   
9       private static AtomicInteger dbNumber = new AtomicInteger();
10  
11      public static String nextUrl() {
12          return "jdbc:h2:mem:test" + dbNumber.incrementAndGet() + ";DB_CLOSE_DELAY=-1";
13      }
14  
15      public static ConnectionProvider connectionProvider() {
16          return new ConnectionProviderFromUrl(nextUrl());
17      }
18  
19      public static Database db() {
20          ConnectionProvider cp = connectionProvider();
21          Connection con = cp.get();
22          createDatabase(con);
23          try {
24              con.close();
25          } catch (SQLException e) {
26              throw new RuntimeException(e);
27          }
28          return new Database(cp);
29      }
30  
31      public static Database createDatabase(ConnectionProvider cp) {
32          Database db = Database.from(cp);
33          Connection con = cp.get();
34          createDatabase(con);
35          try {
36              con.close();
37          } catch (SQLException e) {
38              throw new RuntimeException(e);
39          }
40          return db;
41      }
42  
43      public static void createDatabase(Connection c) {
44          try {
45              c.setAutoCommit(true);
46              c.prepareStatement(
47                      "create table person (name varchar(50) primary key, score int not null,dob date, registered timestamp)")
48                      .execute();
49              c.prepareStatement("insert into person(name,score) values('FRED',21)").execute();
50              c.prepareStatement("insert into person(name,score) values('JOSEPH',34)").execute();
51              c.prepareStatement("insert into person(name,score) values('MARMADUKE',25)").execute();
52  
53              c.prepareStatement(
54                      "create table person_clob (name varchar(50) not null,  document clob)")
55                      .execute();
56              c.prepareStatement(
57                      "create table person_blob (name varchar(50) not null, document blob)")
58                      .execute();
59          } catch (SQLException e) {
60              throw new RuntimeException(e);
61          }
62      }

64  }
