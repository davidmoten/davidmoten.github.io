12  public class NamedParametersTest {

14      @Test
15      public void testSelect() {
16          JdbcQuery r = NamedParameters
17                  .parse("select a, b from tbl where a.name=:name and b.name=:name and c.description = :description");
18          assertEquals("select a, b from tbl where a.name=? and b.name=? and c.description = ?",
19                  r.sql());
20          assertEquals(Arrays.asList("name", "name", "description"), r.names());
21      }
22  
23      @Test
24      public void testSelectWithOneNamedParameterAndColonNameInQuotes() {
25          JdbcQuery r = NamedParameters
26                  .parse("select a, b from tbl where a.name=:name and b.name=':name'");
27          assertEquals("select a, b from tbl where a.name=? and b.name=':name'", r.sql());
28          assertEquals(Arrays.asList("name"), r.names());
29      }
30  
31      @Test
32      public void testSelectWithNoNamedParameters() {
33          JdbcQuery r = NamedParameters.parse("select a, b from tbl");
34          assertEquals("select a, b from tbl", r.sql());
35          assertTrue(r.names().isEmpty());
36      }
37  
38      @Test
39      public void testNamedParametersAllMissingParametersShouldDoNothing() {
40          DatabaseCreator.db().select("select name from person where name = :name").count()
41                  .subscribe();
42      }
