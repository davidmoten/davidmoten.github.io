1   package com.github.davidmoten.rx;
2   
3   import rx.Observable;
4   import rx.Observable.Operator;
5   import rx.Subscriber;
6   import rx.functions.Func1;
7   import rx.observers.Subscribers;
8   
9   /**
10   * Converts an Operation (a function converting one Observable into another)
11   * into an {@link Operator}.
12   * 
13   * @param <R>
14   *            to type
15   * @param <T>
16   *            from type
17   */
18  public final class OperationOperator<R, T> implements Operator<R, T> {
19  
20  	public static <R, T> Operator<R, T> toOperator(
21  			Func1<Observable<T>, Observable<R>> operation) {
22  		return new OperationOperator<R, T>(operation);
23  	}
24  
25  	/**
26  	 * The operation to convert.
27  	 */
28  	private final Func1<Observable<T>, Observable<R>> operation;
29  
30  	/**
31  	 * Constructor.
32  	 * 
33  	 * @param operation
34  	 *            to be converted into {@link Operator}
35  	 */
36  	public OperationOperator(Func1<Observable<T>, Observable<R>> operation) {
37  		this.operation = operation;
38  	}
39  
40  	@Override
41  	public Subscriber<? super T> call(Subscriber<? super R> subscriber) {
42  		SingleSubscribeSubject<T> subject = SingleSubscribeSubject.create();
43  		Subscriber<T> result = Subscribers.from(subject);
44  		subscriber.add(result);
45  		operation.call(subject).unsafeSubscribe(subscriber);
46  		return result;
47  	}
48  
49  }
