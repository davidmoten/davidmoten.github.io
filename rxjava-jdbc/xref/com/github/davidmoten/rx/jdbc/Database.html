54      /**
55       * Constructor.
56       * 
57       * @param cp
58       *            provides connections
59       * @param selectHandler
60       *            handles select queries
61       * @param updateHandler
62       *            handles update queries
63       * @param nonTransactionalSchedulerFactory
64       *            schedules non transactional queries
65       * @param transactionalSchedulerFactory
66       *            schedules transactional queries
67       */
68      public Database(final ConnectionProvider cp, Func1<Observable<ResultSet>, Observable<ResultSet>> selectHandler,
69              Func1<Observable<Integer>, Observable<Integer>> updateHandler,
70              Func0<Scheduler> nonTransactionalSchedulerFactory, Func0<Scheduler> transactionalSchedulerFactory) {
71          Conditions.checkNotNull(cp);
72          this.cp = cp;
73          if (nonTransactionalSchedulerFactory != null)
74              this.nonTransactionalSchedulerFactory = nonTransactionalSchedulerFactory;
75          else
76              this.nonTransactionalSchedulerFactory = IO_SCHEDULER_FACTORY;
77          if (transactionalSchedulerFactory != null)
78              this.transactionalSchedulerFactory = transactionalSchedulerFactory;
79          else
80              this.transactionalSchedulerFactory = SINGLE_THREAD_POOL_SCHEDULER_FACTORY;
81          this.handlers = new Handlers(selectHandler, updateHandler);
82      }

364     /**
365      * Creates a {@link ScheduledExecutorService} based on a single thread pool.
366      * 
367      * @return executor service based on single thread pool.
368      */
369     private static ScheduledExecutorService createSingleThreadExecutor() {
370         return Executors.newScheduledThreadPool(1, new ThreadFactory() {
371             final AtomicInteger counter = new AtomicInteger();
372 
373             @Override
374             public Thread newThread(Runnable r) {
375                 Thread t = new Thread(r, "TransactionThreadPool-" + counter.incrementAndGet());
376                 t.setDaemon(true);
377                 return t;
378             }
379         });
380     }

382     /**
383      * Returns true if and only if integer is non-zero.
384      */
385     private static final Func1<Integer, Boolean> IS_NON_ZERO = new Func1<Integer, Boolean>() {
386         @Override
387         public Boolean call(Integer i) {
388             return i != 0;
389         }
390     };

392     /**
393      * Commits a transaction and resets the current query context so that
394      * further queries will use the asynchronous version by default. All
395      * Observable dependencies must be complete before commit is called.
396      * 
397      * @param depends
398      *            depdencies that must complete before commit occurs.
399      * @return
400      */
401     public Observable<Boolean> commit(Observable<?>... depends) {
402         return commitOrRollback(true, depends);
403     }
404 
405     /**
406      * Commits or rolls back a transaction depending on the <code>commit</code>
407      * parameter and resets the current query context so that further queries
408      * will use the asynchronous version by default. All Observable dependencies
409      * must be complete before commit/rollback is called.
410      * 
411      * @param commit
412      * @param depends
413      * @return
414      */
415     private Observable<Boolean> commitOrRollback(boolean commit, Observable<?>... depends) {
416         String action;
417         if (commit)
418             action = "commit";
419         else
420             action = "rollback";
421         QueryUpdate.Builder u = update(action);
422         for (Observable<?> dep : depends)
423             u = u.dependsOn(dep);
424         Observable<Boolean> result = u.count().map(IS_NON_ZERO);
425         lastTransactionResult.set(result);
426         resetQueryContext();
427         return result;
428     }
429 
430     /**
431      * Rolls back a transaction and resets the current query context so that
432      * further queries will use the asynchronous version by default. All
433      * Observable dependencies must be complete before rollback is called.
434      * 
435      * @param depends
436      *            depdencies that must complete before commit occurs.
437      * @return
438      * 
439      **/
440     public Observable<Boolean> rollback(Observable<?>... depends) {
441         return commitOrRollback(false, depends);
442     }
443 
444     /**
445      * Returns observable that emits true when last transaction committed or
446      * false when last transaction is rolled back.
447      * 
448      * @return
449      */
450     public Observable<Boolean> lastTransactionResult() {
451         Observable<Boolean> o = lastTransactionResult.get();
452         if (o == null)
453             return Observable.empty();
454         else
455             return o;
456     }

458     /**
459      * Revert query context to default asynchronous version.
460      */
461     private void resetQueryContext() {
462         context.set(null);
463     }
