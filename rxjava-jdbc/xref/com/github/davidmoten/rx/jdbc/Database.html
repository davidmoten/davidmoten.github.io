31  	 * Records the result of the last finished transaction (committed =
32  	 * <code>true</code> or rolled back = <code>false</code>).
33  	 */

42  	/**
43  	 * Constructor.
44  	 * 
45  	 * @param cp
46  	 *            provides connections
47  	 * @param threadPoolSize
48  	 *            for asynchronous query context
49  	 */
50  	public Database(final ConnectionProvider cp,
51  			Func1<Observable<ResultSet>, Observable<ResultSet>> selectHandler,
52  			Func1<Observable<Integer>, Observable<Integer>> updateHandler,
53  			Func0<Scheduler> nonTransactionalSchedulerFactory,
54  			Func0<Scheduler> transactionalSchedulerFactory) {
55  		Conditions.checkNotNull(cp);
56  		this.cp = cp;
57  		if (nonTransactionalSchedulerFactory != null)
58  			this.nonTransactionalSchedulerFactory = nonTransactionalSchedulerFactory;
59  		else
60  			this.nonTransactionalSchedulerFactory = COMPUTATION_SCHEDULER_FACTORY;
61  		if (transactionalSchedulerFactory != null)
62  			this.transactionalSchedulerFactory = transactionalSchedulerFactory;
63  		else
64  			this.transactionalSchedulerFactory = SINGLE_THREAD_POOL_SCHEDULER_FACTORY;
65  		this.handlers = new Handlers(selectHandler, updateHandler);
66  	}
67  
68  	private final Func0<Scheduler> COMPUTATION_SCHEDULER_FACTORY = new Func0<Scheduler>() {
69  		@Override
70  		public Scheduler call() {
71  			return Schedulers.computation();
72  		}
73  	};
74  
75  	private final Func0<Scheduler> SINGLE_THREAD_POOL_SCHEDULER_FACTORY = new Func0<Scheduler>() {
76  		@Override
77  		public Scheduler call() {
78  			return Schedulers.executor(createSingleThreadExecutor());
79  		}
80  	};
81  
82  	private static final Func0<Scheduler> CURRENT_THREAD_SCHEDULER_FACTORY = new Func0<Scheduler>() {
83  
84  		@Override
85  		public Scheduler call() {
86  			return Schedulers.trampoline();
87  		}
88  	};

90  	/**
91  	 * Constructor. Thread pool size defaults to
92  	 * <code>{@link Runtime}.getRuntime().availableProcessors()+1</code>. This
93  	 * may be too conservative if the database is on another server. If that is
94  	 * the case then you may want to use a thread pool size equal to the
95  	 * available processors + 1 on the database server.
96  	 * 
97  	 * @param cp
98  	 *            provides connections
99  	 */
100 	public Database(ConnectionProvider cp) {
101 		this(cp, Functions.<Observable<ResultSet>> identity(), Functions
102 				.<Observable<Integer>> identity(), null, null);

213 	 * Returns the thread local current query context (will not return null).
214 	 * 
215 	 * @return
216 	 */
217 	public QueryContext getQueryContext() {
218 		if (context.get() == null) {
219 			context.set(QueryContext.newNonTransactionalQueryContext(cp,
220 					handlers, nonTransactionalSchedulerFactory.call()));
221 		}
222 		return context.get();
223 	}

257 		QueryContext queryContext = QueryContext.newTransactionalQueryContext(

263 	/**
264 	 * Creates a {@link ScheduledExecutorService} based on a single thread pool.
265 	 * 
266 	 * @return executor service based on single thread pool.
267 	 */
268 	private static ScheduledExecutorService createSingleThreadExecutor() {
269 		return Executors.newScheduledThreadPool(1, new ThreadFactory() {
270 			final AtomicInteger counter = new AtomicInteger();
271 
272 			@Override
273 			public Thread newThread(Runnable r) {
274 				Thread t = new Thread(r, "TransactionThreadPool-"
275 						+ counter.incrementAndGet());
276 				t.setDaemon(true);
277 				return t;
278 			}
279 		});
280 	}

282 	/**
283 	 * Returns true if and only if integer is non-zero.
284 	 */
285 	private static final Func1<Integer, Boolean> IS_NON_ZERO = new Func1<Integer, Boolean>() {
286 		@Override
287 		public Boolean call(Integer i) {
288 			return i != 0;
289 		}
290 	};

292 	/**
293 	 * Commits a transaction and resets the current query context so that
294 	 * further queries will use the asynchronous version by default. All
295 	 * Observable dependencies must be complete before commit is called.
296 	 * 
297 	 * @param depends
298 	 *            depdencies that must complete before commit occurs.
299 	 * @return
300 	 */
301 	public Observable<Boolean> commit(Observable<?>... depends) {
302 		return commitOrRollback(true, depends);
303 	}
304 
305 	/**
306 	 * Commits or rolls back a transaction depending on the <code>commit</code>
307 	 * parameter and resets the current query context so that further queries
308 	 * will use the asynchronous version by default. All Observable dependencies
309 	 * must be complete before commit/rollback is called.
310 	 * 
311 	 * @param commit
312 	 * @param depends
313 	 * @return
314 	 */
315 	private Observable<Boolean> commitOrRollback(boolean commit,
316 			Observable<?>... depends) {
317 		String action;
318 		if (commit)
319 			action = "commit";
320 		else
321 			action = "rollback";
322 		QueryUpdate.Builder u = update(action);
323 		for (Observable<?> dep : depends)
324 			u = u.dependsOn(dep);
325 		Observable<Boolean> result = u.getCount().map(IS_NON_ZERO);
326 		lastTransactionResult.set(result);
327 		resetQueryContext();
328 		return result;
329 	}
330 
331 	/**
332 	 * Rolls back a transaction and resets the current query context so that
333 	 * further queries will use the asynchronous version by default. All
334 	 * Observable dependencies must be complete before rollback is called.
335 	 * 
336 	 * @param depends
337 	 *            depdencies that must complete before commit occurs.
338 	 * @return
339 	 * 
340 	 **/
341 	public Observable<Boolean> rollback(Observable<?>... depends) {
342 		return commitOrRollback(false, depends);
343 	}
344 
345 	/**
346 	 * Returns observable that emits true when last transaction committed or
347 	 * false when last transaction is rolled back.
348 	 * 
349 	 * @return
350 	 */
351 	public Observable<Boolean> lastTransactionResult() {
352 		Observable<Boolean> o = lastTransactionResult.get();
353 		if (o == null)
354 			return Observable.empty();
355 		else
356 			return o;
357 	}

359 	/**
360 	 * Revert query context to default asynchronous version.
361 	 */
362 	private void resetQueryContext() {
363 		context.set(null);
364 	}
