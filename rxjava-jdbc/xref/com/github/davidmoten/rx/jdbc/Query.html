1   package com.github.davidmoten.rx.jdbc;
2   
3   import rx.Observable;
4   import rx.Scheduler;
5   
6   /**
7    * A database DML query, either update/insert or select.
8    */
9   public interface Query {
10  
11      /**
12       * Returns the sql statement for this query following JDBC format (? for
13       * parameters for instance).
14       * 
15       * @return jdbc sql
16       */
17      String sql();
18  
19      /**
20       * Returns the parameters for the query in order of appearance as ? markers
21       * in the sql. May emit more than the number of parameters in one run of the
22       * query in which case the query would be run multiple times.
23       * 
24       * @return
25       */
26      Observable<Parameter> parameters();
27  
28      /**
29       * Returns the Observables that have to complete before this query is
30       * started.
31       * 
32       * @return
33       */
34      Observable<?> depends();
35  
36      /**
37       * Returns the query context including {@link ConnectionProvider} and
38       * {@link Scheduler}.
39       * 
40       * @return
41       */
42      QueryContext context();
43  
44  }
