65          return Database.from(Pools.nonBlocking() //

72      private static ConnectionProvider connectionProviderDerby(String url, boolean withStoredProcs) {
73          Connection c;
74          try {
75              c = DriverManager.getConnection(url);
76              createDatabaseDerby(c);
77              if (withStoredProcs) {
78                  addStoredProcs(c);
79              }
80          } catch (SQLException e) {
81              throw new SQLRuntimeException(e);
82          }
83          return new ConnectionProvider() {
84  
85              @Override
86              public Connection get() {
87                  try {
88                      return DriverManager.getConnection(url);
89                  } catch (SQLException e) {
90                      throw new SQLRuntimeException(e);
91                  }
92              }
93  
94              @Override
95              public void close() {
96                  //
97              }
98          };
99      }

289     public static Database create(int maxSize, boolean big, Scheduler scheduler) {
290         NonBlockingConnectionPool pool = Pools.nonBlocking() //
291                 .connectionProvider(connectionProvider(nextUrl(), big)) //
292                 .maxPoolSize(maxSize) //
293                 .scheduler(scheduler) //
294                 .build();
295         return Database.from(pool, () -> {
296             pool.close();
297             scheduler.shutdown();
298         });
299     }

309     private static ConnectionProvider connectionProvider(String url, boolean big) {
310         return new ConnectionProvider() {
311 
312             private final AtomicBoolean once = new AtomicBoolean(false);
313             private final CountDownLatch latch = new CountDownLatch(1);
314 
315             @Override
316             public Connection get() {
317                 try {
318                     if (once.compareAndSet(false, true)) {
319                         try {
320                             Connection c = DriverManager.getConnection(url);
321                             createDatabase(c, big);
322                             c.setAutoCommit(false);
323                             return c;
324                         } finally {
325                             latch.countDown();
326                         }
327                     } else {
328                         if (latch.await(30, TimeUnit.SECONDS)) {
329                             return DriverManager.getConnection(url);
330                         } else {
331                             throw new TimeoutException("big database timed out on creation");
332                         }
333                     }
334                 } catch (SQLException | InterruptedException | TimeoutException e) {
335                     throw new SQLRuntimeException(e);
336                 }
337             }
338 
339             @Override
340             public void close() {
341                 //
342             }
343         };
344     }

354     private static void createDatabase(Connection c, boolean big) {
355         try {
356             c.setAutoCommit(true);
357             c.prepareStatement(
358                     "create table person (name varchar(50) primary key, score int not null, date_of_birth date, registered timestamp)")
359                     .execute();
360             if (big) {
361                 List<String> lines = IOUtils.readLines(DatabaseCreator.class.getResourceAsStream("/big.txt"),
362                         StandardCharsets.UTF_8);
363                 lines.stream().map(line -> line.split("\t")).forEach(items -> {
364                     try {
365                         c.prepareStatement("insert into person(name,score) values('" + items[0] + "',"
366                                 + Integer.parseInt(items[1]) + ")").execute();
367                     } catch (SQLException e) {
368                         throw new SQLRuntimeException(e);
369                     }
370                 });
371             } else {
372                 exec(c, "insert into person(name,score,registered) values('FRED',21, {ts '2015-09-17 18:47:52.69Z'})");
373                 exec(c, "insert into person(name,score) values('JOSEPH',34)");
374                 exec(c, "insert into person(name,score) values('MARMADUKE',25)");
375             }
376 
377             exec(c, "create table person_clob (name varchar(50) not null,  document clob)");
378             exec(c, "create table person_blob (name varchar(50) not null, document blob)");
379 
380             exec(c, "create table address (address_id int primary key, full_address varchar(255) not null)");
381             exec(c, "insert into address(address_id, full_address) values(1,'57 Something St, El Barrio, Big Place')");
382             exec(c, "create table note(id bigint auto_increment primary key, text varchar(255))");
383         } catch (SQLException e) {
384             throw new SQLRuntimeException(e);
385         } catch (IOException e) {
386             throw new RuntimeException(e);
387         }
388     }
