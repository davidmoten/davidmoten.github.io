17      public void testFromQueueName() {

19          AmazonSQSClient sqs = Mockito.mock(AmazonSQSClient.class);
20          String queueUrl = "https://myqueue";
21          Mockito.when(sqs.getQueueUrl(Mockito.any(GetQueueUrlRequest.class)))
22                  .thenReturn(new GetQueueUrlResult().withQueueUrl(queueUrl));
