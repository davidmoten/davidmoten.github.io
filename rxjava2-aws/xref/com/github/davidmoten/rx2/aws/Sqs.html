72      public static String sendToQueueUsingS3(AmazonSQS sqs, String queueUrl, AmazonS3 s3, String bucketName,
73              Map<String, String> headers, byte[] message, Callable<String> s3IdFactory) {
74          Preconditions.checkNotNull(sqs);
75          Preconditions.checkNotNull(s3);
76          Preconditions.checkNotNull(queueUrl);
77          Preconditions.checkNotNull(bucketName);
78          Preconditions.checkNotNull(message);
79          String s3Id = uncheckedCall(s3IdFactory);
80          final ObjectMetadata metadata = new ObjectMetadata();
81          metadata.setContentLength(message.length);
82          headers //
83                  .entrySet() //
84                  .forEach(h -> metadata.setHeader(h.getKey(), h.getValue()));
85          s3.putObject(bucketName, s3Id, new ByteArrayInputStream(message), metadata);
86          try {
87              sqs.sendMessage(queueUrl, s3Id);
88          } catch (final RuntimeException e) {
89              try {
90                  s3.deleteObject(bucketName, s3Id);
91                  throw e;
92              } catch (final RuntimeException e2) {
93                  throw new io.reactivex.exceptions.CompositeException(e, e2);
94              }
95          }
96          return s3Id;
97      }

194     static Flowable<SqsMessage> messages(Callable<AmazonSQS> sqsFactory, Optional<Callable<AmazonS3>> s3Factory,
195             SqsQueue queue, Optional<String> bucketName, Optional<Flowable<Integer>> waitTimesSeconds,
196             Consumer<? super String> logger, Runnable prePoll, Consumer<? super Optional<Throwable>> postPoll) {
197         Preconditions.checkNotNull(sqsFactory);
198         Preconditions.checkNotNull(s3Factory);
199         Preconditions.checkNotNull(queue);
200         Preconditions.checkNotNull(bucketName);
201         Preconditions.checkNotNull(waitTimesSeconds);
202         Preconditions.checkNotNull(logger);
203         Preconditions.checkNotNull(prePoll);
204         Preconditions.checkNotNull(postPoll);
205         return Flowable.using(sqsFactory,
206                 sqs -> createFlowableWithSqs(sqs, s3Factory, sqsFactory, queue, bucketName,
207                         waitTimesSeconds, logger, prePoll, postPoll),
208                 sqs -> sqs.shutdown());
209     }

211     private static Flowable<SqsMessage> createFlowableWithSqs(AmazonSQS sqs, Optional<Callable<AmazonS3>> s3Factory,
212             Callable<AmazonSQS> sqsFactory, SqsQueue queue, Optional<String> bucketName,
213             Optional<Flowable<Integer>> waitTimesSeconds, Consumer<? super String> logger,
214             Runnable prePoll, Consumer<? super Optional<Throwable>> postPoll) {
215 
216         return Flowable.using(() -> s3Factory.map(x -> uncheckedCall(x)), //
217                 s3 -> createFlowableWithS3(sqs, s3Factory, sqsFactory, queue, bucketName, s3,
218                         waitTimesSeconds, logger, prePoll, postPoll),
219                 s3 -> s3.ifPresent(Util::shutdown));
220     }

222     private static Flowable<SqsMessage> createFlowableWithS3(AmazonSQS sqs, Optional<Callable<AmazonS3>> s3Factory,
223             Callable<AmazonSQS> sqsFactory, SqsQueue queue, Optional<String> bucketName, Optional<AmazonS3> s3,
224             Optional<Flowable<Integer>> waitTimesSeconds, Consumer<? super String> logger, Runnable prePoll,
225             Consumer<? super Optional<Throwable>> postPoll) {
226         final Service service = new Service(s3Factory, sqsFactory, s3, sqs, queue.getQueueUrl(sqs), bucketName);
227         if (waitTimesSeconds.isPresent()) {
228             return createFlowablePolling(waitTimesSeconds.get(), service, logger, prePoll, postPoll);
229         } else {
230             return createFlowableContinousLongPolling(service, logger, prePoll, postPoll);
231         }
232     }

234     private static Flowable<SqsMessage> createFlowablePolling(Flowable<Integer> waitTimesSeconds, Service service,
235             Consumer<? super String> logger, Runnable prePoll, Consumer<? super Optional<Throwable>> postPoll) {
236         return waitTimesSeconds.flatMap(n -> get(n, service, logger, prePoll, postPoll), 1);

239     private static Flowable<SqsMessage> get(int waitTimeSeconds, Service service,
240             Consumer<? super String> logger, Runnable prePoll,
241             Consumer<? super Optional<Throwable>> postPoll) {
242         return Flowable.fromCallable(() -> {
243             logger.accept("polling for messages");
244             return sqsMessages(service, waitTimeSeconds, prePoll, postPoll);
245         }) //
246                 .concatWith(//
247                         Flowable.fromCallable(() -> sqsMessages(service, 0, prePoll, postPoll)) //
248                                 .repeat()) //
249                 .takeWhile(x -> !x.isEmpty()) //
250                 .flatMapIterable(x -> x) //
251                 .filter(Optional::isPresent) //
252                 .map(Optional::get);
253     }

262     static List<Message> messages(Supplier<? extends List<Message>> supplier, Runnable prePoll,
263             Consumer<? super Optional<Throwable>> postPoll) {
264         prePoll.run();
265         List<Message> list;
266         try {
267             list = supplier.get();
268         } catch (Throwable t) {
269             postPoll.accept(Optional.of(t));
270             throw t;
271         }
272         postPoll.accept(Optional.empty());
273         return list;
274     }

276     private static List<Optional<SqsMessage>> sqsMessages(Service service, int waitTimeSeconds,
277             Runnable prePoll, Consumer<? super Optional<Throwable>> postPoll) {
278         return messages(service, waitTimeSeconds, prePoll, postPoll) //
279                 .stream() //
280                 .map(m -> Sqs.getNextMessage(m, service)) //
281                 .collect(Collectors.toList());
282     }

285             Consumer<? super String> logger,
286             Runnable prePoll,

293     private static final class ContinuousLongPollingSyncOnSubscribe
294             implements Callable<State>, BiConsumer<State, Emitter<SqsMessage>> {
295 
296         private final Service service;
297 
298         private ReceiveMessageRequest request;
299         private Consumer<? super String> logger;
300         private Runnable prePoll;
301         private Consumer<? super Optional<Throwable>> postPoll;
302 
303         ContinuousLongPollingSyncOnSubscribe(Service service, Consumer<? super String> logger, Runnable prePoll,
304                                                     Consumer<? super Optional<Throwable>> postPoll) {
305             this.service = service;
306             this.logger = logger;
307             this.prePoll = prePoll;
308             this.postPoll = postPoll;
309         }
310 
311         @Override
312         public State call() {
313             request = new ReceiveMessageRequest(service.queueUrl) //
314                     .withWaitTimeSeconds(20) //
315                     .withMaxNumberOfMessages(10);
316             return new State(new LinkedList<>());
317         }
318 
319         @Override
320         public void accept(State state, Emitter<SqsMessage> emitter) throws Exception {
321             final Queue<Message> q = state.queue;
322             Optional<SqsMessage> next = Optional.empty();
323             while (!next.isPresent()) {
324                 while (q.isEmpty()) {
325                     logger.accept("long polling for messages on queue=" + service.queueUrl);
326                     List<Message> list = messages(//
327                             () -> service.sqs.receiveMessage(request).getMessages(), //
328                             prePoll, // 
329                             postPoll);
330                     q.addAll(list);
331                 }
332                 final Message message = q.poll();
333                 next = getNextMessage(message, service);
334             }
335             emitter.onNext(next.get());
336         }
337     }
338 
339     static Optional<SqsMessage> getNextMessage(Message message, Service service) {
340         if (service.bucketName.isPresent()) {
341             final String s3Id = message.getBody();
342             if (!service.s3.get().doesObjectExist(service.bucketName.get(), s3Id)) {
343                 service.sqs.deleteMessage(service.queueUrl, message.getReceiptHandle());
344                 return Optional.empty();
345             } else {
346                 final S3Object object = service.s3.get().getObject(service.bucketName.get(), s3Id);
347                 final byte[] content = readAndClose(object.getObjectContent());
348                 final long timestamp = object.getObjectMetadata().getLastModified().getTime();
349                 final SqsMessage mb = new SqsMessage(message.getReceiptHandle(), content, timestamp, Optional.of(s3Id),
350                         service);
351                 return Optional.of(mb);
352             }
353         } else {
354             final SqsMessage mb = new SqsMessage(message.getReceiptHandle(),
355                     message.getBody().getBytes(StandardCharsets.UTF_8), System.currentTimeMillis(), Optional.empty(),
356                     service);
357             return Optional.of(mb);
358         }
359     }

375     @VisibleForTesting
376     static byte[] readAndClose(InputStream is) {
377         Preconditions.checkNotNull(is);
378         try {
379             final ByteArrayOutputStream bos = new ByteArrayOutputStream();
380             final byte[] bytes = new byte[8192];
381             int n;
382             while ((n = is.read(bytes)) != -1) {
383                 bos.write(bytes, 0, n);
384             }
385             return bos.toByteArray();
386         } catch (final IOException e) {
387             throw new RuntimeException(e);
388         }
