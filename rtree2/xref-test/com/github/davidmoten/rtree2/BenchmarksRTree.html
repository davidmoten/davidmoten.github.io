275     private long search(RTree<?, ?> tree, Rectangle r) {
276         return Iterables.size(tree.search(r));
277     }

279     private long searchGreek(RTree<Object, Point> tree) {

284     private static Rectangle searchRectangle() {
285         final Rectangle r;
286         if (precision == Precision.DOUBLE) {
287             r = DEFAULT_GREEK_RECTANGLE;
288         } else {
289             r = Geometries.rectangle(40f, 27.0f, 40.5f, 27.5f);
290         }
291         return r;
292     }
293 
294     private long searchNearestGreek(RTree<Object, Point> tree) {
295         final Point p;
296         if (precision == Precision.DOUBLE) {
297             p = Geometries.point(40.0, 27.0);
298         } else {
299             p = Geometries.point(40.0f, 27.0f);
300         }
301         return Iterables.size(tree.nearest(p, 1, 300));
302     }

304     private long searchGreekWithBackpressure(RTree<Object, Point> tree, final Blackhole bh) {
305         // should return 22 results

313     private RTree<Object, Point> insertPoint(RTree<Object, Point> tree) {
314         if (precision == Precision.DOUBLE) {
315             return tree.add(new Object(), Geometries.point(Math.random() * 1000, Math.random() * 1000));
316         } else {
317             return tree.add(new Object(), Geometries.point((float) Math.random() * 1000, (float) Math.random() * 1000));
318         }
319     }

321     static boolean longRun = true;
322     
323     public static void main(String[] args) {
324         RTree<Object, Point> tree = RTree.maxChildren(28).star().<Object, Point>create(entries);
325         Rectangle r = searchRectangle();
326         if (longRun) {
327             while (true) {
328                 if (Iterables.size(tree.search(r)) == 0) {
329                     System.out.println("unexpected");
330                 }
331             }
332         } else {
333             
334             long t = System.currentTimeMillis();
335             long warmupTimeSeconds = 10;
336             long benchmarkTimeSeconds = 10;
337             long t2 = -1;
338             long count = 0;
339             while (true) {
340                 if (Iterables.size(tree.search(r))== 0) {
341                     System.out.println("zero!!");
342                 };
343                 count++;
344                 if (count % 10000 == 0) {
345                     if (t2 == -1) {
346                         if (System.currentTimeMillis() - t > TimeUnit.SECONDS.toMillis(warmupTimeSeconds)) {
347                             t2 = System.currentTimeMillis();
348                         }
349                     } else if (System.currentTimeMillis() - t2 > TimeUnit.SECONDS.toMillis(benchmarkTimeSeconds)) {
350                         break;
351                     }
352                 }
353             }
354             double ratePerSecond = count * 1000.0 / (System.currentTimeMillis() - t2);
355             DecimalFormat df = new DecimalFormat("0.000");
356             System.out.println("ratePerSecond=" + df.format(ratePerSecond / 1000000.0) + "m");
357         }
358     }
