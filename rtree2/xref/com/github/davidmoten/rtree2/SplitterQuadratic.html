21      @SuppressWarnings("unchecked")
22      @Override
23      public <T extends HasGeometry> ListPair<T> split(List<T> items, int minSize) {
24          Preconditions.checkArgument(items.size() >= 2);
25  
26          // according to
27          // http://en.wikipedia.org/wiki/R-tree#Splitting_an_overflowing_node
28  
29          // find the worst combination pairwise in the list and use them to start
30          // the two groups
31          final Pair<T> worstCombination = worstCombination(items);
32  
33          // worst combination to have in the same node is now e1,e2.
34  
35          // establish a group around e1 and another group around e2
36          final List<T> group1 = Lists.newArrayList(worstCombination.value1());
37          final List<T> group2 = Lists.newArrayList(worstCombination.value2());
38  
39          final List<T> remaining = new ArrayList<T>(items);
40          remaining.remove(worstCombination.value1());
41          remaining.remove(worstCombination.value2());
42  
43          final int minGroupSize = items.size() / 2;
44  
45          // now add the remainder to the groups using least mbr area increase
46          // except in the case where minimumSize would be contradicted
47          while (remaining.size() > 0) {
48              assignRemaining(group1, group2, remaining, minGroupSize);
49          }
50          return new ListPair<T>(group1, group2);
51      }
52  
53      private <T extends HasGeometry> void assignRemaining(final List<T> group1, final List<T> group2,
54              final List<T> remaining, final int minGroupSize) {
55          final Rectangle mbr1 = Util.mbr(group1);
56          final Rectangle mbr2 = Util.mbr(group2);
57          final T item1 = getBestCandidateForGroup(remaining, group1, mbr1);
58          final T item2 = getBestCandidateForGroup(remaining, group2, mbr2);
59          final boolean area1LessThanArea2 = item1.geometry().mbr().add(mbr1).area() <= item2.geometry().mbr().add(mbr2)
60                  .area();
61  
62          if (area1LessThanArea2 && (group2.size() + remaining.size() - 1 >= minGroupSize)
63                  || !area1LessThanArea2 && (group1.size() + remaining.size() == minGroupSize)) {
64              group1.add(item1);
65              remaining.remove(item1);
66          } else {
67              group2.add(item2);
68              remaining.remove(item2);
69          }
70      }
71  
72      @VisibleForTesting
73      static <T extends HasGeometry> T getBestCandidateForGroup(List<T> list, List<T> group, Rectangle groupMbr) {
74          // TODO reduce allocations by not using Optional
75          Optional<T> minEntry = empty();
76          Optional<Double> minArea = empty();
77          for (final T entry : list) {
78              final double area = groupMbr.add(entry.geometry().mbr()).area();
79              if (!minArea.isPresent() || area < minArea.get()) {
80                  minArea = of(area);
81                  minEntry = of(entry);
82              }
83          }
84          return minEntry.get();
85      }
86  
87      @VisibleForTesting
88      static <T extends HasGeometry> Pair<T> worstCombination(List<T> items) {
89          // TODO reduce allocations by not using Optional
90          Optional<T> e1 = empty();
91          Optional<T> e2 = empty();
92          {
93              Optional<Double> maxArea = empty();
94              for (int i = 0; i < items.size(); i++) {
95                  for (int j = i + 1; j < items.size(); j++) {
96                      T entry1 = items.get(i);
97                      T entry2 = items.get(j);
98                      final double area = entry1.geometry().mbr().add(entry2.geometry().mbr()).area();
99                      if (!maxArea.isPresent() || area > maxArea.get()) {
100                         e1 = of(entry1);
101                         e2 = of(entry2);
102                         maxArea = of(area);
103                     }
104                 }
105             }
106         }
107         if (e1.isPresent())
108             return new Pair<T>(e1.get(), e2.get());
109         else
110             // all items are the same item
111             return new Pair<T>(items.get(0), items.get(1));
112     }
